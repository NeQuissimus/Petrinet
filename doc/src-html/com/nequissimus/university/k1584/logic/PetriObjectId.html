001package com.nequissimus.university.k1584.logic;
002
003import java.util.HashSet;
004import java.util.Set;
005import java.util.UUID;
006
007/**
008 * Class that generates a unique ID for each Petri object.
009 * @author Tim Steinbach
010 */
011public final class PetriObjectId {
012
013    /**
014     * All ids that have been used.
015     */
016    private static final Set<String> USED_IDS = new HashSet<String>();
017
018    /**
019     * Hide constructor.
020     */
021    private PetriObjectId() {
022    }
023
024    /**
025     * Add an id to the list of used ones.
026     * @param id Used id
027     */
028    public static void addUsedId(final String id) {
029
030        PetriObjectId.USED_IDS.add(id);
031
032    }
033
034    /**
035     * Return the next available id.
036     * @return Unique id
037     */
038    public static String getId() {
039
040        UUID uid = UUID.randomUUID();
041        String id = uid.toString();
042
043        while (PetriObjectId.USED_IDS.contains(id)) {
044
045            uid = UUID.randomUUID();
046            id = uid.toString();
047
048        }
049
050        PetriObjectId.USED_IDS.add(id);
051
052        return id;
053    }
054
055    /**
056     * Empty the list of used ids.
057     */
058    public static void resetUsedIds() {
059
060        PetriObjectId.USED_IDS.clear();
061
062    }
063
064}
