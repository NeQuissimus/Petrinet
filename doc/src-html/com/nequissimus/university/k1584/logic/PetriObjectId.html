001package com.nequissimus.university.k1584.logic;
002
003/**
004 * Class that generates a unique ID for each Petri object.
005 * @author Tim Steinbach
006 */
007final class PetriObjectId {
008
009    /**
010     * Id counter.
011     */
012    private static long nextId = 0;
013
014    /**
015     * Hide constructor.
016     */
017    private PetriObjectId() {
018    }
019
020    /**
021     * Return the next available id.
022     * @return Unique id
023     */
024    public static String getId() {
025        return Long.toHexString(PetriObjectId.nextId++);
026    }
027
028}
