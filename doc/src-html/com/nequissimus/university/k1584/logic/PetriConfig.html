001package com.nequissimus.university.k1584.logic;
002
003import java.io.FileInputStream;
004import java.io.IOException;
005import java.io.InputStream;
006import java.util.Properties;
007
008/**
009 * Load the properties file into this configuration object.<br />
010 * This class has default values, in case the configuration file cannot be loaded<br /><br />
011 * Default values:<br />
012 * PlaceName = "" // Initial name for a place<br />
013 * TransitionName = "" // Initial name for a transition<br />
014 * CanvasWidth = 1000 // Width for the editing canvas<br />
015 * CanvasHeight = 1000 // Height for the editing canvas<br />
016 * CanvasWindowWidth = 300 // Width for the visible part of the canvas<br />
017 * CanvasWindowHeight = 300 // Height for the visible part of the canvas<br />
018 * WindowTitle = "Tim Schram - q7485417" // Window title<br />
019 * WindowWidth = 500 // Window width<br />
020 * WindowHeight = 400 // Window height<br />
021 * WindowX = 100 // Initial X coordinate for window<br />
022 * WindowY = 100 // Initial Y coordinate for window<br />
023 * ApplicationName = "Petrinet" // Application name<br />
024 * @author Tim Schram
025 *
026 */
027public class PetriConfig extends Properties {
028
029    private static final long serialVersionUID = -6320268661029188694L;
030
031    public static final String PLACE_NAME = "PlaceName";
032    public static final String TRANSITION_NAME = "TransitionName";
033    public static final String CANVAS_WIDTH = "CanvasWidth";
034    public static final String CANVAS_HEIGHT = "CanvasHeight";
035    public static final String CANVAS_WINDOW_WIDTH = "CanvasWindowWidth";
036    public static final String CANVAS_WINDOW_HEIGHT = "CanvasWindowHeight";
037    public static final String WINDOW_TITLE = "WindowTitle";
038    public static final String WINDOW_WIDTH = "WindowWidth";
039    public static final String WINDOW_HEIGHT = "WindowHeight";
040    public static final String WINDOW_X = "WindowX";
041    public static final String WINDOW_Y = "WindowY";
042    public static final String APPLICATION_NAME = "ApplicationName";
043
044    private static PetriConfig config = null;
045    
046    private PetriConfig() {
047        
048        super();
049        
050        this.readProperties();
051        
052    }
053    
054    /**
055     * Instead of using the constructor, this method is used to make this a singleton.
056     * This is to make sure the properties file is only read once.
057     * @return Configuration instance
058     */
059    public static PetriConfig getInstance() {
060        
061        if (config == null)
062            config = new PetriConfig();
063        
064        return config;
065        
066    }
067    
068    /**
069     * Fill the configuration object with default values.
070     */
071    private void getDefaults() {        
072
073        this.put(PLACE_NAME, "");
074        this.put(TRANSITION_NAME, "");
075        this.put(CANVAS_HEIGHT, 1000);
076        this.put(CANVAS_WIDTH, 1000);
077        this.put(CANVAS_WINDOW_HEIGHT, 300);
078        this.put(CANVAS_WINDOW_WIDTH, 300);
079        this.put(WINDOW_TITLE, "Tim Schram - q7485417");
080        this.put(WINDOW_WIDTH, 500);
081        this.put(WINDOW_HEIGHT, 400);
082        this.put(WINDOW_X, 100);
083        this.put(WINDOW_Y, 100);
084        this.put(APPLICATION_NAME, "Petrinet");
085        
086    }
087    
088    /**
089     * Read the properties file 'config.properties'
090     * If reading that file fails, a default configuration will be used
091     */
092    private void readProperties() {
093
094        InputStream is = null;
095        
096        try {
097            
098            is = new FileInputStream("config.properties");
099            this.load(is);
100            
101        } catch(final IOException e) {
102            
103            // If something went wrong with reading the properties file, the default values
104            // will still be in the needed in the configuration object
105            this.getDefaults();
106            
107        } finally {
108            
109            if(null != is) 
110                try {is.close();} catch(final IOException e) {}
111            
112        }       
113
114    }
115    
116    /**
117     * Return a property value as String
118     * @param key Key to look for
119     * @return Value as String
120     */
121    public String get(final String key) {
122        
123        return String.valueOf(super.get(key));
124        
125    }
126    
127}
