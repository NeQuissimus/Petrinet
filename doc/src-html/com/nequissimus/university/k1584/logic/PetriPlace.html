001package com.nequissimus.university.k1584.logic;
002
003import java.awt.Dimension;
004import java.awt.Point;
005
006/**
007 * Place class for a Petri net.
008 * @author Tim Steinbach
009 */
010public class PetriPlace extends PetriObject {
011
012    /**
013     * Number of markings set.
014     */
015    private int markings = 0;
016
017    /**
018     * Create a new logical place with a given name.
019     * @param name Name
020     */
021    PetriPlace(final String name) {
022
023        super(name);
024
025    }
026
027    /**
028     * Create a new logical place with a given name.
029     * @param name Name
030     * @param id Id
031     */
032    PetriPlace(final String name, final String id) {
033
034        super(name, id);
035
036    }
037
038    @Override
039    public final String toString() {
040        return "PetriPlace [" + this.getId() + "]";
041    }
042
043    @Override
044    protected final PetriPlace clone() {
045
046        final PetriPlace clone =
047            new PetriPlace(this.getName(), PetriObjectId.getId());
048
049        clone.setMarkings(this.getMarkings());
050        clone.setPosition(new Point(this.getPosition()));
051        clone.setSize(new Dimension(this.getSize()));
052
053        return clone;
054
055    }
056
057    /**
058     * Decrease the value of markings by one.
059     */
060    final void decreaseMarkings() {
061
062        if (this.markings > 0) {
063            this.markings--;
064        }
065
066    }
067
068    /**
069     * Get the number of markings set for this place.
070     * @return Number of markings
071     */
072    final int getMarkings() {
073        return this.markings;
074    }
075
076    /**
077     * Increase the value of markings by one.
078     */
079    final void increaseMarkings() {
080
081        if (this.markings < Integer.MAX_VALUE) {
082            this.markings++;
083        }
084
085    }
086
087    /**
088     * Set the value of markings.
089     * @param value If value < 0, value = 0 will be assumed
090     */
091    final void setMarkings(final int value) {
092
093        int tmpValue = value;
094
095        if (tmpValue < 0) {
096            tmpValue = 0;
097        }
098
099        this.markings = tmpValue;
100
101    }
102
103}
