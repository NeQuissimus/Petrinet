032package com.nequissimus.university.k1584.logic;
033
034import java.util.ArrayList;
035import java.util.List;
036
037import com.nequissimus.library.data.Singleton;
038
039/**
040 * Container that holds all snapshots of Petri nets as well as the current one.
041 * @author Tim Steinbach
042 */
043public class PetriSnapshots {
044
045    /**
046     * Configuration.
047     */
048    private static PetriConfig config;
049
050    /**
051     * Currently active Petri net.
052     */
053    private PetriNet current = null;
054
055    /**
056     * Petri nets.
057     */
058    private final List<PetriNet> nets = new ArrayList<PetriNet>();
059
060    /**
061     * Create a new snapshots object.
062     */
063    public PetriSnapshots() {
064
065        PetriSnapshots.config = Singleton.getObject(PetriConfig.class);
066
067    }
068
069    /**
070     * Create a new snapshots object and inject a custom configuration.
071     * @param config Custom configuration
072     */
073    PetriSnapshots(final PetriConfig config) {
074
075        PetriSnapshots.config = config;
076
077    }
078
079    /**
080     * Add a new Petri net to the list of snapshots.
081     * @param net Net to be added
082     */
083    public final void add(final PetriNet net) {
084
085        this.nets.add(net);
086
087    }
088
089    /**
090     * Create a new Petri net and add it to the snapshots.
091     * @param name Name for new net
092     * @return New Petri net
093     */
094    public final PetriNet add(final String name) {
095
096        final PetriNet net = new PetriNet(name);
097        this.nets.add(net);
098
099        return net;
100
101    }
102
103    /**
104     * Delete a logical net from the snapshots.
105     * @param net Net to be removed
106     */
107    public final void delete(final PetriNet net) {
108
109        this.nets.remove(net);
110
111        if (net == this.current) {
112
113            this.current = null;
114            this.current = this.getCurrent();
115
116        }
117
118    }
119
120    /**
121     * Find a logical net by its name.
122     * @param name Name
123     * @return Logical net with net. NULL if no such net could be found
124     */
125    public final PetriNet getByName(final String name) {
126
127        for (final PetriNet net : this.nets) {
128
129            if (net.getName().equals(name)) {
130                return net;
131            }
132
133        }
134
135        return null;
136
137    }
138
139    /**
140     * Get the currently active logical net.
141     * @return Currently active net
142     */
143    public final PetriNet getCurrent() {
144
145        if (this.nets.isEmpty()) {
146
147            this.current = this.add(PetriSnapshots.config.getNetName());
148
149        }
150
151        if ((null == this.current) && (this.nets.size() > 0)) {
152
153            this.current = this.nets.get(this.nets.size() - 1);
154
155        }
156
157        return this.current;
158
159    }
160
161    /**
162     * Get the list of nets.
163     * @return All snapshots
164     */
165    public final List<PetriNet> getNets() {
166
167        return this.nets;
168
169    }
170
171    /**
172     * Rename a Petri net.
173     * @param net Net to be renamed
174     * @param newName New name
175     */
176    public final void rename(final PetriNet net, final String newName) {
177
178        net.setName(newName);
179
180    }
181
182    /**
183     * Set a new currently active net.
184     * @param net Active net
185     */
186    public final void setCurrent(final PetriNet net) {
187
188        if (null != net) {
189            this.current = net;
190        }
191
192    }
193
194}
