172    @Override
173    public final PetriNet clone() {
174
175        try {
176
177            super.clone();
178
179        } catch (final CloneNotSupportedException e) {
180
181            return null;
182
183        }
184
185        final PetriNet result = new PetriNet(this.name);
186
187        final Set<PetriPlace> newPlaces = new HashSet<PetriPlace>();
188        final Set<PetriTransition> newTransitions =
189            new HashSet<PetriTransition>();
190
191        // Use old ids with new places to map when creating the transitions
192        final Map<String, PetriPlace> oldIdWithNewPlace =
193            new HashMap<String, PetriPlace>();
194
195        for (final PetriPlace petriPlace : this.places) {
196
197            final PetriPlace newPlace = petriPlace.clone();
198
199            oldIdWithNewPlace.put(petriPlace.getId(), newPlace);
200
201            newPlaces.add(newPlace);
202
203        }
204
205        for (final PetriTransition petriTransition : this.transitions) {
206
207            final PetriTransition transition = petriTransition.clone();
208
209            final Set<PetriPlace> input = petriTransition.getInput();
210            final Set<PetriPlace> output = petriTransition.getOutput();
211
212            for (final PetriPlace petriPlace : input) {
213
214                final String oldId = petriPlace.getId();
215
216                final PetriPlace newPlace = oldIdWithNewPlace.get(oldId);
217                transition.addInput(newPlace);
218
219            }
220
221            for (final PetriPlace petriPlace : output) {
222
223                final String oldId = petriPlace.getId();
224
225                final PetriPlace newPlace = oldIdWithNewPlace.get(oldId);
226                transition.addOutput(newPlace);
227
228            }
229
230            newTransitions.add(transition);
231
232        }
233
234        result.addPlaces(newPlaces);
235        result.addTransitions(newTransitions);
236
237        return result;
238
239    }

516    /**
517     * Remove a Petri object. This method decides whether the object is a place
518     * or a transition and then calls the correct method for the respective
519     * sub-type.
520     * @param object Petri object
521     */
522    public final void remove(final PetriObject object) {
523
524        if (object instanceof PetriPlace) {
525            this.remove((PetriPlace) object);
526        } else if (object instanceof PetriTransition) {
527            this.remove((PetriTransition) object);
528        }
529
530    }

532    /**
533     * Remove a place and all edges touching the object.
534     * @param place Place to be removed
535     */
536    public final void remove(final PetriPlace place) {
537
538        for (final PetriTransition transition : this.transitions) {
539
540            transition.removeInput(place);
541            transition.removeOutput(place);
542
543        }
544
545        this.places.remove(place);
546        this.removeFromMarkings(place);
547
548    }

651    /**

653     * All places for the net will be set to the correct number of tokens.
654     * @param marking Marking to switch to
655     */
656    public final void switchMarking(final PetriMarking marking) {
657
658        this.currentMarking = marking;
659
660        for (final PetriPlace place : this.places) {
661
662            place.setTokens(marking.getTokens(place));

664        }
