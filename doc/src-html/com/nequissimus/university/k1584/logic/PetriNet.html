001package com.nequissimus.university.k1584.logic;
002
003import java.awt.Dimension;
004import java.awt.Point;
005import java.util.HashSet;
006import java.util.Properties;
007import java.util.Set;
008
009/**
010 * Main Petri net class that holds all information and methods about the net.
011 * This is basically the delegate API of the Petri net logic, no other classes
012 * need to be accessed to work with the net. Since the methods to work with the
013 * objects are not public, this class provides all necessary access to the net's
014 * as well as each component's behaviour.
015 * @author Tim Schram
016 */
017public class PetriNet {
018
019    /**
020     * All logical places for this net.
021     */
022    private final Set<PetriPlace> places;
023
024    /**
025     * All logical transitions for this net.
026     */
027    private final Set<PetriTransition> transitions;
028
029    /**
030     * Configuration.
031     */
032    private final Properties config;
033
034    /**
035     * This net's name.
036     */
037    private String name;
038
039    /**
040     * Create a new net with the given name.
041     * @param name Name
042     */
043    public PetriNet(final String name) {
044
045        this.places = new HashSet<PetriPlace>();
046        this.transitions = new HashSet<PetriTransition>();
047        this.config = PetriConfig.getInstance();
048
049        this.name = name;
050
051    }
052
053    /**
054     * Get all places for this net.
055     * @return Places
056     */
057    public final Set<PetriPlace> getPlaces() {
058        return this.places;
059    }
060
061    /**
062     * Get all transitions for this net.
063     * @return Transitions
064     */
065    public final Set<PetriTransition> getTransitions() {
066        return this.transitions;
067    }
068
069    /**
070     * Get this net's name.
071     * @return Name
072     */
073    public final String getName() {
074        return this.name;
075    }
076
077    /**
078     * Set this net's name.
079     * @param newName Name
080     */
081    final void setName(final String newName) {
082        this.name = newName;
083    }
084
085    /**
086     * Remove a place and all edges touching the object.
087     * @param place Place to be removed
088     */
089    public final void remove(final PetriPlace place) {
090
091        for (final PetriTransition transition : this.transitions) {
092
093            transition.removeInput(place);
094            transition.removeOutput(place);
095
096        }
097
098        this.places.remove(place);
099
100    }
101
102    /**
103     * Remove a transition and all edges touching the object.
104     * @param transition Transition to be removed
105     */
106    public final void remove(final PetriTransition transition) {
107
108        this.transitions.remove(transition);
109
110    }
111
112    /**
113     * Add a new place to the net.
114     * @return Returns the newly created place
115     */
116    public final PetriPlace addPlace() {
117
118        final PetriPlace newPlace =
119            new PetriPlace(this.config.getProperty(PetriConfig.PLACE_NAME));
120        this.places.add(newPlace);
121
122        return newPlace;
123
124    }
125
126    /**
127     * Add a new transition to the net.
128     * @return Return the newly created transition
129     */
130    public final PetriTransition addTransition() {
131
132        final PetriTransition newTransition =
133            new PetriTransition(
134                this.config.getProperty(PetriConfig.TRANSITION_NAME));
135        this.transitions.add(newTransition);
136
137        return newTransition;
138
139    }
140
141    /**
142     * Connect a place to a transition (place -> transition).
143     * @param from Connect from this place
144     * @param to Connect to this transition
145     */
146    public final void connect(final PetriPlace from,
147        final PetriTransition to) {
148
149        to.addInput(from);
150
151    }
152
153    /**
154     * Connect a transition to a place (transition -> place).
155     * @param from Connect from this transition
156     * @param to Connect to this place
157     */
158    public final void connect(final PetriTransition from,
159        final PetriPlace to) {
160
161        from.addOutput(to);
162
163    }
164
165    /**
166     * Make a transition occur (Take one marking away from each input place and
167     * give one to each output place) if the transition is active. Nothing
168     * happens if the transition is not active.
169     * @param transition Transition to occur
170     */
171    public final void occur(final PetriTransition transition) {
172
173        transition.occur();
174
175    }
176
177    /**
178     * Check whether a transition is active. A transition is active if each of
179     * its input places has at least one marking.
180     * @param transition Transition to be checked
181     * @return Whether the transition is active
182     */
183    public final boolean isActive(final PetriTransition transition) {
184
185        return transition.isActive();
186
187    }
188
189    /**
190     * Increase the number of markings for the given place.
191     * @param place Increase number of markings for this place
192     */
193    public final void increaseMarkings(final PetriPlace place) {
194
195        place.increaseMarkings();
196
197    }
198
199    /**
200     * Decrease the number of markings for the given place.
201     * @param place Decrease number of markings for this place
202     */
203    public final void decreaseMarkings(final PetriPlace place) {
204
205        place.decreaseMarkings();
206
207    }
208
209    @Override
210    public final String toString() {
211        return "PetriNet [places=" + this.places + ", transitions="
212            + this.transitions + ", config=" + this.config + "]";
213    }
214
215    /**
216     * Rename an object.
217     * @param object Petri net object to be renamed
218     * @param name New name
219     */
220    public final void rename(final PetriObject object, final String name) {
221
222        object.setName(name);
223
224    }
225
226    /**
227     * Get a Petri net object's name.
228     * @param object Petri net object
229     * @return Object's name
230     */
231    public final String getName(final PetriObject object) {
232
233        return object.getName();
234
235    }
236
237    /**
238     * Get the number of markings set for the place.
239     * @param place Petri net place
240     * @return Number of markings set for place
241     */
242    public final int getMarkings(final PetriPlace place) {
243
244        return place.getMarkings();
245
246    }
247
248    /**
249     * Get an object's position.
250     * @param object Object
251     * @return Position
252     */
253    public final Point getPosition(final PetriObject object) {
254
255        return object.getPosition();
256
257    }
258
259    /**
260     * Get an object's size.
261     * @param object Object
262     * @return Size
263     */
264    public final Dimension getSize(final PetriObject object) {
265
266        return object.getSize();
267
268    }
269
270    /**
271     * Get all input places for a transition.
272     * @param transition Transition
273     * @return Incoming places
274     */
275    public final Set<PetriPlace> getInputEdges(
276        final PetriTransition transition) {
277
278        return transition.getInput();
279
280    }
281
282    /**
283     * Get all output places for a transition.
284     * @param transition Transition
285     * @return Outgoing places
286     */
287    public final Set<PetriPlace> getOutputEdges(
288        final PetriTransition transition) {
289
290        return transition.getOutput();
291
292    }
293
294    /**
295     * Set size for a given object.
296     * @param object Object
297     * @param size Size
298     */
299    public final void
300        setSize(final PetriObject object, final Dimension size) {
301
302        object.setSize(size);
303
304    }
305
306    /**
307     * Set position for a given object.
308     * @param object Object
309     * @param position Position
310     */
311    public final void setPosition(final PetriObject object,
312        final Point position) {
313
314        object.setPosition(position);
315
316    }
317
318}
