069/**
070 * Transform a {@link PetriNet} object into EPNML 1.1 and back. This markup is
071 * XML-based. The following is a basic example. <br />
072 * <br />
073 * &lt;pnml&gt; <br />
074 * &lt;net type="http://www.yasper.org/specs/epnml-1.1" id="net1"&gt; <br />
075 * &lt;transition id="transition1"&gt; <br />
076 * &lt;graphics&gt; <br />
077 * &lt;position x="78" y="78"/&gt; <br />
078 * &lt;dimension x="24" y="24"/&gt; <br />
079 * &lt;/graphics&gt; <br />
080 * &lt;/transition&gt; <br />
081 * &lt;place id="place1"&gt; <br />
082 * &lt;graphics&gt; <br />
083 * &lt;position x="78" y="12"/&gt; <br />
084 * &lt;dimension x="20" y="20"/&gt; <br />
085 * &lt;/graphics&gt; <br />
086 * &lt;initialMarking&gt; <br />
087 * &lt;text&gt;1&lt;/text&gt; <br />
088 * &lt;/initialMarking&gt; <br />
089 * &lt;/place&gt; <br />
090 * &lt;place id="place2"&gt; <br />
091 * &lt;graphics&gt; <br />
092 * &lt;position x="144" y="78"/&gt; <br />
093 * &lt;dimension x="20" y="20"/&gt; <br />
094 * &lt;/graphics&gt; <br />
095 * &lt;/place&gt; <br />
096 * &lt;arc id="arc1" source="place1" target="transition1"/&gt; <br />
097 * &lt;arc id="arc2" source="transition1" target="place2"/&gt; <br />
098 * &lt;/net&gt;
099 * @see http://www.petriweb.org/specs/epnml-1.1/pnmldef.pdf
100 * @see http://www.petriweb.org/specs/simple.pnml
101 * @author Tim Steinbach
102 */
103public final class PetriMarkup {

128    /**
129     * Load a file and get all Petri nets from it.
130     * @param file File to load
131     * @return Petri nets
132     * @throws PnmlException Error while parsing PNML
133     */
134    public static PetriSnapshots loadPnmlFile(final File file)
135        throws PnmlException {
136
137        PetriSnapshots logic = null;
138
139        final DocumentBuilderFactory factory =
140            DocumentBuilderFactory.newInstance();
141
142        DocumentBuilder docBuilder;
143
144        try {
145
146            docBuilder = factory.newDocumentBuilder();
147            final Document doc = docBuilder.parse(file);
148            logic = PetriMarkup.toPetrinet(doc);
149
150        } catch (final Exception e) {
151
152            throw new PnmlException(e);
153
154        }
155
156        return logic;
157
158    }

160    /**
161     * Convert all Petri nets and write PNML to file.
162     * @param file File to write to
163     * @param nets Petri nets
164     * @throws PnmlException Error while parsing PNML
165     */
166    public static void savePnmlFile(final File file,
167        final PetriSnapshots nets) throws PnmlException {
168
169        File tmpFile = file;
170
171        final String extension = PetriMarkup.CONFIG.getFileExtension();
172        final String fileName = tmpFile.getAbsolutePath();
173
174        if (!fileName.endsWith(extension)) {
175            tmpFile = new File(fileName + "." + extension);
176        }
177
178        final Document doc = PetriMarkup.toPnml(nets);
179
180        final Source source = new DOMSource(doc);
181        final Result result = new StreamResult(tmpFile);
182
183        try {
184
185            final Transformer xformer =
186                TransformerFactory.newInstance().newTransformer();
187            xformer.setOutputProperty(OutputKeys.OMIT_XML_DECLARATION,
188                "yes");
189            xformer.transform(source, result);
190
191        } catch (final Exception t) {
192
193            throw new PnmlException(t);
194
195        }
196
197    }

199    /**
200     * Convert a EPNML document into a {@link PetriNet} with all its snapshots.
201     * @param doc EPNML document
202     * @return PetriSnapshots object with all PetriNets
203     * @throws PnmlException Error parsing the file
204     */
205    public static PetriSnapshots toPetrinet(final Document doc)
206        throws PnmlException {
207
208        final PetriSnapshots result = new PetriSnapshots();
209
210        final Element root = doc.getDocumentElement();
211
212        XmlUtil.removeEmptyTextNodes(root);
213
214        if (root.hasChildNodes()) {
215
216            PetriMarkup.addAllNets(root, result);
217
218        }
219
220        return result;
221
222    }

224    /**
225     * Convert all Petri nets from the {@link PetriSnapshots} object into an
226     * EPNML document.
227     * @param nets PetriSnapshots element with Petri nets
228     * @return EPNML document
229     * @throws PnmlException Thrown if the XML parser cannot work with the input
230     *             from the snapshots object
231     */
232    public static Document toPnml(final PetriSnapshots nets)
233        throws PnmlException {
234
235        Document pnml = null;
236
237        try {
238
239            final DocumentBuilderFactory docFactory =
240                DocumentBuilderFactory.newInstance();
241            final DocumentBuilder docBuilder =
242                docFactory.newDocumentBuilder();
243
244            pnml = docBuilder.newDocument();
245
246            final PetrinetToMarkup pnmlBuilder = new PetrinetToMarkup(pnml);
247
248            final Element root = pnmlBuilder.createRoot();
249
250            final List<PetriNet> list = nets.getNets();
251
252            PetriMarkup.addAllPetriNets(list, pnmlBuilder, root);
253
254        } catch (final ParserConfigurationException e) {
255
256            throw new PnmlException(e);
257
258        }
259
260        return pnml;
261
262    }

264    /**
265     * Add all edges to a PNML net element.
266     * @param petriNet Petri net to work with
267     * @param transitions Transitions to read edges from
268     * @param pnmlBuilder PNML builder
269     * @param net Net element to add to
270     */
271    private static void addAllEdges(final PetriNet petriNet,
272        final Set<PetriTransition> transitions,
273        final PetrinetToMarkup pnmlBuilder, final Element net) {
274
275        int edgeId = 0;
276        final String edgeIdPrefix = PetriMarkup.CONFIG.getEdgeIdPrefix();
277
278        for (final PetriTransition petriTransition : transitions) {
279
280            final Set<PetriPlace> inEdges =
281                petriNet.getInputEdges(petriTransition);
282
283            final String targetId = petriNet.getId(petriTransition);
284
285            for (final PetriPlace petriPlace : inEdges) {
286
287                final String sourceId = petriNet.getId(petriPlace);
288
289                pnmlBuilder.addEdge(net, edgeIdPrefix + edgeId++, sourceId,
290                    targetId);
291
292            }
293
294            final Set<PetriPlace> outEdges =
295                petriNet.getOutputEdges(petriTransition);
296
297            final String sourceId = targetId;
298
299            for (final PetriPlace petriPlace : outEdges) {
300
301                final String targetId2 = petriNet.getId(petriPlace);
302
303                pnmlBuilder.addEdge(net, edgeIdPrefix + edgeId++, sourceId,
304                    targetId2);
305
306            }
307
308        }
309
310    }

313     * Add all markings to the net root element.
314     * @param petriNet Logical net
315     * @param pnmlBuilder Document builder
316     * @param net Net root element to add to
317     */
318    private static void addAllMarkings(final PetriNet petriNet,
319        final PetrinetToMarkup pnmlBuilder, final Element net) {
320
321        final Set<PetriMarking> markings = petriNet.getMarkings();
322
323        if (markings.size() > 0) {
324
325            final Element markingsRoot = pnmlBuilder.createMarkingsRoot();
326
327            PetriMarkup.addMarkings(petriNet, pnmlBuilder, markings,
328                markingsRoot);
329
330            net.appendChild(markingsRoot);
331
332        }
333
334    }

336    /**
337     * Add all nets from the root element to the resulting logic.
338     * @param root XML root element
339     * @param result Logical result
340     * @throws PnmlException Error parsing the file
341     */
342    private static void addAllNets(final Element root,
343        final PetriSnapshots result) throws PnmlException {
344
345        // Get individual nets
346
347        final NodeList nets = root.getChildNodes();
348
349        final int size = nets.getLength();
350
351        for (int i = 0; i < size; i++) {
352
353            final Node node = nets.item(i);
354
355            // Makes loading the file much more error-resistant
356            if (node instanceof Element) {
357
358                final Element net = (Element) node;
359
360                PetriMarkup.addNextNet(result, net);
361
362            }
363
364        }
365
366    }

368    /**
369     * Add all Petri nets to the root element.
370     * @param list List of Petri nets
371     * @param pnmlBuilder PNML to markup builder
372     * @param root Root element
373     */
374    private static void addAllPetriNets(final List<PetriNet> list,
375        final PetrinetToMarkup pnmlBuilder, final Element root) {
376
377        for (final PetriNet petriNet : list) {
378
379            final Element net =
380                pnmlBuilder.addNet(root, petriNet.getName());
381
382            final Set<PetriPlace> places = petriNet.getPlaces();
383
384            PetriMarkup.addAllPlaces(petriNet, places, pnmlBuilder, net);
385
386            final Set<PetriTransition> transitions =
387                petriNet.getTransitions();
388
389            PetriMarkup.addAllTransitions(petriNet, transitions,
390                pnmlBuilder, net);
391
392            PetriMarkup
393                .addAllEdges(petriNet, transitions, pnmlBuilder, net);
394
395            PetriMarkup.addAllMarkings(petriNet, pnmlBuilder, net);
396
397        }
398
399    }

401    /**
402     * Add all places to a PNML net element.
403     * @param petriNet Petri net to work with
404     * @param places Set of places to be added
405     * @param pnmlBuilder PNML builder
406     * @param net Net element
407     */
408    private static void addAllPlaces(final PetriNet petriNet,
409        final Set<PetriPlace> places, final PetrinetToMarkup pnmlBuilder,
410        final Element net) {
411
412        for (final PetriPlace petriPlace : places) {
413
414            final String id =
415                petriNet.getId(petriPlace) + "_"
416                    + petriNet.getName(petriPlace);
417            final int value = petriNet.getTokens(petriPlace);
418
419            final Element place = pnmlBuilder.addPlace(net, id, value);
420            final Point position = petriNet.getPosition(petriPlace);
421            final Dimension size = petriNet.getSize(petriPlace);
422
423            pnmlBuilder.addGraphics(place, position, size);
424
425        }

456    /**
457     * Add the connection between a place and a transition.
458     * @param logicalNet Logical net
459     * @param elem Arc element
460     */
461    private static void
462        addArc(final PetriNet logicalNet, final Element elem) {
463
464        final String sourceId = elem.getAttribute(PnmlElements.EDGE_SOURCE);
465        final String targetId = elem.getAttribute(PnmlElements.EDGE_TARGET);
466
467        PetriTransition transition = null;
468        PetriPlace place = null;
469
470        boolean isInput = false;
471
472        transition = logicalNet.getTransitionById(sourceId);
473
474        if (null == transition) {
475
476            transition = logicalNet.getTransitionById(targetId);
477            place = logicalNet.getPlaceById(sourceId);
478            isInput = true;
479
480        } else {
481
482            place = logicalNet.getPlaceById(targetId);
483
484        }
485
486        if (isInput) {
487
488            logicalNet.connect(place, transition);
489
490        } else {
491
492            logicalNet.connect(transition, place);
493
494        }
495
496    }

498    /**
499     * Add markings to the logical net.
500     * @param net Logical net
501     * @param root Markings root
502     * @throws PnmlException Malformed PNML file
503     */
504    private static void addMarkings(final PetriNet net, final Element root)
505        throws PnmlException {
506
507        final NodeList markings = root.getChildNodes();
508
509        for (int i = 0; i < markings.getLength(); i++) {
510
511            final Node child = markings.item(i);
512
513            if ((child instanceof Element)
514                && (PnmlElements.MARKING.equals(child.getNodeName()))) {
515
516                final Element markingElem = (Element) child;
517
518                final String name =
519                    markingElem.getAttribute(PnmlElements.MARKING_NAME);
520                final String id =
521                    markingElem.getAttribute(PnmlElements.MARKING_ID);
522
523                final PetriMarking marking = net.createMarking(name, id);
524
525                final NodeList places = markingElem.getChildNodes();
526
527                for (int j = 0; j < places.getLength(); j++) {
528
529                    final Node placeNode = places.item(j);
530
531                    if ((placeNode instanceof Element)
532                        && (PnmlElements.MPLACE.equals(placeNode
533                            .getNodeName()))) {
534
535                        final Element placeElem = (Element) placeNode;
536
537                        final String placeId =
538                            placeElem.getAttribute(PnmlElements.MPLACE_ID);
539                        final int placeTokens =
540                            Integer.valueOf(placeElem
541                                .getAttribute(PnmlElements.MPLACE_TOKENS));
542
543                        final PetriPlace place = net.getPlaceById(placeId);
544
545                        marking.setTokens(place, placeTokens);
546
547                    } else {
548
549                        throw new PnmlException("File malformed!");
550
551                    }
552
553                }
554
555            } else {
556
557                throw new PnmlException("File malformed!");
558
559            }
560
561        }
562
563    }

565    /**
566     * Iterate through all markings and add them to the markings root node.
567     * @param petriNet Logical net
568     * @param pnmlBuilder Document builder
569     * @param markings Set of markings
570     * @param markingsRoot Markings root node
571     */
572    private static void addMarkings(final PetriNet petriNet,
573        final PetrinetToMarkup pnmlBuilder,
574        final Set<PetriMarking> markings, final Element markingsRoot) {
575
576        for (final PetriMarking petriMarking : markings) {
577
578            final String id = petriMarking.getId();
579            final String name = petriMarking.getName();
580
581            final Element marking = pnmlBuilder.createMarking(id, name);
582
583            final Map<PetriPlace, Integer> tokens =
584                petriMarking.getAllTokens();
585
586            final Set<Entry<PetriPlace, Integer>> tokenSet =
587                tokens.entrySet();
588            for (final Entry<PetriPlace, Integer> entry : tokenSet) {
589
590                pnmlBuilder.addMarkingPlace(marking,
591                    petriNet.getId(entry.getKey()), entry.getValue());
592
593            }
594
595            markingsRoot.appendChild(marking);
596
597        }
598
599    }

601    /**
602     * Method that adds the next Petri net from its XML representation to the
603     * logical net.
604     * @param result Resulting logic
605     * @param net Net element
606     * @throws PnmlException Error because of malformed file
607     */
608    private static void addNextNet(final PetriSnapshots result,
609        final Element net) throws PnmlException {
610
611        final String netName = net.getAttribute(PnmlElements.NET_ID);
612
613        final PetriNet logicalNet = result.add(netName);
614
615        // We need this for the initial creation, will be deleted
616        final PetriMarking tmpMarking =
617            logicalNet.createNewMarking(PetriMarkup.TMP_MARKING);
618
619        // Places, transitions, arcs, markings
620        final NodeList pta = net.getChildNodes();
621
622        final int ptaSize = pta.getLength();
623
624        int markingsIndex = -1;
625
626        for (int j = 0; j < ptaSize; j++) {
627
628            final Node elemNode = pta.item(j);
629
630            if (elemNode instanceof Element) {
631
632                final Element elem = (Element) elemNode;
633
634                final String nodeName = elem.getNodeName();
635
636                if (PnmlElements.PLACE.equals(nodeName)) {
637
638                    PetriMarkup.addPlace(logicalNet, elem);
639
640                } else if (PnmlElements.TRANSITION.equals(nodeName)) {
641
642                    PetriMarkup.addTransition(logicalNet, elem);
643
644                } else if (PnmlElements.EDGE.equals(nodeName)) {
645
646                    PetriMarkup.addArc(logicalNet, elem);
647
648                } else {
649
650                    // Skip the markings elements, throw an error if anything
651                    // unknown has been found
652                    if (PnmlElements.MARKINGS.equals(nodeName)) {
653
654                        markingsIndex = j;
655
656                    } else {
657
658                        throw new PnmlException("File malformed!");
659
660                    }
661
662                }
663
664            }
665
666        }
667
668        if (markingsIndex == -1) { // Could not find any markings in file
669
670            logicalNet.createNewMarking(PetriMarkup.MSG.getNullMarking());
671
672        } else { // Found markings, so add them
673
674            final Element markingsElement =
675                (Element) pta.item(markingsIndex);
676
677            PetriMarkup.addMarkings(logicalNet, markingsElement);
678
679        }
680
681        logicalNet.deleteMarking(tmpMarking);
682
683        final PetriMarking marking =
684            logicalNet.getMarkings().iterator().next();
685        logicalNet.switchMarking(marking);
686
687    }

689    /**
690     * Add a place to the logical net.
691     * @param logicalNet Logical net
692     * @param elem PNML element for place
693     * @throws PnmlException Error parsing the file
694     */
695    private static void addPlace(final PetriNet logicalNet,
696        final Element elem) throws PnmlException {
697
698        final String nameId = elem.getAttribute(PnmlElements.PLACE_ID);
699        final String name = nameId.substring(nameId.indexOf("_") + 1);
700        final String id = nameId.substring(0, nameId.indexOf("_"));
701
702        PetriObjectId.addUsedId(id);
703
704        final Element graphics =
705            PetriMarkup.getElementByName(elem, PnmlElements.GRAPHICS);
706        final Element markings =
707            PetriMarkup.getElementByName(elem,
708                PnmlElements.PLACE_INITIAL_MARKING);
709
710        final int markingsValue =
711            Integer.parseInt(PetriMarkup.getTextValue(markings,
712                PnmlElements.PLACE_INITIAL_MARKING_TEXT));
713
714        final int[] graphicsValues =
715            PetriMarkup.getGraphicsValues(graphics);
716        final int x = graphicsValues[0];
717        final int y = graphicsValues[1];
718        final int width = graphicsValues[2];
719        final int height = graphicsValues[3];
720
721        final PetriPlace place = logicalNet.addPlace(name, id);
722        for (int i = 0; i < markingsValue; i++) {
723            logicalNet.increaseTokens(place);
724        }
725        logicalNet.setPosition(place, new Point(x, y));
726        logicalNet.setSize(place, new Dimension(width, height));
727
728    }

761    /**
762     * Find an element under a root element.
763     * @param root Root element
764     * @param name Name of child
765     * @return Child element
766     * @throws PnmlException Error finding child element
767     */
768    private static Element getElementByName(final Element root,
769        final String name) throws PnmlException {
770
771        final NodeList children = root.getChildNodes();
772        final int size = children.getLength();
773
774        for (int i = 0; i < size; i++) {
775
776            final Node node = children.item(i);
777
778            if (node instanceof Element) {
779
780                final Element elem = (Element) node;
781                if ((null != elem) && (name.equals(elem.getNodeName()))) {
782                    return elem;
783                }
784
785            }
786
787        }
788
789        throw new PnmlException("Element not found");
790
791    }

793    /**
794     * Get the graphics values form a graphics PNML element.
795     * @param graphics Graphics PNML element
796     * @return {x, y, width, height}
797     * @throws PnmlException Error parsing file
798     */
799    private static int[] getGraphicsValues(final Node graphics)
800        throws PnmlException {
801
802        final int[] result = new int[4];
803
804        final Element graphics1 = (Element) graphics.getFirstChild();
805        final Element graphics2 = (Element) graphics.getLastChild();
806
807        if ((null == graphics1) || (null == graphics2)) {
808            throw new PnmlException("File malformed!");
809        } else {
810
811            Element position = null;
812            Element dimension = null;
813
814            if ((PnmlElements.POSITION.equals(graphics1.getNodeName()))
815                && (PnmlElements.DIMENSION.equals(graphics2.getNodeName()))) {
816
817                position = graphics1;
818                dimension = graphics2;
819
820            } else if ((PnmlElements.POSITION.equals(graphics2
821                .getNodeName()))
822                && (PnmlElements.DIMENSION.equals(graphics1.getNodeName()))) {
823
824                position = graphics2;
825                dimension = graphics1;
826
827            } else {
828                throw new PnmlException("File malformed!");
829            }
830
831            final String xText =
832                position.getAttribute(PnmlElements.POSITION_X);
833            final String yText =
834                position.getAttribute(PnmlElements.POSITION_Y);
835            final String widthText =
836                dimension.getAttribute(PnmlElements.DIMENSION_WIDTH);
837            final String heightText =
838                dimension.getAttribute(PnmlElements.DIMENSION_HEIGHT);
839
840            if ((null == xText) || (null == yText) || (null == widthText)
841                || (null == heightText)) {
842                throw new PnmlException("File malformed!");
843            } else {
844                result[0] = Integer.parseInt(xText);
845                result[1] = Integer.parseInt(yText);
846                result[2] = Integer.parseInt(widthText);
847                result[3] = Integer.parseInt(heightText);
848            }
849
850        }
851
852        return result;
853
854    }

857     * Get the text value of a tag under the root element.
858     * @param root Root element
859     * @param tag Tag to find
860     * @return Text value of tag under root
861     * @throws PnmlException Error finding tag
862     */
863    private static String
864        getTextValue(final Element root, final String tag)
865            throws PnmlException {
866
867        final NodeList list = root.getChildNodes();
868        final int size = list.getLength();
869
870        for (int i = 0; i < size; i++) {
871
872            final Node node = list.item(i);
873
874            if (node instanceof Element) {
875
876                final Element elem = (Element) node;
877
878                if ((null != elem) && (tag.equals(elem.getNodeName()))) {
879                    return elem.getTextContent();
880                }
881
882            }
883
884        }
885
886        throw new PnmlException("Tag not found!");
887
888    }
