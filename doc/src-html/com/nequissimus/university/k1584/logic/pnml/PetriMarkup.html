032/**
033 * Transform a {@link PetriNet} object into EPNML 1.1 and back. This markup is
034 * XML-based. The following is a basic example. <br />
035 * <br />
036 * &lt;pnml&gt; <br />
037 * &lt;net type="http://www.yasper.org/specs/epnml-1.1" id="net1"&gt; <br />
038 * &lt;transition id="transition1"&gt; <br />
039 * &lt;graphics&gt; <br />
040 * &lt;position x="78" y="78"/&gt; <br />
041 * &lt;dimension x="24" y="24"/&gt; <br />
042 * &lt;/graphics&gt; <br />
043 * &lt;/transition&gt; <br />
044 * &lt;place id="place1"&gt; <br />
045 * &lt;graphics&gt; <br />
046 * &lt;position x="78" y="12"/&gt; <br />
047 * &lt;dimension x="20" y="20"/&gt; <br />
048 * &lt;/graphics&gt; <br />
049 * &lt;initialMarking&gt; <br />
050 * &lt;text&gt;1&lt;/text&gt; <br />
051 * &lt;/initialMarking&gt; <br />
052 * &lt;/place&gt; <br />
053 * &lt;place id="place2"&gt; <br />
054 * &lt;graphics&gt; <br />
055 * &lt;position x="144" y="78"/&gt; <br />
056 * &lt;dimension x="20" y="20"/&gt; <br />
057 * &lt;/graphics&gt; <br />
058 * &lt;/place&gt; <br />
059 * &lt;arc id="arc1" source="place1" target="transition1"/&gt; <br />
060 * &lt;arc id="arc2" source="transition1" target="place2"/&gt; <br />
061 * &lt;/net&gt;
062 * @see http://www.petriweb.org/specs/epnml-1.1/pnmldef.pdf
063 * @see http://www.petriweb.org/specs/simple.pnml
064 * @author Tim Steinbach
065 */
066public final class PetriMarkup {

079    /**
080     * Load a file and get all Petri nets from it.
081     * @param file File to load
082     * @return Petri nets
083     * @throws PnmlException Error while parsing PNML
084     */
085    public static PetriSnapshots loadPnmlFile(final File file)
086        throws PnmlException {
087
088        PetriSnapshots logic = null;
089
090        final DocumentBuilderFactory factory =
091            DocumentBuilderFactory.newInstance();
092        DocumentBuilder docBuilder;
093
094        try {
095
096            docBuilder = factory.newDocumentBuilder();
097            final Document doc = docBuilder.parse(file);
098            logic = PetriMarkup.toPetrinet(doc);
099
100        } catch (final Exception e) {
101
102            throw new PnmlException(e);
103
104        }
105
106        return logic;
107
108    }

110    /**
111     * Convert all Petri nets and write PNML to file.
112     * @param file File to write to
113     * @param nets Petri nets
114     * @throws PnmlException Error while parsing PNML
115     */
116    public static void savePnmlFile(final File file,
117        final PetriSnapshots nets) throws PnmlException {
118
119        File tmpFile = file;
120
121        final String extension = PetriMarkup.CONFIG.getFileExtension();
122        final String fileName = tmpFile.getAbsolutePath();
123
124        if (!fileName.endsWith(extension)) {
125            tmpFile = new File(fileName + "." + extension);
126        }
127
128        final Document doc = PetriMarkup.toPnml(nets);
129
130        final Source source = new DOMSource(doc);
131        final Result result = new StreamResult(tmpFile);
132
133        try {
134
135            final Transformer xformer =
136                TransformerFactory.newInstance().newTransformer();
137            xformer.setOutputProperty(OutputKeys.OMIT_XML_DECLARATION,
138                "yes");
139            xformer.transform(source, result);
140
141        } catch (final Exception t) {
142
143            throw new PnmlException(t);
144
145        }
146
147    }

149    /**
150     * Convert a EPNML document into a {@link PetriNet} with all its snapshots.
151     * @param doc EPNML document
152     * @return PetriSnapshots object with all PetriNets
153     * @throws PnmlException Error parsing the file
154     */
155    public static PetriSnapshots toPetrinet(final Document doc)
156        throws PnmlException {
157
158        final PetriSnapshots result = new PetriSnapshots();
159
160        final Element root = doc.getDocumentElement();
161
162        if (root.hasChildNodes()) {
163
164            PetriMarkup.addAllNets(root, result);
165
166        }
167
168        return result;
169
170    }

172    /**
173     * Convert all Petri nets from the {@link PetriSnapshots} object into an
174     * EPNML document.
175     * @param nets PetriSnapshots element with Petri nets
176     * @return EPNML document
177     * @throws PnmlException Thrown if the XML parser cannot work with the input
178     *             from the snapshots object
179     */
180    public static Document toPnml(final PetriSnapshots nets)
181        throws PnmlException {
182
183        Document pnml = null;
184
185        try {
186
187            final DocumentBuilderFactory docFactory =
188                DocumentBuilderFactory.newInstance();
189            final DocumentBuilder docBuilder =
190                docFactory.newDocumentBuilder();
191
192            pnml = docBuilder.newDocument();
193
194            final PetrinetToMarkup pnmlBuilder = new PetrinetToMarkup(pnml);
195
196            final Element root = pnmlBuilder.createRoot();
197
198            final List<PetriNet> list = nets.getNets();
199
200            PetriMarkup.addAllPetriNets(list, pnmlBuilder, root);
201
202        } catch (final ParserConfigurationException e) {
203
204            throw new PnmlException(e);
205
206        }
207
208        return pnml;
209
210    }

212    /**
213     * Add all edges to a PNML net element.
214     * @param petriNet Petri net to work with
215     * @param transitions Transitions to read edges from
216     * @param pnmlBuilder PNML builder
217     * @param net Net element to add to
218     */
219    private static void addAllEdges(final PetriNet petriNet,
220        final Set<PetriTransition> transitions,
221        final PetrinetToMarkup pnmlBuilder, final Element net) {
222
223        int edgeId = 0;
224        final String edgeIdPrefix = PetriMarkup.CONFIG.getEdgeIdPrefix();
225
226        for (final PetriTransition petriTransition : transitions) {
227
228            final Set<PetriPlace> inEdges =
229                petriNet.getInputEdges(petriTransition);
230
231            final String targetId = petriNet.getId(petriTransition);
232
233            for (final PetriPlace petriPlace : inEdges) {
234
235                final String sourceId = petriNet.getId(petriPlace);
236
237                pnmlBuilder.addEdge(net, edgeIdPrefix + edgeId++, sourceId,
238                    targetId);
239
240            }
241
242            final Set<PetriPlace> outEdges =
243                petriNet.getOutputEdges(petriTransition);
244
245            final String sourceId = targetId;
246
247            for (final PetriPlace petriPlace : outEdges) {
248
249                final String targetId2 = petriNet.getId(petriPlace);
250
251                pnmlBuilder.addEdge(net, edgeIdPrefix + edgeId++, sourceId,
252                    targetId2);
253
254            }
255
256        }
257
258    }

260    /**
261     * Add all nets from the root element to the resulting logic.
262     * @param root XML root element
263     * @param result Logical result
264     * @throws PnmlException Error parsing the file
265     */
266    private static void addAllNets(final Element root,
267        final PetriSnapshots result) throws PnmlException {
268
269        // Get individual nets
270
271        final NodeList nets = root.getChildNodes();
272
273        final int size = nets.getLength();
274
275        for (int i = 0; i < size; i++) {
276
277            final Node node = nets.item(i);
278
279            // Makes loading the file much more error-resistant
280            if (node instanceof Element) {
281
282                final Element net = (Element) node;
283
284                PetriMarkup.addNextNet(result, net);
285
286            }
287
288        }
289
290    }

292    /**
293     * Add all Petri nets to the root element.
294     * @param list List of Petri nets
295     * @param pnmlBuilder PNML to markup builder
296     * @param root Root element
297     */
298    private static void addAllPetriNets(final List<PetriNet> list,
299        final PetrinetToMarkup pnmlBuilder, final Element root) {
300
301        for (final PetriNet petriNet : list) {
302
303            final Element net =
304                pnmlBuilder.addNet(root, petriNet.getName());
305
306            final Set<PetriPlace> places = petriNet.getPlaces();
307
308            PetriMarkup.addAllPlaces(petriNet, places, pnmlBuilder, net);
309
310            final Set<PetriTransition> transitions =
311                petriNet.getTransitions();
312
313            PetriMarkup.addAllTransitions(petriNet, transitions,
314                pnmlBuilder, net);
315
316            PetriMarkup
317                .addAllEdges(petriNet, transitions, pnmlBuilder, net);
318
319        }
320
321    }

323    /**
324     * Add all places to a PNML net element.
325     * @param petriNet Petri net to work with
326     * @param places Set of places to be added
327     * @param pnmlBuilder PNML builder
328     * @param net Net element
329     */
330    private static void addAllPlaces(final PetriNet petriNet,
331        final Set<PetriPlace> places, final PetrinetToMarkup pnmlBuilder,
332        final Element net) {
333
334        for (final PetriPlace petriPlace : places) {
335
336            final String id =
337                petriNet.getId(petriPlace) + "_"
338                    + petriNet.getName(petriPlace);
339            final int value = petriNet.getMarkings(petriPlace);
340
341            final Element place = pnmlBuilder.addPlace(net, id, value);
342            final Point position = petriNet.getPosition(petriPlace);
343            final Dimension size = petriNet.getSize(petriPlace);
344
345            pnmlBuilder.addGraphics(place, position, size);
346
347        }
348
349    }

354     * @param transitions Set of transitions to be added
355     * @param pnmlBuilder PNML builder
356     * @param net Net element

358    private static void addAllTransitions(final PetriNet petriNet,

378    /**
379     * Add the connection between a place and a transition.
380     * @param logicalNet Logical net
381     * @param elem Arc element
382     */
383    private static void
384        addArc(final PetriNet logicalNet, final Element elem) {
385
386        final String sourceId = elem.getAttribute(PnmlElements.EDGE_SOURCE);
387        final String targetId = elem.getAttribute(PnmlElements.EDGE_TARGET);
388
389        PetriTransition transition = null;
390        PetriPlace place = null;
391
392        boolean isInput = false;
393
394        transition = logicalNet.getTransitionById(sourceId);
395
396        if (null == transition) {
397
398            transition = logicalNet.getTransitionById(targetId);
399            place = logicalNet.getPlaceById(sourceId);
400            isInput = true;
401
402        } else {
403
404            place = logicalNet.getPlaceById(targetId);
405
406        }
407
408        if (isInput) {
409
410            logicalNet.connect(place, transition);
411
412        } else {
413
414            logicalNet.connect(transition, place);
415
416        }
417
418    }

420    /**
421     * Method that adds the next Petri net from its XML representation to the
422     * logical net.
423     * @param result Resulting logic
424     * @param net Net element
425     * @throws PnmlException Error because of malformed file
426     */
427    private static void addNextNet(final PetriSnapshots result,
428        final Element net) throws PnmlException {
429
430        final String netName = net.getAttribute(PnmlElements.NET_ID);
431
432        final PetriNet logicalNet = result.add(netName);
433
434        // Places, transitions, arcs
435        final NodeList pta = net.getChildNodes();
436
437        final int ptaSize = pta.getLength();
438
439        for (int j = 0; j < ptaSize; j++) {
440
441            final Element elem = (Element) pta.item(j);
442
443            final String nodeName = elem.getNodeName();
444
445            if (PnmlElements.PLACE.equals(nodeName)) {
446
447                PetriMarkup.addPlace(logicalNet, elem);
448
449            } else if (PnmlElements.TRANSITION.equals(nodeName)) {
450
451                PetriMarkup.addTransition(logicalNet, elem);
452
453            } else if (PnmlElements.EDGE.equals(nodeName)) {
454
455                PetriMarkup.addArc(logicalNet, elem);
456
457            } else {
458                throw new PnmlException("File malformed!");
459            }
460
461        }
462    }

464    /**
465     * Add a place to the logical net.
466     * @param logicalNet Logical net
467     * @param elem PNML element for place
468     * @throws PnmlException Error parsing the file
469     */
470    private static void addPlace(final PetriNet logicalNet,
471        final Element elem) throws PnmlException {
472
473        final String nameId = elem.getAttribute(PnmlElements.PLACE_ID);
474        final String name = nameId.substring(nameId.indexOf("_") + 1);
475        final String id = nameId.substring(0, nameId.indexOf("_"));
476
477        PetriObjectId.addUsedId(id);
478
479        final Element graphics =
480            PetriMarkup.getElementByName(elem, PnmlElements.GRAPHICS);
481        final Element markings =
482            PetriMarkup.getElementByName(elem,
483                PnmlElements.PLACE_INITIAL_MARKING);
484
485        final int markingsValue =
486            Integer.parseInt(PetriMarkup.getTextValue(markings,
487                PnmlElements.PLACE_INITIAL_MARKING_TEXT));
488
489        final int[] graphicsValues =
490            PetriMarkup.getGraphicsValues(graphics);
491        final int x = graphicsValues[0];
492        final int y = graphicsValues[1];
493        final int width = graphicsValues[2];
494        final int height = graphicsValues[3];
495
496        final PetriPlace place = logicalNet.addPlace(name, id);
497        for (int i = 0; i < markingsValue; i++) {
498            logicalNet.increaseMarkings(place);
499        }
500        logicalNet.setPosition(place, new Point(x, y));
501        logicalNet.setSize(place, new Dimension(width, height));
502
503    }

507     * @param logicalNet Logical net
508     * @param elem PNML element
509     * @throws PnmlException Error parsing element

511    private static void addTransition(final PetriNet logicalNet,
512        final Element elem) throws PnmlException {
513
514        final String nameId = elem.getAttribute(PnmlElements.TRANSITION_ID);
515        final String name = nameId.substring(nameId.indexOf("_") + 1);
516        final String id = nameId.substring(0, nameId.indexOf("_"));
517
518        PetriObjectId.addUsedId(id);
519
520        final Element graphics =
521            PetriMarkup.getElementByName(elem, PnmlElements.GRAPHICS);
522        final int[] graphicsValues =
523            PetriMarkup.getGraphicsValues(graphics);
524        final int x = graphicsValues[0];
525        final int y = graphicsValues[1];
526        final int width = graphicsValues[2];
527        final int height = graphicsValues[3];
528
529        final PetriTransition transition =
530            logicalNet.addTransition(name, id);
531        logicalNet.setSize(transition, new Dimension(width, height));
532        logicalNet.setPosition(transition, new Point(x, y));
533
534    }

536    /**
537     * Find an element under a root element.
538     * @param root Root element
539     * @param name Name of child
540     * @return Child element
541     * @throws PnmlException Error finding child element
542     */
543    private static Element getElementByName(final Element root,
544        final String name) throws PnmlException {
545
546        final NodeList children = root.getChildNodes();
547        final int size = children.getLength();
548
549        for (int i = 0; i < size; i++) {
550
551            final Element node = (Element) children.item(i);
552            if ((null != node) && (name.equals(node.getNodeName()))) {
553                return node;
554            }
555
556        }
557
558        throw new PnmlException("Element not found");
559
560    }

562    /**
563     * Get the graphics values form a graphics PNML element.
564     * @param graphics Graphics PNML element
565     * @return {x, y, width, height}
566     * @throws PnmlException Error parsing file
567     */
568    private static int[] getGraphicsValues(final Node graphics)
569        throws PnmlException {
570
571        final int[] result = new int[4];
572
573        final Element graphics1 = (Element) graphics.getFirstChild();
574        final Element graphics2 = (Element) graphics.getLastChild();
575
576        if ((null == graphics1) || (null == graphics2)) {
577            throw new PnmlException("File malformed!");
578        } else {
579
580            Element position = null;
581            Element dimension = null;
582
583            if ((PnmlElements.POSITION.equals(graphics1.getNodeName()))
584                && (PnmlElements.DIMENSION.equals(graphics2.getNodeName()))) {
585
586                position = graphics1;
587                dimension = graphics2;
588
589            } else if ((PnmlElements.POSITION.equals(graphics2
590                .getNodeName()))
591                && (PnmlElements.DIMENSION.equals(graphics1.getNodeName()))) {
592
593                position = graphics2;
594                dimension = graphics1;
595
596            } else {
597                throw new PnmlException("File malformed!");
598            }
599
600            final String xText =
601                position.getAttribute(PnmlElements.POSITION_X);
602            final String yText =
603                position.getAttribute(PnmlElements.POSITION_Y);
604            final String widthText =
605                dimension.getAttribute(PnmlElements.DIMENSION_WIDTH);
606            final String heightText =
607                dimension.getAttribute(PnmlElements.DIMENSION_HEIGHT);
608
609            if ((null == xText) || (null == yText) || (null == widthText)
610                || (null == heightText)) {
611                throw new PnmlException("File malformed!");
612            } else {
613                result[0] = Integer.parseInt(xText);
614                result[1] = Integer.parseInt(yText);
615                result[2] = Integer.parseInt(widthText);
616                result[3] = Integer.parseInt(heightText);
617            }
618
619        }
620
621        return result;
622
623    }

625    /**
626     * Get the text value of a tag under the root element.
627     * @param root Root element
628     * @param tag Tag to find
629     * @return Text value of tag under root
630     * @throws PnmlException Error finding tag
631     */
632    private static String
633        getTextValue(final Element root, final String tag)
634            throws PnmlException {
635
636        final NodeList list = root.getChildNodes();
637        final int size = list.getLength();
638
639        for (int i = 0; i < size; i++) {
640
641            final Element elem = (Element) list.item(i);
642
643            if ((null != elem) && (tag.equals(elem.getNodeName()))) {
644                return elem.getTextContent();
645            }
646
647        }
648
649        throw new PnmlException("Tag not found!");
650
651    }
