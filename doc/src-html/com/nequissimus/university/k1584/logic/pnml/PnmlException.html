032package com.nequissimus.university.k1584.logic.pnml;
033
034/**
035 * This exception will be thrown when there is something wrong while converting
036 * a Petrinet into EPNML or back.
037 * @author Tim Steinbach
038 */
039public class PnmlException extends Exception {
040
041    /**
042     * Serializable UID.
043     */
044    private static final long serialVersionUID = -165689230946792569L;
045
046    /**
047     * Create new PnmlException.
048     * @param e Root exception
049     */
050    public PnmlException(final Exception e) {
051
052        super(e);
053
054    }
055
056    /**
057     * Create new PnmlException.
058     * @param msg Message
059     */
060    public PnmlException(final String msg) {
061
062        super("PnmlException: " + msg);
063
064    }
065
066}
