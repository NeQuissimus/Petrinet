032package com.nequissimus.university.k1584.logic.pnml;
033
034/**
035 * All necessary XML elements for EPNML and their attributes. Some attributes
036 * have default values that should not be changed. The naming of these constants
037 * follows the following logic:<br />
038 * <br />
039 * XYZ = Element XYZ <br />
040 * XYZ_ABC = Attribute ABC for element XYZ<br />
041 * XYZ_ABC_VALUE = Default value for attribute ABC of element XYZ
042 * @author Tim Steinbach
043 */
044public final class PnmlElements {
045
046    /**
047     * PNML tag for a dimension.
048     */
049    public static final String DIMENSION = "dimension";
050
051    /**
052     * Height attribute for a dimension.
053     */
054    public static final String DIMENSION_HEIGHT = "y";
055
056    /**
057     * Width attribute for a dimension.
058     */
059    public static final String DIMENSION_WIDTH = "x";
060
061    /**
062     * PNML tag for an edge.
063     */
064    public static final String EDGE = "arc";
065
066    /**
067     * ID attribute for an edge.
068     */
069    public static final String EDGE_ID = "id";
070
071    /**
072     * Source attribute for an edge.
073     */
074    public static final String EDGE_SOURCE = "source";
075
076    /**
077     * Target attribute for an edge.
078     */
079    public static final String EDGE_TARGET = "target";
080
081    /**
082     * PNML tag for graphics.
083     */
084    public static final String GRAPHICS = "graphics";
085
086    /**
087     * Root for markings.
088     */
089    public static final String MARKINGS = "markings";
090
091    /**
092     * Root for individual marking.
093     */
094    public static final String MARKING = "marking";
095
096    /**
097     * Marking id attribute.
098     */
099    public static final String MARKING_ID = "id";
100
101    /**
102     * Marking name attribute.
103     */
104    public static final String MARKING_NAME = "name";
105
106    /**
107     * Place in marking.
108     */
109    public static final String MPLACE = "place";
110
111    /**
112     * Marking place id attribute.
113     */
114    public static final String MPLACE_ID = "id";
115
116    /**
117     * Marking place number of tokens attribute.
118     */
119    public static final String MPLACE_TOKENS = "tokens";
120
121    /**
122     * PNML tag for a net.
123     */
124    public static final String NET = "net";
125
126    /**
127     * ID attribute for net.
128     */
129    public static final String NET_ID = "id";
130
131    /**
132     * Type attribute for net.
133     */
134    public static final String NET_TYPE = "type";
135
136    /**
137     * Default type value for net value.
138     */
139    public static final String NET_TYPE_VALUE =
140        "http://www.yasper.org/specs/epnml-1.1";
141
142    /**
143     * PNML tag for a place.
144     */
145    public static final String PLACE = "place";
146
147    /**
148     * ID attribute for a place.
149     */
150    public static final String PLACE_ID = "id";
151
152    /**
153     * Node for markings.
154     */
155    public static final String PLACE_INITIAL_MARKING = "initialMarking";
156
157    /**
158     * Text element holding the markings value.
159     */
160    public static final String PLACE_INITIAL_MARKING_TEXT = "text";
161
162    /**
163     * PNML tag for a position.
164     */
165    public static final String POSITION = "position";
166
167    /**
168     * X attribute for a position.
169     */
170    public static final String POSITION_X = "x";
171
172    /**
173     * Y attribute for a position.
174     */
175    public static final String POSITION_Y = "y";
176
177    /**
178     * PNML tag for a root.
179     */
180    public static final String ROOT = "pnml";
181
182    /**
183     * PNML tag for a transition.
184     */
185    public static final String TRANSITION = "transition";
186
187    /**
188     * ID attribute for transition.
189     */
190    public static final String TRANSITION_ID = "id";
191
192    /**
193     * Hide constructor.
194     */
195    private PnmlElements() {
196    }
197
198}
