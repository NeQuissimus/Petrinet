001package com.nequissimus.university.k1584.logic.pnml;
002
003import java.awt.Dimension;
004import java.awt.Point;
005
006import org.w3c.dom.Document;
007import org.w3c.dom.Element;
008
009/**
010 * Class that can turn Petri net elements into PNML.
011 * @author Tim Steinbach
012 */
013class PetrinetToMarkup {
014
015    /**
016     * PNML document.
017     */
018    private final Document doc;
019
020    /**
021     * Create new PNML parser for a document.
022     * @param doc PNML document
023     */
024    PetrinetToMarkup(final Document doc) {
025
026        this.doc = doc;
027
028    }
029
030    /**
031     * Get PNML document.
032     * @return PNML document
033     */
034    Document getDoc() {
035
036        return this.doc;
037
038    }
039
040    /**
041     * Create a root element for this document.
042     * @return Root element
043     */
044    Element createRoot() {
045
046        final Element root = this.doc.createElement(PnmlElements.ROOT);
047        this.doc.appendChild(root);
048
049        return root;
050
051    }
052
053    /**
054     * Add a Petri net element to a given root element.
055     * @param root Root element
056     * @param id Petri net ID
057     * @return Net element
058     */
059    Element addNet(final Element root, final String id) {
060
061        final Element net = this.doc.createElement(PnmlElements.NET);
062
063        net.setAttribute(PnmlElements.NET_TYPE, PnmlElements.NET_TYPE_VALUE);
064        net.setAttribute(PnmlElements.NET_ID, id);
065
066        root.appendChild(net);
067
068        return net;
069
070    }
071
072    /**
073     * Add a transition to a net element.
074     * @param net Net element
075     * @param id Transition ID
076     * @return Transition element
077     */
078    Element addTransition(final Element net, final String id) {
079
080        final Element transition =
081            this.doc.createElement(PnmlElements.TRANSITION);
082
083        transition.setAttribute(PnmlElements.TRANSITION_ID, id);
084
085        net.appendChild(transition);
086
087        return transition;
088
089    }
090
091    /**
092     * Add graphics information for Petri object.
093     * @param petriObject Petri object
094     * @param position Position on the canvas
095     * @param size Size
096     */
097    void addGraphics(final Element petriObject, final Point position,
098        final Dimension size) {
099
100        final Element graphics =
101            this.doc.createElement(PnmlElements.GRAPHICS);
102
103        final Element pos = this.doc.createElement(PnmlElements.POSITION);
104
105        pos.setAttribute(PnmlElements.POSITION_X,
106            String.valueOf(position.x));
107        pos.setAttribute(PnmlElements.POSITION_Y,
108            String.valueOf(position.y));
109
110        final Element dim = this.doc.createElement(PnmlElements.DIMENSION);
111
112        dim.setAttribute(PnmlElements.DIMENSION_HEIGHT,
113            String.valueOf(size.height));
114        dim.setAttribute(PnmlElements.DIMENSION_WIDTH,
115            String.valueOf(size.width));
116
117        graphics.appendChild(pos);
118        graphics.appendChild(dim);
119
120        petriObject.appendChild(graphics);
121
122    }
123
124    /**
125     * Add a place to a net element.
126     * @param net Net element
127     * @param id Place ID
128     * @return Place element
129     */
130    Element addPlace(final Element net, final String id) {
131
132        final Element place = this.doc.createElement(PnmlElements.PLACE);
133
134        place.setAttribute(PnmlElements.PLACE_ID, id);
135
136        net.appendChild(place);
137
138        return place;
139
140    }
141
142    /**
143     * Add edge to a net element.
144     * @param net Net element
145     * @param id Edge ID
146     * @param sourceId ID for source Petri object
147     * @param targetId ID for target Petri object
148     */
149    void addEdge(final Element net, final String id, final String sourceId,
150        final String targetId) {
151
152        final Element edge = this.doc.createElement(PnmlElements.EDGE);
153
154        edge.setAttribute(PnmlElements.EDGE_ID, id);
155        edge.setAttribute(PnmlElements.EDGE_SOURCE, sourceId);
156        edge.setAttribute(PnmlElements.EDGE_TARGET, targetId);
157
158        net.appendChild(edge);
159
160    }
161
162}
