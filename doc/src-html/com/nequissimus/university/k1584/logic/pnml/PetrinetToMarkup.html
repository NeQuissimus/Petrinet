001package com.nequissimus.university.k1584.logic.pnml;

003import static com.nequissimus.university.k1584.logic.pnml.PnmlElements.DIMENSION;
004import static com.nequissimus.university.k1584.logic.pnml.PnmlElements.DIMENSION_HEIGHT;
005import static com.nequissimus.university.k1584.logic.pnml.PnmlElements.DIMENSION_WIDTH;
006import static com.nequissimus.university.k1584.logic.pnml.PnmlElements.EDGE;
007import static com.nequissimus.university.k1584.logic.pnml.PnmlElements.EDGE_ID;
008import static com.nequissimus.university.k1584.logic.pnml.PnmlElements.EDGE_SOURCE;
009import static com.nequissimus.university.k1584.logic.pnml.PnmlElements.EDGE_TARGET;
010import static com.nequissimus.university.k1584.logic.pnml.PnmlElements.GRAPHICS;
011import static com.nequissimus.university.k1584.logic.pnml.PnmlElements.NET_ID;
012import static com.nequissimus.university.k1584.logic.pnml.PnmlElements.NET_TYPE;
013import static com.nequissimus.university.k1584.logic.pnml.PnmlElements.NET_TYPE_VALUE;
014import static com.nequissimus.university.k1584.logic.pnml.PnmlElements.PLACE;
015import static com.nequissimus.university.k1584.logic.pnml.PnmlElements.PLACE_ID;
016import static com.nequissimus.university.k1584.logic.pnml.PnmlElements.POSITION;
017import static com.nequissimus.university.k1584.logic.pnml.PnmlElements.POSITION_X;
018import static com.nequissimus.university.k1584.logic.pnml.PnmlElements.POSITION_Y;
019import static com.nequissimus.university.k1584.logic.pnml.PnmlElements.ROOT;
020import static com.nequissimus.university.k1584.logic.pnml.PnmlElements.TRANSITION;
021import static com.nequissimus.university.k1584.logic.pnml.PnmlElements.TRANSITION_ID;
022
023import java.awt.Dimension;
024import java.awt.Point;
025
026import org.w3c.dom.Document;
027import org.w3c.dom.Element;
028
029class PetrinetToMarkup {
030
031    private final Document doc;
032
033    PetrinetToMarkup(final Document doc) {
034
035        this.doc = doc;
036
037    }
038
039    Document getDoc() {
040
041        return this.doc;
042
043    }
044
045    Element createRoot() {
046
047        final Element root = this.doc.createElement(ROOT);
048        this.doc.appendChild(root);
049
050        return root;
051
052    }
053
054    Element addNet(final Element root, final String id) {
055
056        final Element net = this.doc.createElement(PnmlElements.NET);
057
058        net.setAttribute(NET_TYPE, NET_TYPE_VALUE);
059        net.setAttribute(NET_ID, id);
060
061        root.appendChild(net);
062
063        return net;
064
065    }
066
067    Element addTransition(final Element net, final String id) {
068
069        final Element transition = this.doc.createElement(TRANSITION);
070
071        transition.setAttribute(TRANSITION_ID, id);
072
073        net.appendChild(transition);
074
075        return transition;
076
077    }
078
079    void addGraphics(final Element petriObject, final Point position,
080        final Dimension size) {
081
082        final Element graphics = this.doc.createElement(GRAPHICS);
083
084        final Element pos = this.doc.createElement(POSITION);
085
086        pos.setAttribute(POSITION_X, String.valueOf(position.x));
087        pos.setAttribute(POSITION_Y, String.valueOf(position.y));
088
089        final Element dim = this.doc.createElement(DIMENSION);
090
091        dim.setAttribute(DIMENSION_HEIGHT, String.valueOf(size.height));
092        dim.setAttribute(DIMENSION_WIDTH, String.valueOf(size.width));
093
094        graphics.appendChild(pos);
095        graphics.appendChild(dim);
096
097        petriObject.appendChild(graphics);
098
099    }
100
101    Element addPlace(final Element net, final String id) {
102
103        final Element place = this.doc.createElement(PLACE);
104
105        place.setAttribute(PLACE_ID, id);
106
107        net.appendChild(place);
108
109        return place;
110
111    }
112
113    void addEdge(final Element net, final String id, final String sourceId,
114        final String targetId) {
115
116        final Element edge = this.doc.createElement(EDGE);
117
118        edge.setAttribute(EDGE_ID, id);
119        edge.setAttribute(EDGE_SOURCE, sourceId);
120        edge.setAttribute(EDGE_TARGET, targetId);
121
122        net.appendChild(edge);
123
124    }
125
126}
