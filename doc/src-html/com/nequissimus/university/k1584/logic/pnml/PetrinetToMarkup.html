032package com.nequissimus.university.k1584.logic.pnml;
033
034import java.awt.Dimension;
035import java.awt.Point;
036
037import org.w3c.dom.Document;
038import org.w3c.dom.Element;
039
040/**
041 * Class that can turn Petri net elements into PNML.
042 * @author Tim Steinbach
043 */
044class PetrinetToMarkup {
045
046    /**
047     * PNML document.
048     */
049    private final Document doc;
050
051    /**
052     * Create new PNML parser for a document.
053     * @param doc PNML document
054     */
055    PetrinetToMarkup(final Document doc) {
056
057        this.doc = doc;
058
059    }
060
061    /**
062     * Add edge to a net element.
063     * @param net Net element
064     * @param id Edge ID
065     * @param sourceId ID for source Petri object
066     * @param targetId ID for target Petri object
067     */
068    void addEdge(final Element net, final String id, final String sourceId,
069        final String targetId) {
070
071        final Element edge = this.doc.createElement(PnmlElements.EDGE);
072
073        edge.setAttribute(PnmlElements.EDGE_ID, id);
074        edge.setAttribute(PnmlElements.EDGE_SOURCE, sourceId);
075        edge.setAttribute(PnmlElements.EDGE_TARGET, targetId);
076
077        net.appendChild(edge);
078
079    }
080
081    /**
082     * Add graphics information for Petri object.
083     * @param petriObject Petri object
084     * @param position Position on the canvas
085     * @param size Size
086     */
087    void addGraphics(final Element petriObject, final Point position,
088        final Dimension size) {
089
090        final Element graphics =
091            this.doc.createElement(PnmlElements.GRAPHICS);
092
093        final Element pos = this.doc.createElement(PnmlElements.POSITION);
094
095        pos.setAttribute(PnmlElements.POSITION_X,
096            String.valueOf(position.x));
097        pos.setAttribute(PnmlElements.POSITION_Y,
098            String.valueOf(position.y));
099
100        final Element dim = this.doc.createElement(PnmlElements.DIMENSION);
101
102        dim.setAttribute(PnmlElements.DIMENSION_HEIGHT,
103            String.valueOf(size.height));
104        dim.setAttribute(PnmlElements.DIMENSION_WIDTH,
105            String.valueOf(size.width));
106
107        graphics.appendChild(pos);
108        graphics.appendChild(dim);
109
110        petriObject.appendChild(graphics);
111
112    }
113
114    /**
115     * Add a place to a marking element.
116     * @param marking Marking to add to
117     * @param placeId Id of place
118     * @param tokens Number of tokens
119     */
120    void addMarkingPlace(final Element marking, final String placeId,
121        final int tokens) {
122
123        final Element place = this.doc.createElement(PnmlElements.MPLACE);
124        place.setAttribute(PnmlElements.MPLACE_ID, placeId);
125        place.setAttribute(PnmlElements.MPLACE_TOKENS,
126            String.valueOf(tokens));
127
128        marking.appendChild(place);
129
130    }
131
132    /**
133     * Add a Petri net element to a given root element.
134     * @param root Root element
135     * @param id Petri net ID
136     * @return Net element
137     */
138    Element addNet(final Element root, final String id) {
139
140        final Element net = this.doc.createElement(PnmlElements.NET);
141
142        net.setAttribute(PnmlElements.NET_TYPE, PnmlElements.NET_TYPE_VALUE);
143        net.setAttribute(PnmlElements.NET_ID, id);
144
145        root.appendChild(net);
146
147        return net;
148
149    }
150
151    /**
152     * Add a place to a net element.
153     * @param net Net element
154     * @param id Place ID
155     * @param markings Value of markings
156     * @return Place element
157     */
158    Element
159        addPlace(final Element net, final String id, final int markings) {
160
161        final Element place = this.doc.createElement(PnmlElements.PLACE);
162
163        place.setAttribute(PnmlElements.PLACE_ID, id);
164
165        final Element marking =
166            this.doc.createElement(PnmlElements.PLACE_INITIAL_MARKING);
167        final Element value =
168            this.doc.createElement(PnmlElements.PLACE_INITIAL_MARKING_TEXT);
169        value.setTextContent(String.valueOf(markings));
170
171        marking.appendChild(value);
172        place.appendChild(marking);
173
174        net.appendChild(place);
175
176        return place;
177
178    }
179
180    /**
181     * Add a transition to a net element.
182     * @param net Net element
183     * @param id Transition ID
184     * @return Transition element
185     */
186    Element addTransition(final Element net, final String id) {
187
188        final Element transition =
189            this.doc.createElement(PnmlElements.TRANSITION);
190
191        transition.setAttribute(PnmlElements.TRANSITION_ID, id);
192
193        net.appendChild(transition);
194
195        return transition;
196
197    }
198
199    /**
200     * Create a new marking node.
201     * @param id Marking id
202     * @param name Marking name
203     * @return Marking node element
204     */
205    Element createMarking(final String id, final String name) {
206
207        final Element marking =
208            this.doc.createElement(PnmlElements.MARKING);
209        marking.setAttribute(PnmlElements.MARKING_ID, id);
210        marking.setAttribute(PnmlElements.MARKING_NAME, name);
211
212        return marking;
213
214    }
215
216    /**
217     * Create a new markings root node.
218     * @return Markings root node
219     */
220    Element createMarkingsRoot() {
221
222        final Element markingsRoot =
223            this.doc.createElement(PnmlElements.MARKINGS);
224
225        return markingsRoot;
226
227    }
228
229    /**
230     * Create a root element for this document.
231     * @return Root element
232     */
233    Element createRoot() {
234
235        final Element root = this.doc.createElement(PnmlElements.ROOT);
236        this.doc.appendChild(root);
237
238        return root;
239
240    }
241
242    /**
243     * Get PNML document.
244     * @return PNML document
245     */
246    Document getDoc() {
247
248        return this.doc;
249
250    }
251
252}
