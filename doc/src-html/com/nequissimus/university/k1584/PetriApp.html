001package com.nequissimus.university.k1584;
002
003import java.awt.EventQueue;
004
005import com.nequissimus.university.k1584.logic.PetriConfig;
006
007/**
008 * This is the application launcher which causes the editor to be created and
009 * started.
010 * @author Tim Steinbach
011 */
012public final class PetriApp {
013
014    /**
015     * Configuration.
016     */
017    private static final PetriConfig CONFIG = PetriConfig.getInstance();
018
019    /**
020     * Hide constructor.
021     */
022    private PetriApp() {
023    }
024
025    /**
026     * Point of entry for executing the application.
027     * @param args Command line arguments
028     */
029    public static void main(final String[] args) {
030
031        // This is needed for Mac OS X only - It changes the application name in
032        // the menu bar
033        // from the fully-qualified class name to the given value
034        System.setProperty(
035            "com.apple.mrj.application.apple.menu.about.name",
036            PetriApp.CONFIG.getApplicationName());
037
038        EventQueue.invokeLater(PetriController.getInstance());
039
040    }
041
042}
