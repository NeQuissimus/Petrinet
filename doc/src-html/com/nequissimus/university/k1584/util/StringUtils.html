001package com.nequissimus.university.k1584.util;
002
003/**
004 * Class that provides common methods when working with {@link String}.
005 * @author Tim Steinbach
006 */
007public final class StringUtils {
008
009    /**
010     * Hide constructor.
011     */
012    private StringUtils() {
013    }
014
015    /**
016     * This method capitalizes the first letter of the word and turns the rest
017     * of it into lower case.<br />
018     * It also replaces underscores (_) with spaces. E.g.: fOo becomes Foo, foo
019     * becomes Foo, Foo stays Foo, F_oo becomes F oo.
020     * @param word Word to transform
021     * @return Word in name form
022     */
023    public static String asName(final String word) {
024
025        final StringBuffer result = new StringBuffer();
026
027        if (null != word) {
028
029            final char first = Character.toUpperCase(word.charAt(0));
030
031            // Make sure first is not a symbol
032            if (Character.isUpperCase(first)) {
033
034                final String withSpaces =
035                    word.substring(1).toLowerCase().replace("_", " ");
036
037                result.append(first).append(withSpaces);
038
039            }
040
041        }
042
043        return result.toString();
044
045    }
046}
