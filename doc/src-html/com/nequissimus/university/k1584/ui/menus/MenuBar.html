001package com.nequissimus.university.k1584.ui.menus;
002
003import javax.swing.JMenu;
004import javax.swing.JMenuBar;
005import javax.swing.JMenuItem;
006
007import com.nequissimus.university.k1584.ui.MessagePool;
008import com.nequissimus.university.k1584.ui.actions.CanvasLargerAction;
009import com.nequissimus.university.k1584.ui.actions.CanvasSmallerAction;
010import com.nequissimus.university.k1584.ui.actions.ChangeIconSizeAction;
011import com.nequissimus.university.k1584.ui.actions.CloseWindowAction;
012import com.nequissimus.university.k1584.ui.actions.SaveFileAction;
013import com.nequissimus.university.k1584.ui.enums.IconSize;
014
015/**
016 * Menu bar for the application window. This menu bar automatically provides all
017 * essential options, such as "save", "close", "load" etc.
018 * @author Tim Steinbach
019 */
020public class MenuBar extends JMenuBar {
021
022    /**
023     * Serializable UID.
024     */
025    private static final long serialVersionUID = -415488481991662662L;
026
027    /**
028     * Message pool.
029     */
030    private static final MessagePool MSG = MessagePool.getInstance();
031
032    /**
033     * Create a new menu bar instance for a given window.
034     */
035    public MenuBar() {
036
037        super();
038
039        this.resetMenu();
040
041    }
042
043    /**
044     * Add all essential menu items to the menu bar.
045     */
046    public final void resetMenu() {
047
048        this.resetFileMenu();
049        this.resetViewMenu();
050
051    }
052
053    /**
054     * Add all items found under "file".
055     */
056    private void resetFileMenu() {
057
058        final JMenu menu = new JMenu(MenuBar.MSG.getMenuFile());
059
060        JMenuItem item = new JMenuItem(MenuBar.MSG.getMenuFileSave());
061        item.addActionListener(new SaveFileAction());
062        menu.add(item);
063
064        item = new JMenuItem(MenuBar.MSG.getMenuFileClose());
065        item.addActionListener(new CloseWindowAction());
066        menu.add(item);
067
068        this.add(menu);
069
070    }
071
072    /**
073     * Add all items found under "view".
074     */
075    private void resetViewMenu() {
076
077        final JMenu menu = new JMenu(MenuBar.MSG.getMenuView());
078
079        this.resetIconSizeViewMenu(menu);
080        this.resetCanvasSizeViewMenu(menu);
081
082        this.add(menu);
083
084    }
085
086    /**
087     * Add all items found under "View"-&gt;"Canvas Size".
088     * @param menu View menu
089     */
090    private void resetCanvasSizeViewMenu(final JMenu menu) {
091
092        final JMenu subMenu =
093            new JMenu(MenuBar.MSG.getMenuViewCanvasSize());
094
095        final JMenuItem itemSmaller =
096            new JMenuItem(MenuBar.MSG.getMenuViewCanvasSizeSmaller());
097        final JMenuItem itemLarger =
098            new JMenuItem(MenuBar.MSG.getMenuViewCanvasSizeLarger());
099
100        itemSmaller.addActionListener(new CanvasSmallerAction());
101        itemLarger.addActionListener(new CanvasLargerAction());
102
103        subMenu.add(itemSmaller);
104        subMenu.add(itemLarger);
105
106        menu.add(subMenu);
107
108    }
109
110    /**
111     * Add all items found under "View"-&gt;"Icon Size".
112     * @param menu View menu to add to
113     */
114    private void resetIconSizeViewMenu(final JMenu menu) {
115
116        final JMenu subMenu = new JMenu(MenuBar.MSG.getMenuViewIconSize());
117
118        JMenuItem item =
119            new JMenuItem(MenuBar.MSG.getMenuViewIconSizeLarge());
120        item.addActionListener(new ChangeIconSizeAction(IconSize.LARGE));
121        subMenu.add(item);
122
123        item = new JMenuItem(MenuBar.MSG.getMenuViewIconSizeMedium());
124        item.addActionListener(new ChangeIconSizeAction(IconSize.MEDIUM));
125        subMenu.add(item);
126
127        item = new JMenuItem(MenuBar.MSG.getMenuViewIconSizeSmall());
128        item.addActionListener(new ChangeIconSizeAction(IconSize.SMALL));
129        subMenu.add(item);
130
131        item = new JMenuItem(MenuBar.MSG.getMenuViewIconSizeVerySmall());
132        item.addActionListener(new ChangeIconSizeAction(IconSize.VERY_SMALL));
133        subMenu.add(item);
134
135        menu.add(subMenu);
136
137    }
138
139}
