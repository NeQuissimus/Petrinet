001package com.nequissimus.university.k1584.ui.menus;
002
003import javax.swing.JMenuItem;
004
005import com.nequissimus.university.k1584.ui.MessagePool;
006import com.nequissimus.university.k1584.ui.actions.DecreaseMarkingsAction;
007import com.nequissimus.university.k1584.ui.actions.IncreaseMarkingsAction;
008import com.nequissimus.university.k1584.ui.elements.PlaceLabel;
009
010/**
011 * Context menu for Petri place icons on the canvas.
012 * @author Tim Steinbach
013 */
014public final class CanvasPlaceMenu extends CanvasIconMenu {
015
016    /**
017     * Serializable UID.
018     */
019    private static final long serialVersionUID = 6468863481232638652L;
020
021    /**
022     * Message pool.
023     */
024    private static final MessagePool MSG = MessagePool.getInstance();
025
026    /**
027     * Create a new menu for places.
028     * @param invoker Invoking place
029     */
030    public CanvasPlaceMenu(final PlaceLabel invoker) {
031        super(invoker);
032    }
033
034    @Override
035    void addCustomItems() {
036
037        JMenuItem item =
038            new JMenuItem(CanvasPlaceMenu.MSG.getIconMenuIncreaseMarkings());
039        item.addActionListener(new IncreaseMarkingsAction((PlaceLabel) this
040            .getPetriLabel()));
041        this.add(item);
042
043        item =
044            new JMenuItem(CanvasPlaceMenu.MSG.getIconMenuDecreaseMarkings());
045        item.addActionListener(new DecreaseMarkingsAction((PlaceLabel) this
046            .getPetriLabel()));
047        this.add(item);
048
049    }
050
051}
