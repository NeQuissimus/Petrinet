001package com.nequissimus.university.k1584.ui.menus;
002
003import javax.swing.JPopupMenu;
004
005import com.nequissimus.university.k1584.ui.elements.AbstractLabel;
006
007/**
008 * Abstract menu that defines all methods for context menus.
009 * @author Tim Steinbach
010 */
011public abstract class AbstractContextMenu extends JPopupMenu {
012
013    /**
014     * Serializable UID.
015     */
016    private static final long serialVersionUID = -7674897734685610675L;
017
018    /**
019     * Invoking component.
020     */
021    private final AbstractLabel invoker;
022
023    /**
024     * Create a new context menu for a label.
025     * @param invoker Label
026     */
027    public AbstractContextMenu(final AbstractLabel invoker) {
028
029        super();
030        this.invoker = invoker;
031
032        this.setupMenu();
033
034    }
035
036    /**
037     * Return invoking Petri label.
038     * @return Petri label
039     */
040    final AbstractLabel getPetriLabel() {
041
042        return this.invoker;
043
044    }
045
046    /**
047     * Add menu items and actions.
048     */
049    abstract void setupMenu();
050
051}
