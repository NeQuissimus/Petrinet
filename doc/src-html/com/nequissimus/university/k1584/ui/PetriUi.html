001package com.nequissimus.university.k1584.ui;
002
003import java.awt.Dimension;
004import java.awt.Point;
005import java.util.Map;
006
007import javax.print.attribute.standard.Severity;
008
009import com.nequissimus.university.k1584.ui.elements.AbstractLabel;
010import com.nequissimus.university.k1584.ui.elements.Arrow;
011import com.nequissimus.university.k1584.ui.elements.Canvas;
012import com.nequissimus.university.k1584.ui.elements.PlaceLabel;
013import com.nequissimus.university.k1584.ui.elements.TransitionLabel;
014import com.nequissimus.university.k1584.ui.elements.Window;
015import com.nequissimus.university.k1584.ui.enums.IconSize;
016
017/**
018 * Interface providing all methods needed to work with the Petri net UI.
019 * @author Tim Steinbach
020 */
021public interface PetriUi {
022
023    /**
024     * Add an arrow to the UI.
025     * @param arrow Arrow to add
026     */
027    void addArrow(final Arrow arrow);
028
029    /**
030     * Add a new Petri place to the UI and return the label component.
031     * @return Label component
032     */
033    PlaceLabel addPlace();
034
035    /**
036     * Add a new Petri place to the UI with a given name.
037     * @param name Label name
038     * @return Label component
039     */
040    PlaceLabel addPlace(final String name);
041
042    /**
043     * Add a new Petri transition to the UI ad return the label component.
044     * @return Label component
045     */
046    TransitionLabel addTransition();
047
048    /**
049     * Add a new transition to the UI with a given name.
050     * @param name Label name
051     * @return Label component
052     */
053    TransitionLabel addTransition(final String name);
054
055    /**
056     * Calculate a new canvas size.<br />
057     * Canvas sizes are calculated in 500px increments and always square.<br />
058     * If the given minimum size was (1600, 2100), the calculated size would be
059     * (2500, 2500). If the minimum was (1600, 1600), the result would be (2000,
060     * 2000).
061     * @param minimum Minimum canvas size
062     * @return New canvas size
063     */
064    Dimension calculateCanvasSize(final Dimension minimum);
065
066    /**
067     * Clean the UI, remove all components.
068     */
069    void clean();
070
071    /**
072     * Decrease the number of markings on the place.
073     * @param label Place
074     */
075    void decreaseMarkings(final PlaceLabel label);
076
077    /**
078     * Get the main canvas.
079     * @return Editor canvas
080     */
081    Canvas getCanvas();
082
083    /**
084     * Return the editor canvas' size.
085     * @return Canvas size
086     */
087    Dimension getCanvasSize();
088
089    /**
090     * Get the size of the icons used for displaying the Petri net components.
091     * @return Icon size
092     */
093    IconSize getIconSize();
094
095    /**
096     * Get the minimum size needed to display all components on the canvas.
097     * @return Minimum size
098     */
099    Dimension getMinCanvasSize();
100
101    /**
102     * Get the application window.
103     * @return Application window
104     */
105    Window getWindow();
106
107    /**
108     * Hide the application window.
109     */
110    void hideWindow();
111
112    /**
113     * Highlight a label component by setting a background colour.
114     * @param label Label to highlight
115     */
116    void highlightLabel(final AbstractLabel label);
117
118    /**
119     * Increase the number of markings on the place.
120     * @param label Place
121     */
122    void increaseMarkings(final PlaceLabel label);
123
124    /**
125     * Change a transition's UI whether it is active or not.
126     * @param label Transition
127     * @param active Active or not
128     */
129    void markTransitionActive(final TransitionLabel label,
130        final boolean active);
131
132    /**
133     * Move a label component to a new location.
134     * @param label Label component
135     * @param location New location
136     */
137    void moveLabel(final AbstractLabel label, final Point location);
138
139    /**
140     * Redraw the canvas.<br />
141     * This method should take care of the entire process, not just call
142     * {@link java.awt.Component#repaint}.
143     */
144    void redrawCanvas();
145
146    /**
147     * Remove an arrow from the UI.
148     * @param arrow Arrow to remove
149     */
150    void removeArrow(final Arrow arrow);
151
152    /**
153     * Remove a label component representing a Petri object.
154     * @param label Label component
155     */
156    void removeLabel(final AbstractLabel label);
157
158    /**
159     * Rename a label component.
160     * @param label Label to be renamed
161     * @param name New name
162     */
163    void renameLabel(final AbstractLabel label, final String name);
164
165    /**
166     * Report a message to the user by opening an appropriate dialog.
167     * @param severity Message severity
168     * @param message Message text
169     */
170    void reportMessage(final Severity severity, final String message);
171
172    /**
173     * Resize all the arrow components' canvases.
174     * @param size New size
175     */
176    void resizeArrowCanvas(final Dimension size);
177
178    /**
179     * Resize the editor canvas to a new size.<br />
180     * The canvas can only be resized to a size larger than its
181     * {@link java.awt.Component#getPreferredSize()}, so that its components
182     * will be visible after resizing.
183     * @param size New size
184     * @return Whether the canvas has been resized
185     */
186    boolean resizeCanvas(final Dimension size);
187
188    /**
189     * Resize the canvas by a certain amount of pixels in each direction.
190     * @param difference Number of pixels
191     * @return Whether the canvas has been resized
192     * @see #resizeArrowCanvas(Dimension)
193     */
194    boolean resizeCanvas(final int difference);
195
196    /**
197     * Resize the editor window.
198     * @param size New size
199     */
200    void resizeEditorWindow(final Dimension size);
201
202    /**
203     * Set the canvas to a new size.
204     * @param size New size
205     */
206    void setCanvasSize(final Dimension size);
207
208    /**
209     * Set a new size for all icons representing Petri net components.
210     * @param size New icon size
211     */
212    void setIconSize(final IconSize size);
213
214    /**
215     * Show the application window.
216     */
217    void showWindow();
218
219    /**
220     * Remove the highlighted background from a label.
221     * @param label Label to remove background from
222     */
223    void unhighlightLabel(final AbstractLabel label);
224
225    /**
226     * Update the markings on places, e.g. after a transition has occurred.
227     * @param places Places to update with their new values
228     */
229    void updateMarkings(final Map<PlaceLabel, Integer> places);
230
231}
