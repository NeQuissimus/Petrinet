001package com.nequissimus.university.k1584.ui.actions;
002
003import java.awt.event.ActionEvent;
004import java.awt.event.ActionListener;
005
006import com.nequissimus.university.k1584.PetriController;
007import com.nequissimus.university.k1584.ui.elements.AbstractLabel;
008
009/**
010 * Action used to connect a label with another one.
011 * @author Tim Steinbach
012 */
013public class ConnectArrowAction implements ActionListener {
014
015    /**
016     * Label invoking this action.
017     */
018    private final AbstractLabel invoker;
019
020    /**
021     * Create a new connect action.
022     * @param invoker Label invoking the action
023     */
024    public ConnectArrowAction(final AbstractLabel invoker) {
025        this.invoker = invoker;
026    }
027
028    @Override
029    public final void actionPerformed(final ActionEvent arg0) {
030        PetriController.getInstance().arrowConnect(this.invoker);
031    }
032
033}
