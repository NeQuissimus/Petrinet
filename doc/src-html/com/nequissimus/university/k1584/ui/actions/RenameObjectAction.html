001package com.nequissimus.university.k1584.ui.actions;
002
003import java.awt.Dimension;
004import java.awt.event.ActionEvent;
005import java.awt.event.ActionListener;
006
007import javax.swing.Icon;
008import javax.swing.JOptionPane;
009
010import com.nequissimus.library.util.IconUtil;
011import com.nequissimus.university.k1584.PetriController;
012import com.nequissimus.university.k1584.logic.PetriConfig;
013import com.nequissimus.university.k1584.ui.MessagePool;
014import com.nequissimus.university.k1584.ui.elements.AbstractLabel;
015
016/**
017 * Action that allows the user to rename a given label object.
018 * @author Tim Steinbach
019 */
020public final class RenameObjectAction implements ActionListener {
021
022    /**
023     * Controller.
024     */
025    private static final PetriController CONTROLLER = PetriController
026        .getInstance();
027
028    /**
029     * Configuration.
030     */
031    private static final PetriConfig CONFIG = PetriConfig.getInstance();
032
033    /**
034     * Message pool.
035     */
036    private static final MessagePool MSG = MessagePool.getInstance();
037
038    /**
039     * Invoking object.
040     */
041    private final AbstractLabel invoker;
042
043    /**
044     * Create a new action allowing the user to rename the invoking object.
045     * @param invoker Invoking object
046     */
047    public RenameObjectAction(final AbstractLabel invoker) {
048
049        this.invoker = invoker;
050
051    }
052
053    @Override
054    public void actionPerformed(final ActionEvent arg0) {
055
056        final Icon icon =
057            IconUtil.getIcon(RenameObjectAction.CONFIG.getImageQuestion()
058                .getFile(), new Dimension(100, 100));
059
060        final String newName =
061            (String) JOptionPane.showInputDialog(
062                RenameObjectAction.CONTROLLER.getWindow(),
063                RenameObjectAction.MSG.getIconMenuRenameTitle(),
064                RenameObjectAction.MSG.getIconMenuRenameWindowTitle(),
065                JOptionPane.QUESTION_MESSAGE, icon, null,
066                this.invoker.getText());
067
068        if (newName != null) {
069
070            RenameObjectAction.CONTROLLER
071                .renameLabel(this.invoker, newName);
072
073        }
074
075    }
076
077}
