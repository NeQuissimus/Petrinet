032package com.nequissimus.university.k1584.ui.actions;
033
034import java.awt.event.ActionEvent;
035import java.awt.event.ActionListener;
036
037import javax.swing.JOptionPane;
038
039import com.nequissimus.library.data.Singleton;
040import com.nequissimus.university.k1584.PetriController;
041import com.nequissimus.university.k1584.logic.PetriMarking;
042import com.nequissimus.university.k1584.ui.MessagePool;
043
044/**
045 * Select a marking for the current net.
046 * @author Tim Steinbach
047 */
048public final class SelectMarkingAction implements ActionListener {
049
050    /**
051     * Controller.
052     */
053    private static final PetriController CONTROLLER = PetriController
054        .getInstance();
055
056    /**
057     * Message pool.
058     */
059    private static final MessagePool MSG = Singleton
060        .getObject(MessagePool.class);
061
062    @Override
063    public void actionPerformed(final ActionEvent e) {
064
065        final Object[] markings =
066            SelectMarkingAction.CONTROLLER.getMarkings().toArray();
067
068        final PetriMarking active =
069            SelectMarkingAction.CONTROLLER.getActiveMarking();
070
071        final PetriMarking selected =
072            (PetriMarking) JOptionPane.showInputDialog(
073                SelectMarkingAction.CONTROLLER.getWindow(),
074                SelectMarkingAction.MSG.getSelectMarkingDialog(),
075                SelectMarkingAction.MSG.getSelectMarkingDialogTitle(),
076                JOptionPane.DEFAULT_OPTION, null, markings, active);
077
078        if ((null != selected) && (active != selected)) {
079
080            SelectMarkingAction.CONTROLLER.switchMarking(selected);
081
082        }
083
084    }
085}
