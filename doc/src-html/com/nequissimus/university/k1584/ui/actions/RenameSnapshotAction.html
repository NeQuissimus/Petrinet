001package com.nequissimus.university.k1584.ui.actions;
002
003import java.awt.event.ActionEvent;
004import java.awt.event.ActionListener;
005import java.util.ArrayList;
006import java.util.List;
007
008import javax.print.attribute.standard.Severity;
009import javax.swing.JOptionPane;
010
011import com.nequissimus.university.k1584.PetriController;
012import com.nequissimus.university.k1584.logic.PetriNet;
013import com.nequissimus.university.k1584.ui.MessagePool;
014import com.nequissimus.university.k1584.ui.elements.Window;
015
016/**
017 * Action that lets the user choose the snapshot to be renamed and then opens a
018 * dialog for the user to enter the new name.
019 * @author Tim Steinbach
020 */
021public class RenameSnapshotAction implements ActionListener {
022
023    /**
024     * Controller.
025     */
026    private static final PetriController CONTROLLER = PetriController
027        .getInstance();
028
029    /**
030     * Message pool.
031     */
032    private static final MessagePool MSG = MessagePool.getInstance();
033
034    @Override
035    public final void actionPerformed(final ActionEvent e) {
036
037        final Window window = RenameSnapshotAction.CONTROLLER.getWindow();
038
039        final Object[] snapshots =
040            RenameSnapshotAction.CONTROLLER.getSnapshots().toArray();
041
042        final PetriNet active =
043            RenameSnapshotAction.CONTROLLER.getActiveLogic();
044
045        final PetriNet selected =
046            (PetriNet) JOptionPane.showInputDialog(window,
047                RenameSnapshotAction.MSG.getSnapshotSelectDialog(),
048                RenameSnapshotAction.MSG.getSnapshotRename(),
049                JOptionPane.DEFAULT_OPTION, null, snapshots, active);
050
051        if (null != selected) {
052
053            final List<String> takenNames = this.getTakenNames();
054
055            boolean foundOrCancelled = false;
056
057            final String oldName = selected.getName();
058
059            while (!foundOrCancelled) {
060
061                final String newName =
062                    (String) JOptionPane.showInputDialog(window,
063                        RenameSnapshotAction.MSG.getSnapshotRenameDialog(),
064                        RenameSnapshotAction.MSG.getSnapshotRename(),
065                        JOptionPane.QUESTION_MESSAGE, null, null, oldName);
066
067                if ((null != newName) && (takenNames.contains(newName))) {
068
069                    RenameSnapshotAction.CONTROLLER.reportMessage(
070                        Severity.REPORT, RenameSnapshotAction.MSG
071                            .getMsgErrorSnapshotNameTaken());
072
073                } else {
074
075                    foundOrCancelled = true;
076
077                    if (null != newName) {
078                        selected.setName(newName);
079                    }
080
081                }
082
083            }
084
085        }
086
087    }
088
089    /**
090     * Get a list of all snapshot names that have been taken so far.
091     * @return List of used names
092     */
093    private List<String> getTakenNames() {
094
095        final List<PetriNet> nets =
096            RenameSnapshotAction.CONTROLLER.getSnapshots();
097        final List<String> names = new ArrayList<String>();
098
099        for (final PetriNet petriNet : nets) {
100
101            names.add(petriNet.getName());
102
103        }
104
105        return names;
106
107    }
108
109}
