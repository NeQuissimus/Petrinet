001package com.nequissimus.university.k1584.ui.actions;
002
003import java.awt.Dimension;
004import java.awt.event.ActionEvent;
005import java.awt.event.ActionListener;
006import java.util.ArrayList;
007import java.util.List;
008
009import javax.print.attribute.standard.Severity;
010import javax.swing.Icon;
011import javax.swing.JOptionPane;
012
013import com.nequissimus.library.util.IconUtil;
014import com.nequissimus.university.k1584.PetriController;
015import com.nequissimus.university.k1584.logic.PetriConfig;
016import com.nequissimus.university.k1584.logic.PetriNet;
017import com.nequissimus.university.k1584.ui.MessagePool;
018
019/**
020 * Action for creating a new snapshot. Snapshots have to have unique names.
021 * @author Tim Steinbach
022 */
023public final class CreateSnapshotAction implements ActionListener {
024
025    /**
026     * Controller.
027     */
028    private static final PetriController CONTROLLER = PetriController
029        .getInstance();
030
031    /**
032     * Configuration.
033     */
034    private static final PetriConfig CONFIG = PetriConfig.getInstance();
035
036    /**
037     * Message pool.
038     */
039    private static final MessagePool MSG = MessagePool.getInstance();
040
041    @Override
042    public void actionPerformed(final ActionEvent e) {
043
044        final List<String> usedNames = this.getTakenNames();
045
046        final Icon icon =
047            IconUtil.getIcon(CreateSnapshotAction.CONFIG.getImageQuestion()
048                .getFile(), new Dimension(100, 100));
049
050        boolean foundOrCancelled = false;
051
052        String newName = null;
053
054        while (!foundOrCancelled) {
055
056            newName =
057                (String) JOptionPane.showInputDialog(
058                    CreateSnapshotAction.CONTROLLER.getWindow(),
059                    CreateSnapshotAction.MSG.getSnapshotRenameDialog(),
060                    CreateSnapshotAction.MSG.getSnapshotCreate(),
061                    JOptionPane.QUESTION_MESSAGE, icon, null,
062                    CreateSnapshotAction.CONFIG.getNetName());
063
064            if ((null != newName) && (usedNames.contains(newName))) {
065
066                CreateSnapshotAction.CONTROLLER
067                    .reportMessage(Severity.REPORT,
068                        CreateSnapshotAction.MSG
069                            .getMsgErrorSnapshotNameTaken());
070
071            } else {
072
073                foundOrCancelled = true;
074
075            }
076
077        }
078
079        if (null != newName) {
080
081            CreateSnapshotAction.CONTROLLER.createSnapshot(newName);
082
083        }
084
085    }
086
087    /**
088     * Get a list of all snapshot names that have been taken so far.
089     * @return List of used names
090     */
091    private List<String> getTakenNames() {
092
093        final List<PetriNet> nets =
094            CreateSnapshotAction.CONTROLLER.getSnapshots();
095        final List<String> names = new ArrayList<String>();
096
097        for (final PetriNet petriNet : nets) {
098
099            names.add(petriNet.getName());
100
101        }
102
103        return names;
104
105    }
106
107}
