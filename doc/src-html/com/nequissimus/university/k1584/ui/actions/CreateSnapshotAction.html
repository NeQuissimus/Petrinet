032package com.nequissimus.university.k1584.ui.actions;
033
034import java.awt.Dimension;
035import java.awt.event.ActionEvent;
036import java.awt.event.ActionListener;
037import java.util.ArrayList;
038import java.util.List;
039
040import javax.print.attribute.standard.Severity;
041import javax.swing.Icon;
042import javax.swing.JOptionPane;
043
044import com.nequissimus.library.data.Singleton;
045import com.nequissimus.library.util.IconUtil;
046import com.nequissimus.university.k1584.PetriController;
047import com.nequissimus.university.k1584.logic.PetriConfig;
048import com.nequissimus.university.k1584.logic.PetriNet;
049import com.nequissimus.university.k1584.ui.MessagePool;
050
051/**
052 * Action for creating a new snapshot. Snapshots have to have unique names.
053 * @author Tim Steinbach
054 */
055public final class CreateSnapshotAction implements ActionListener {
056
057    /**
058     * Configuration.
059     */
060    private static final PetriConfig CONFIG = Singleton
061        .getObject(PetriConfig.class);
062
063    /**
064     * Controller.
065     */
066    private static final PetriController CONTROLLER = PetriController
067        .getInstance();
068
069    /**
070     * Message pool.
071     */
072    private static final MessagePool MSG = Singleton
073        .getObject(MessagePool.class);
074
075    @Override
076    public void actionPerformed(final ActionEvent e) {
077
078        final List<String> usedNames = this.getTakenNames();
079
080        final Icon icon =
081            IconUtil.getIcon(CreateSnapshotAction.CONFIG.getImageQuestion()
082                .getFile(), new Dimension(100, 100));
083
084        boolean foundOrCancelled = false;
085
086        String newName = null;
087
088        while (!foundOrCancelled) {
089
090            newName =
091                (String) JOptionPane.showInputDialog(
092                    CreateSnapshotAction.CONTROLLER.getWindow(),
093                    CreateSnapshotAction.MSG.getSnapshotRenameDialog(),
094                    CreateSnapshotAction.MSG.getSnapshotCreate(),
095                    JOptionPane.QUESTION_MESSAGE, icon, null,
096                    CreateSnapshotAction.CONFIG.getNetName());
097
098            if ((null != newName) && (usedNames.contains(newName))) {
099
100                CreateSnapshotAction.CONTROLLER
101                    .reportMessage(Severity.REPORT,
102                        CreateSnapshotAction.MSG
103                            .getMsgErrorSnapshotNameTaken());
104
105            } else {
106
107                foundOrCancelled = true;
108
109            }
110
111        }
112
113        if (null != newName) {
114
115            CreateSnapshotAction.CONTROLLER.createSnapshot(newName);
116
117        }
118
119    }
120
121    /**
122     * Get a list of all snapshot names that have been taken so far.
123     * @return List of used names
124     */
125    private List<String> getTakenNames() {
126
127        final List<PetriNet> nets =
128            CreateSnapshotAction.CONTROLLER.getSnapshots();
129        final List<String> names = new ArrayList<String>();
130
131        for (final PetriNet petriNet : nets) {
132
133            names.add(petriNet.getName());
134
135        }
136
137        return names;
138
139    }
140
141}
