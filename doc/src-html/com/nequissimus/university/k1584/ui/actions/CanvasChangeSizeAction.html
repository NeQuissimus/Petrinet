032package com.nequissimus.university.k1584.ui.actions;
033
034import java.awt.event.ActionEvent;
035import java.awt.event.ActionListener;
036
037import com.nequissimus.university.k1584.PetriController;
038
039/**
040 * This abstract action helps implement actions that will resize the canvas.
041 * @author Tim Steinbach
042 */
043public abstract class CanvasChangeSizeAction implements ActionListener {
044
045    /**
046     * Controller.
047     */
048    private static final PetriController CONTROLLER = PetriController
049        .getInstance();
050
051    /**
052     * Amount of pixels to change.
053     */
054    private final int change;
055
056    /**
057     * Create a new action that will change the canvas by a number of pixels.
058     * @param change Number of pixels the canvas will be changed
059     */
060    protected CanvasChangeSizeAction(final int change) {
061
062        this.change = change;
063
064    }
065
066    @Override
067    public final void actionPerformed(final ActionEvent e) {
068
069        CanvasChangeSizeAction.CONTROLLER.resizeCanvas(this.change);
070
071    }
072
073}
