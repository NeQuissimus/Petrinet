599    /**
600     * Read all entries from the lang.properties file.
601     */
602    private void readProperties() {
603
604        this.getDefaults();
605
606        InputStream is = null;
607
608        try {
609
610            is = this.getClass().getResourceAsStream("/lang.properties");
611            this.load(is);
612
613        } catch (final Exception e) {
614
615            // If something went wrong with reading the properties file, the
616            // default values
617            // will still be in the needed in the configuration object
618
619            System.out.println("Reading the language file failed");
620
621        } finally {
622
623            if (null != is) {
624                try {
625                    is.close();
626                } catch (final IOException e) {
627                    System.out.println("Could not close config stream");
628                }
629            }
630
631        }
632
633    }
