641    /**
642     * Read all entries from the lang.properties file.
643     */
644    private void readProperties() {
645
646        this.getDefaults();
647
648        InputStream is = null;
649
650        try {
651
652            is = this.getClass().getResourceAsStream("/lang.properties");
653            this.load(is);
654
655        } catch (final Exception e) {
656
657            // If something went wrong with reading the properties file, the
658            // default values
659            // will still be in the needed in the configuration object
660
661            System.out.println("Reading the language file failed");
662
663        } finally {
664
665            if (null != is) {
666                try {
667                    is.close();
668                } catch (final IOException e) {
669                    System.out.println("Could not close config stream");
670                }
671            }
672
673        }
674
675    }
