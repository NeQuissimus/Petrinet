010 * These messages will be read from /lang.properties.<br />

021 * MenuFileClose = "Close" // File menu for closing the application<br />

023 * MenuViewCanvasSize = "Canvas size" // Editor canvas size<br />

156    /**
157     * Read all entries from the lang.properties file.
158     */
159    private void readProperties() {
160
161        this.getDefaults();
162
163        InputStream is = null;
164
165        try {
166
167            is = new FileInputStream("lang.properties");
168            this.load(is);
169
170        } catch (final IOException e) {
171
172            // If something went wrong with reading the properties file, the
173            // default values
174            // will still be in the needed in the configuration object
175
176            System.out.println("Reading the language file failed");
177
178        } finally {
179
180            if (null != is) {
181                try {
182                    is.close();
183                } catch (final IOException e) {
184                    System.out.println("Could not close config stream");
185                }
186            }
187
188        }
189
190    }
