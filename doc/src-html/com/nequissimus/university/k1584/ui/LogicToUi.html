001package com.nequissimus.university.k1584.ui;
002
003import java.awt.Dimension;
004import java.awt.Point;
005import java.util.Set;
006
007import com.nequissimus.university.k1584.PetriController;
008import com.nequissimus.university.k1584.data.BiMap;
009import com.nequissimus.university.k1584.data.TwoKeyMap;
010import com.nequissimus.university.k1584.logic.PetriNet;
011import com.nequissimus.university.k1584.logic.PetriObject;
012import com.nequissimus.university.k1584.logic.PetriPlace;
013import com.nequissimus.university.k1584.logic.PetriTransition;
014import com.nequissimus.university.k1584.ui.elements.AbstractLabel;
015import com.nequissimus.university.k1584.ui.elements.Arrow;
016import com.nequissimus.university.k1584.ui.elements.PlaceLabel;
017import com.nequissimus.university.k1584.ui.elements.TransitionLabel;
018import com.nequissimus.university.k1584.ui.enums.IconSize;
019
020/**
021 * This class takes a logical net and draws all components onto the UI canvas.
022 * @author Tim Steinbach
023 */
024public final class LogicToUi {
025
026    /**
027     * Hide constructor.
028     */
029    private LogicToUi() {
030    }
031
032    /**
033     * Draw a Petri net onto the canvas used by the application's controller.
034     * @param net Petri net
035     */
036    public static void convert(final PetriNet net) {
037
038        // Get all components
039        final PetriController controller = PetriController.getInstance();
040        final PetriUi ui = controller.getUi();
041        final BiMap<PetriObject, AbstractLabel> objects =
042            controller.getObjects();
043        final TwoKeyMap<PlaceLabel, TransitionLabel, Arrow> arrows =
044            controller.getArrows();
045
046        // Clean the controller
047        objects.clear();
048        arrows.clear();
049        controller.arrowConnect(null);
050        controller.arrowDisconnect(null);
051
052        // Clean the canvas
053        ui.clean();
054
055        // Set icon size
056        final PetriPlace firstPlace = net.getPlaces().iterator().next();
057        final Dimension size = net.getSize(firstPlace);
058        final IconSize iconSize = IconSize.getIconSize(size);
059        ui.setIconSize(iconSize);
060
061        // Add places
062        final Set<PetriPlace> places = net.getPlaces();
063
064        for (final PetriPlace place : places) {
065
066            final PlaceLabel label = ui.addPlace(net.getName(place));
067            ui.moveLabel(label, net.getPosition(place));
068            objects.put(place, label);
069
070        }
071
072        // Add transitions
073        final Set<PetriTransition> transitions = net.getTransitions();
074
075        for (final PetriTransition transition : transitions) {
076
077            final TransitionLabel label =
078                ui.addTransition(net.getName(transition));
079            ui.moveLabel(label, net.getPosition(transition));
080            objects.put(transition, label);
081
082        }
083
084        // Add arrows
085        for (final PetriTransition transition : transitions) {
086
087            final TransitionLabel transitionLabel =
088                (TransitionLabel) objects.get(transition);
089
090            // Incoming arrows
091            final Set<PetriPlace> sources = net.getInputEdges(transition);
092
093            for (final PetriPlace source : sources) {
094
095                final PlaceLabel sourceLabel =
096                    (PlaceLabel) objects.get(source);
097
098                final Arrow arrow = new Arrow(sourceLabel, transitionLabel);
099
100                ui.addArrow(arrow);
101                arrows.put(sourceLabel, transitionLabel, arrow);
102
103            }
104
105            // Outgoing arrows
106            final Set<PetriPlace> targets = net.getOutputEdges(transition);
107
108            for (final PetriPlace target : targets) {
109
110                final PlaceLabel targetLabel =
111                    (PlaceLabel) objects.get(target);
112
113                final Arrow arrow =
114                    new Arrow(transitionLabel, transitionLabel);
115
116                ui.addArrow(arrow);
117                arrows.put(targetLabel, transitionLabel, arrow);
118
119            }
120
121        }
122
123        // Set canvas to size larger than needed by coordinates
124        int largest = LogicToUi.findLargestCoord(net, places, transitions);
125        largest += ui.getIconSize().getSize().height;
126        final Dimension newSize =
127            ui.calculateCanvasSize(new Dimension(largest, largest));
128        ui.setCanvasSize(newSize);
129
130    }
131
132    /**
133     * Find the largest coordinate for all elements.
134     * @param net Petri net to work with
135     * @param places Places to go through
136     * @param transitions Transition to go through
137     * @return Largest coordinate for Petri objects
138     */
139    private static int
140        findLargestCoord(final PetriNet net, final Set<PetriPlace> places,
141            final Set<PetriTransition> transitions) {
142
143        int largest = 0;
144
145        for (final PetriTransition petriTransition : transitions) {
146
147            final Point point = net.getPosition(petriTransition);
148
149            if (point.x > largest) {
150                largest = point.x;
151            }
152
153            if (point.y > largest) {
154                largest = point.y;
155            }
156
157        }
158
159        for (final PetriPlace petriPlace : places) {
160
161            final Point point = net.getPosition(petriPlace);
162
163            if (point.x > largest) {
164                largest = point.x;
165            }
166
167            if (point.y > largest) {
168                largest = point.y;
169            }
170
171        }
172
173        return largest;
174
175    }

177}
