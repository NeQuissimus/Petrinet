032package com.nequissimus.university.k1584.ui;
033
034import java.awt.Dimension;
035import java.awt.Point;
036import java.util.HashSet;
037import java.util.Iterator;
038import java.util.Set;
039
040import com.nequissimus.library.data.BiMap;
041import com.nequissimus.library.data.TwoKeyMap;
042import com.nequissimus.university.k1584.PetriController;
043import com.nequissimus.university.k1584.logic.PetriNet;
044import com.nequissimus.university.k1584.logic.PetriObject;
045import com.nequissimus.university.k1584.logic.PetriPlace;
046import com.nequissimus.university.k1584.logic.PetriTransition;
047import com.nequissimus.university.k1584.ui.elements.AbstractLabel;
048import com.nequissimus.university.k1584.ui.elements.Arrow;
049import com.nequissimus.university.k1584.ui.elements.PlaceLabel;
050import com.nequissimus.university.k1584.ui.elements.TransitionLabel;
051import com.nequissimus.university.k1584.ui.enums.IconSize;
052
053/**
054 * This class takes a logical net and draws all components onto the UI canvas.
055 * @author Tim Steinbach
056 */
057public final class LogicToUi {
058
059    /**
060     * Controller.
061     */
062    private static final PetriController CONTROLLER = PetriController
063        .getInstance();
064
065    /**
066     * Hide constructor.
067     */
068    private LogicToUi() {
069    }
070
071    /**
072     * Draw a Petri net onto the canvas used by the application's controller.
073     * @param net Petri net
074     */
075    public static void convert(final PetriNet net) {
076
077        // Get all components
078
079        final PetriUi ui = LogicToUi.CONTROLLER.getUi();
080        final BiMap<PetriObject, AbstractLabel> objects =
081            LogicToUi.CONTROLLER.getObjects();
082        final TwoKeyMap<AbstractLabel, AbstractLabel, Arrow> arrows =
083            LogicToUi.CONTROLLER.getArrows();
084
085        // Clean the controller
086        objects.clear();
087        arrows.clear();
088        LogicToUi.CONTROLLER.arrowConnect(null);
089        LogicToUi.CONTROLLER.arrowDisconnect(null);
090
091        // Clean the canvas
092        ui.clean();
093
094        // Set icon size
095        Set<PetriPlace> places = net.getPlaces();
096
097        if (null != places) {
098
099            final Iterator<PetriPlace> placeIterator = places.iterator();
100
101            if (placeIterator.hasNext()) {
102
103                final PetriPlace firstPlace = placeIterator.next();
104                final Dimension size = net.getSize(firstPlace);
105                final IconSize iconSize = IconSize.getIconSize(size);
106                ui.setIconSize(iconSize);
107
108            }
109
110        } else {
111            places = new HashSet<PetriPlace>();
112        }
113
114        // Add places
115
116        for (final PetriPlace place : places) {
117
118            final PlaceLabel label = ui.addPlace(net.getName(place));
119            final int markings = net.getTokens(place);
120
121            label.setTokens(markings);
122            ui.moveLabel(label, net.getPosition(place));
123            objects.put(place, label);
124
125        }
126
127        // Add transitions
128        final Set<PetriTransition> transitions = net.getTransitions();
129
130        for (final PetriTransition transition : transitions) {
131
132            final TransitionLabel label =
133                ui.addTransition(net.getName(transition));
134            ui.moveLabel(label, net.getPosition(transition));
135            objects.put(transition, label);
136
137        }
138
139        // Add arrows
140        for (final PetriTransition transition : transitions) {
141
142            final TransitionLabel transitionLabel =
143                (TransitionLabel) objects.get(transition);
144
145            // Incoming arrows
146            final Set<PetriPlace> sources = net.getInputEdges(transition);
147
148            for (final PetriPlace source : sources) {
149
150                final PlaceLabel sourceLabel =
151                    (PlaceLabel) objects.get(source);
152
153                final Arrow arrow = new Arrow(sourceLabel, transitionLabel);
154
155                ui.addArrow(arrow);
156                arrows.put(sourceLabel, transitionLabel, arrow);
157
158            }
159
160            // Outgoing arrows
161            final Set<PetriPlace> targets = net.getOutputEdges(transition);
162
163            for (final PetriPlace target : targets) {
164
165                final PlaceLabel targetLabel =
166                    (PlaceLabel) objects.get(target);
167
168                final Arrow arrow = new Arrow(transitionLabel, targetLabel);
169
170                ui.addArrow(arrow);
171                arrows.put(targetLabel, transitionLabel, arrow);
172
173            }
174
175        }
176
177        // Set canvas to size larger than needed by coordinates
178        int largest = LogicToUi.findLargestCoord(net, places, transitions);
179        largest += ui.getIconSize().getSize().height;
180        final Dimension newSize =
181            ui.calculateCanvasSize(new Dimension(largest, largest));
182        ui.resizeCanvas(newSize);
183        ui.resizeArrowCanvas(newSize);
184
185    }
186
187    /**
188     * Find the largest coordinate for all elements.
189     * @param net Petri net to work with
190     * @param places Places to go through
191     * @param transitions Transition to go through
192     * @return Largest coordinate for Petri objects
193     */
194    private static int
195        findLargestCoord(final PetriNet net, final Set<PetriPlace> places,
196            final Set<PetriTransition> transitions) {
197
198        int largest = 0;
199
200        for (final PetriTransition petriTransition : transitions) {
201
202            final Point point = net.getPosition(petriTransition);
203
204            if (point.x > largest) {
205                largest = point.x;
206            }
207
208            if (point.y > largest) {
209                largest = point.y;
210            }
211
212        }
213
214        for (final PetriPlace petriPlace : places) {
215
216            final Point point = net.getPosition(petriPlace);
217
218            if (point.x > largest) {
219                largest = point.x;
220            }
221
222            if (point.y > largest) {
223                largest = point.y;
224            }
225
226        }
227
228        return largest;
229
230    }
