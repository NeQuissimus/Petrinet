001package com.nequissimus.university.k1584.ui;

003import java.awt.Point;
004import java.util.HashSet;
005import java.util.Set;
006
007import com.nequissimus.university.k1584.PetriController;
008import com.nequissimus.university.k1584.logic.PetriNet;
009import com.nequissimus.university.k1584.logic.PetriObject;
010import com.nequissimus.university.k1584.logic.PetriPlace;
011import com.nequissimus.university.k1584.logic.PetriTransition;
012import com.nequissimus.university.k1584.ui.elements.AbstractLabel;
013import com.nequissimus.university.k1584.ui.elements.PetriCanvas;
014import com.nequissimus.university.k1584.ui.elements.PetriPlaceLabel;
015import com.nequissimus.university.k1584.ui.elements.PetriTransitionLabel;
016import com.nequissimus.university.k1584.ui.elements.PetriWindow;
017
018/**
019 * This class takes a logical net and draws all components onto the UI canvas.
020 * @author Tim Steinbach
021 */
022public final class LogicToUi {
023
024    /**
025     * Hide constructor.
026     */
027    private LogicToUi() {
028    }
029
030    /**
031     * Draw a Petri net onto the canvas used by the application's controller.
032     * This method utilizes the controller's currentNet!
033     * @param net Petri net
034     */
035    public static void convert(final PetriNet net) {
036
037        final PetriController controller = PetriController.getInstance();
038        final Set<PetriPlace> places = net.getPlaces();
039        final Set<PetriTransition> transitions = net.getTransitions();
040        final PetriNet currentNet = controller.getCurrentNet();
041
042        final Set<PetriPlaceLabel> placeLabels =
043            new HashSet<PetriPlaceLabel>();
044        final Set<PetriTransitionLabel> transitionLabels =
045            new HashSet<PetriTransitionLabel>();
046
047        // Clean canvas
048        PetriCanvas canvas = PetriWindow.getCanvas();
049        canvas.getCanvas().removeAll();
050
051        // Set up currentNet
052        currentNet.setName(net.getName());
053
054        // Add places
055        for (final PetriPlace petriPlace : places) {
056
057            final Point location = currentNet.getPosition(petriPlace);
058            final PetriPlaceLabel label = new PetriPlaceLabel(petriPlace);
059
060            controller.addLabel(label);
061            placeLabels.add(label);
062
063            final PetriPlace newPlace =
064                currentNet.addPlace(currentNet.getName(petriPlace));
065            currentNet.setPosition(newPlace, location);
066
067        }
068
069        // Add transitions
070        for (final PetriTransition petriTransition : transitions) {
071
072            final Point location = currentNet.getPosition(petriTransition);
073            final PetriTransitionLabel label =
074                new PetriTransitionLabel(petriTransition);
075
076            controller.addLabel(label);
077            transitionLabels.add(label);
078
079            final PetriTransition newTransition =
080                currentNet.addTransition(currentNet
081                    .getName(petriTransition));
082            currentNet.setPosition(newTransition, location);
083
084        }
085
086        // Add edges
087        for (final PetriTransitionLabel label : transitionLabels) {
088
089            PetriTransition transition =
090                (PetriTransition) label.getObject();
091
092            Set<PetriPlace> incoming = currentNet.getInputEdges(transition);
093            Set<PetriPlace> outgoing =
094                currentNet.getOutputEdges(transition);
095
096            for (PetriPlace petriPlace : incoming) {
097
098                // Empty out selected connection
099                controller.arrowConnect(null);
100
101                PetriPlaceLabel placeLabel =
102                    (PetriPlaceLabel) findLabel(placeLabels, petriPlace);
103
104                controller.arrowConnect(label);
105                controller.arrowConnect(placeLabel);
106
107            }
108
109            for (PetriPlace petriPlace : outgoing) {
110
111                // Empty out selected connection
112                controller.arrowConnect(null);
113
114                PetriPlaceLabel placeLabel =
115                    (PetriPlaceLabel) findLabel(placeLabels, petriPlace);
116
117                controller.arrowConnect(placeLabel);
118                controller.arrowConnect(label);
119            }
120
121        }
122
123        // Redraw canvas
124        controller.redrawCanvas();
125
126    }
127
128    /**
129     * Find the matching label for a Petri object.
130     * @param set Set of labels to look through
131     * @param object Object to look for
132     * @return Petri object if found, else NULL
133     */
134    private static AbstractLabel findLabel(
135        final Set<? extends AbstractLabel> set, final PetriObject object) {
136
137        for (AbstractLabel label : set) {
138            final PetriObject tmpObject = label.getObject();
139            if (tmpObject.equals(object)) {
140                return label;
141            }
142        }
143
144        return null;
145
146    }
147
148}
