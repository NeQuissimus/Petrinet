001package com.nequissimus.university.k1584.ui;
002
003import java.awt.Dimension;
004import java.awt.Point;
005import java.util.Iterator;
006import java.util.Set;
007
008import com.nequissimus.library.data.BiMap;
009import com.nequissimus.library.data.TwoKeyMap;
010import com.nequissimus.university.k1584.PetriController;
011import com.nequissimus.university.k1584.logic.PetriNet;
012import com.nequissimus.university.k1584.logic.PetriObject;
013import com.nequissimus.university.k1584.logic.PetriPlace;
014import com.nequissimus.university.k1584.logic.PetriTransition;
015import com.nequissimus.university.k1584.ui.elements.AbstractLabel;
016import com.nequissimus.university.k1584.ui.elements.Arrow;
017import com.nequissimus.university.k1584.ui.elements.PlaceLabel;
018import com.nequissimus.university.k1584.ui.elements.TransitionLabel;
019import com.nequissimus.university.k1584.ui.enums.IconSize;
020
021/**
022 * This class takes a logical net and draws all components onto the UI canvas.
023 * @author Tim Steinbach
024 */
025public final class LogicToUi {
026
027    /**
028     * Controller.
029     */
030    private static final PetriController CONTROLLER = PetriController
031        .getInstance();
032
033    /**
034     * Hide constructor.
035     */
036    private LogicToUi() {
037    }
038
039    /**
040     * Draw a Petri net onto the canvas used by the application's controller.
041     * @param net Petri net
042     */
043    public static void convert(final PetriNet net) {
044
045        // Get all components
046
047        final PetriUi ui = LogicToUi.CONTROLLER.getUi();
048        final BiMap<PetriObject, AbstractLabel> objects =
049            LogicToUi.CONTROLLER.getObjects();
050        final TwoKeyMap<PlaceLabel, TransitionLabel, Arrow> arrows =
051            LogicToUi.CONTROLLER.getArrows();
052
053        // Clean the controller
054        objects.clear();
055        arrows.clear();
056        LogicToUi.CONTROLLER.arrowConnect(null);
057        LogicToUi.CONTROLLER.arrowDisconnect(null);
058
059        // Clean the canvas
060        ui.clean();
061
062        // Set icon size
063        final Set<PetriPlace> places = net.getPlaces();
064
065        if (null != places) {
066
067            final Iterator<PetriPlace> placeIterator = places.iterator();
068
069            if (placeIterator.hasNext()) {
070
071                final PetriPlace firstPlace = placeIterator.next();
072                final Dimension size = net.getSize(firstPlace);
073                final IconSize iconSize = IconSize.getIconSize(size);
074                ui.setIconSize(iconSize);
075
076            }
077
078        }
079
080        // Add places
081
082        for (final PetriPlace place : places) {
083
084            final PlaceLabel label = ui.addPlace(net.getName(place));
085            final int markings = net.getMarkings(place);
086
087            label.setMarkings(markings);
088            ui.moveLabel(label, net.getPosition(place));
089            objects.put(place, label);
090
091        }
092
093        // Add transitions
094        final Set<PetriTransition> transitions = net.getTransitions();
095
096        for (final PetriTransition transition : transitions) {
097
098            final TransitionLabel label =
099                ui.addTransition(net.getName(transition));
100            ui.moveLabel(label, net.getPosition(transition));
101            objects.put(transition, label);
102
103        }
104
105        // Add arrows
106        for (final PetriTransition transition : transitions) {
107
108            final TransitionLabel transitionLabel =
109                (TransitionLabel) objects.get(transition);
110
111            // Incoming arrows
112            final Set<PetriPlace> sources = net.getInputEdges(transition);
113
114            for (final PetriPlace source : sources) {
115
116                final PlaceLabel sourceLabel =
117                    (PlaceLabel) objects.get(source);
118
119                final Arrow arrow = new Arrow(sourceLabel, transitionLabel);
120
121                ui.addArrow(arrow);
122                arrows.put(sourceLabel, transitionLabel, arrow);
123
124            }
125
126            // Outgoing arrows
127            final Set<PetriPlace> targets = net.getOutputEdges(transition);
128
129            for (final PetriPlace target : targets) {
130
131                final PlaceLabel targetLabel =
132                    (PlaceLabel) objects.get(target);
133
134                final Arrow arrow = new Arrow(transitionLabel, targetLabel);
135
136                ui.addArrow(arrow);
137                arrows.put(targetLabel, transitionLabel, arrow);
138
139            }
140
141        }
142
143        // Set canvas to size larger than needed by coordinates
144        int largest = LogicToUi.findLargestCoord(net, places, transitions);
145        largest += ui.getIconSize().getSize().height;
146        final Dimension newSize =
147            ui.calculateCanvasSize(new Dimension(largest, largest));
148        ui.setCanvasSize(newSize);
149
150    }
151
152    /**
153     * Find the largest coordinate for all elements.
154     * @param net Petri net to work with
155     * @param places Places to go through
156     * @param transitions Transition to go through
157     * @return Largest coordinate for Petri objects
158     */
159    private static int
160        findLargestCoord(final PetriNet net, final Set<PetriPlace> places,
161            final Set<PetriTransition> transitions) {
162
163        int largest = 0;
164
165        for (final PetriTransition petriTransition : transitions) {
166
167            final Point point = net.getPosition(petriTransition);
168
169            if (point.x > largest) {
170                largest = point.x;
171            }
172
173            if (point.y > largest) {
174                largest = point.y;
175            }
176
177        }
178
179        for (final PetriPlace petriPlace : places) {
180
181            final Point point = net.getPosition(petriPlace);
182
183            if (point.x > largest) {
184                largest = point.x;
185            }
186
187            if (point.y > largest) {
188                largest = point.y;
189            }
190
191        }
192
193        return largest;
194
195    }
