032package com.nequissimus.university.k1584.ui.enums;
033
034import java.awt.Dimension;
035
036/**
037 * Preset values for icon sizes.<br />
038 * VERY_SMALL: 30x30 px<br />
039 * SMALL: 45x45 px<br />
040 * MEDIUM: 60x60 px<br />
041 * LARGE: 100x100 px<br />
042 * @author Tim Steinbach
043 */
044public enum IconSize {
045
046    /**
047     * Icon size values.
048     */
049    VERY_SMALL(30, 30), SMALL(45, 45), MEDIUM(60, 60), LARGE(100, 100);
050
051    /**
052     * Icon size as dimension object.
053     */
054    private Dimension size;
055
056    /**
057     * Create a new icon size value.
058     * @param width Icon width
059     * @param height Icon height
060     */
061    private IconSize(final int width, final int height) {
062
063        this.size = new Dimension(width, height);
064
065    }
066
067    /**
068     * Try to find an enumeration representation for a given size. Only either
069     * an element's height <b>OR</b> width is checked as the length of the label
070     * size can vary.<br />
071     * This methods returns MEDIUM if no match could be found.
072     * @param size Desired dimension
073     * @return IconSize representation if one was found, MEDIUM otherwise
074     */
075    public static IconSize getIconSize(final Dimension size) {
076
077        for (final IconSize iconSize : IconSize.values()) {
078
079            final Dimension iSize = iconSize.getSize();
080
081            if ((iSize.height == size.height)
082                || (iSize.width == size.width)) {
083                return iconSize;
084            }
085
086        }
087
088        return MEDIUM;
089
090    }
091
092    /**
093     * Get icon size.
094     * @return Icon size
095     */
096    public Dimension getSize() {
097        return this.size;
098    }
099
100}
