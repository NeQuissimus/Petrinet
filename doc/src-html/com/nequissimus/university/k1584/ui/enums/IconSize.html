001package com.nequissimus.university.k1584.ui.enums;
002
003import java.awt.Dimension;
004
005/**
006 * Preset values for icon sizes.<br />
007 * VERY_SMALL: 30x30 px<br />
008 * SMALL: 45x45 px<br />
009 * MEDIUM: 60x60 px<br />
010 * LARGE: 100x100 px<br />
011 * @author Tim Steinbach
012 */
013public enum IconSize {
014
015    /**
016     * Icon size values.
017     */
018    VERY_SMALL(30, 30), SMALL(45, 45), MEDIUM(60, 60), LARGE(100, 100);
019
020    /**
021     * Icon size as dimension object.
022     */
023    private Dimension size;
024
025    /**
026     * Create a new icon size value.
027     * @param width Icon width
028     * @param height Icon height
029     */
030    private IconSize(final int width, final int height) {
031
032        this.size = new Dimension(width, height);
033
034    }
035
036    /**
037     * Get icon size.
038     * @return Icon size
039     */
040    public Dimension getSize() {
041        return this.size;
042    }
043
044    /**
045     * Try to find an enum representation for a given size.<br />
046     * This methods returns MEDIUM if no match could be found.
047     * @param size Desired dimension
048     * @return IconSize representation if one was found, MEDIUM otherwise
049     */
050    public static IconSize getIconSize(final Dimension size) {
051
052        for (final IconSize iconSize : IconSize.values()) {
053
054            if (iconSize.getSize().equals(size)) {
055                return iconSize;
056            }
057
058        }
059
060        return MEDIUM;
061
062    }
063
064}
