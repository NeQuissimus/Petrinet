001package com.nequissimus.university.k1584.ui;
002
003import javax.swing.JLabel;
004
005import com.nequissimus.university.k1584.ui.enums.TextPosition;
006
007public abstract class AbstractLabel extends JLabel {
008
009    private static final long serialVersionUID = 6558385524618595255L;
010    
011    protected String name;
012    
013    public AbstractLabel(final String name) {
014        
015        this.name = name;
016        
017        this.setIcon(getPetriIcon());
018        this.setText(this.name);
019        this.setOpaque(false);
020        
021        this.moveText(TextPosition.RIGHT);
022        
023    }
024    
025    void moveText(final TextPosition pos) {
026        
027        this.setHorizontalTextPosition(pos.getX());
028        this.setVerticalTextPosition(pos.getY());
029        
030    }
031    
032    abstract AbstractIcon getPetriIcon();
033
034}
