001package com.nequissimus.university.k1584.ui;
002
003import java.awt.Color;
004import java.awt.Dimension;
005
006import javax.swing.JPanel;
007import javax.swing.JScrollPane;
008
009import com.nequissimus.university.k1584.logic.PetriConfig;
010
011/**
012 * This is the main canvas on which all the editing will take place.
013 * It consists of this panel that is scrollable and comes with a predefined size.
014 * Its size can be changed to allow for a larger editing canvas.
015 * @see PetriConfig
016 * @author Tim Schram
017 *
018 */
019public class PetriCanvas extends JScrollPane {
020
021    private static final long serialVersionUID = 6158181636171175402L;
022    
023    private final JPanel canvas;
024    private final PetriConfig config;
025    
026    PetriCanvas() {
027        
028        super();
029        
030        this.config = PetriConfig.getInstance();
031        
032        this.canvas = new JPanel();
033        
034        this.resetCanvas();     
035        
036        this.canvas.validate();
037        this.canvas.repaint();
038
039        final int visibleWidth = this.config.getWindowWidth() - this.config.getSidebarWidth();
040        final int visibleHeight = this.config.getWindowHeight() - this.config.getScrollbarHeight();
041        final Dimension visibleSize = new Dimension(visibleWidth, visibleHeight);
042        
043        this.setPreferredSize(visibleSize);
044        this.setSize(visibleSize);
045        
046        this.setViewportView(canvas);
047
048        this.setHorizontalScrollBarPolicy(JScrollPane.HORIZONTAL_SCROLLBAR_ALWAYS);
049        this.setVerticalScrollBarPolicy(JScrollPane.VERTICAL_SCROLLBAR_ALWAYS);
050        
051    }
052    
053    private void resetCanvas() {
054        
055        this.canvas.setBackground(Color.WHITE);
056        this.canvas.setOpaque(true);
057
058        final int canvasWidth = this.config.getCanvasWidth();
059        final int canvasHeight = this.config.getCanvasHeight();
060        
061        final Dimension canvasSize = new Dimension(canvasWidth, canvasHeight);
062        
063        this.canvas.setPreferredSize(canvasSize);
064        this.canvas.setSize(canvasSize);
065        
066    }    
067
068}
