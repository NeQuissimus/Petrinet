001package com.nequissimus.university.k1584.ui;
002
003import javax.swing.JPanel;
004
005import com.nequissimus.university.k1584.logic.PetriConfig;
006import com.nequissimus.university.k1584.ui.enums.TextPosition;
007
008/**
009 * Sidebar UI for adding new elements to the canvas.
010 * @author Tim Steinbach
011 *
012 */
013public class PetriSidebar extends JPanel {
014
015    /**
016     * Serializable UID.
017     */
018    private static final long serialVersionUID = -7934305863238811085L;
019
020    /**
021     * Configuration instance.
022     */
023    private final PetriConfig config = PetriConfig.getInstance();
024
025    /**
026     * Create a new sidebar instance.
027     */
028    PetriSidebar() {
029
030        super();
031
032        final int width = this.config.getSidebarWidth();
033        final int height = this.config.getWindowHeight();
034
035        // Use same color as window background (This prevents ugly effects when
036        // resizing the window)
037        this.setOpaque(false);
038
039        this.addIcons();
040
041        this.setSize(width, height);
042
043    }
044
045    /**
046     * Add the default icons to the sidebar.
047     * (1 place, 1 transition)
048     */
049    private void addIcons() {
050
051        PetriPlaceLabel label = new PetriPlaceLabel();
052        label.moveText(TextPosition.BELOW);
053        label.setBounds(10, 10, 100, 100);
054
055        this.add(label);
056
057        PetriTransitionLabel label2 = new PetriTransitionLabel();
058        label2.moveText(TextPosition.BELOW);
059        label2.setBounds(0, 120, 100, 100);
060
061        this.add(label2);
062
063    }
064
065}
