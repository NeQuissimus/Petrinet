001package com.nequissimus.university.k1584.ui.listener;
002
003import java.awt.event.AdjustmentEvent;
004import java.awt.event.AdjustmentListener;
005
006import com.nequissimus.university.k1584.PetriController;
007
008/**
009 * Listener that refreshed the canvas when the surrounding
010 * {@link javax.swing.JScrollPane} has been moved.
011 * @author Tim Steinbach
012 */
013public class ScrollListener implements AdjustmentListener {
014
015    @Override
016    public final void adjustmentValueChanged(final AdjustmentEvent e) {
017
018        if (e.getValueIsAdjusting()) {
019            return;
020        }
021
022        PetriController.getInstance().redrawCanvas();
023
024    }
025}
