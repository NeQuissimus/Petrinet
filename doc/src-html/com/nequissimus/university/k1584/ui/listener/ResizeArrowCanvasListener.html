001package com.nequissimus.university.k1584.ui.listener;
002
003import java.awt.event.ComponentEvent;
004import java.awt.event.ComponentListener;
005
006import com.nequissimus.university.k1584.PetriController;
007
008/**
009 * Automatically resize the canvas used by the arrows to match the size of the
010 * net canvas.
011 * @author Tim Steinbach
012 */
013public final class ResizeArrowCanvasListener implements ComponentListener {
014
015    @Override
016    public void componentHidden(final ComponentEvent arg0) {
017    }
018
019    @Override
020    public void componentMoved(final ComponentEvent arg0) {
021    }
022
023    @Override
024    public void componentShown(final ComponentEvent arg0) {
025    }
026
027    @Override
028    public void componentResized(final ComponentEvent arg0) {
029
030        PetriController.getInstance().resizeArrowCanvas();
031
032    }
033
034}
