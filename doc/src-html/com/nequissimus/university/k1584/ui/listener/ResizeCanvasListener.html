001package com.nequissimus.university.k1584.ui.listener;
002
003import java.awt.Component;
004import java.awt.Dimension;
005import java.awt.event.ComponentEvent;
006
007import com.nequissimus.university.k1584.logic.PetriConfig;
008
009/**
010 * This Listener extends {@link ResizeListener} by the feature of resizing
011 * another component in height according to the original source.
012 * The slave component will always be the same height as the original source component.
013 * @author Tim Schram
014 *
015 */
016public class ResizeCanvasListener extends ResizeListener {
017    
018    private Component slave = null;
019    PetriConfig config = PetriConfig.getInstance();
020    
021    public ResizeCanvasListener(final Dimension minSize, Component slave) {
022        
023        super(minSize);
024        
025        this.slave = slave;
026        
027    }
028    
029    @Override
030    public void componentResized(final ComponentEvent e) {
031        
032        super.componentResized(e);
033        
034        if (this.slave != null) {
035            
036            final Dimension size = this.slave.getSize();
037            
038            final Component component = (Component) e.getSource();
039            
040            size.height = component.getSize().height - this.config.getScrollbarHeight();
041            size.width = component.getSize().width - this.config.getSidebarWidth();
042            
043            this.slave.setSize(size);
044            
045            this.slave.validate();
046            this.slave.repaint();
047            
048        }
049        
050    }
051
052}
