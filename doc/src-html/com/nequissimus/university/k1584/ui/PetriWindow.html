001package com.nequissimus.university.k1584.ui;
002
003import java.awt.Dimension;
004import java.awt.Point;
005
006import javax.swing.JFrame;
007
008import com.nequissimus.university.k1584.logic.PetriConfig;
009import com.nequissimus.university.k1584.ui.enums.IconSize;
010import com.nequissimus.university.k1584.ui.listener.ResizeCanvasListener;
011import com.nequissimus.university.k1584.ui.listener.ResizeSidebarListener;
012
013
014/**
015 * The main window that holds all the major UI elements, such as the editing canvas, menus etc.
016 * @author Tim Schram
017 *
018 */
019public class PetriWindow extends JFrame {
020
021    private static final long serialVersionUID = 6276277357529619473L;
022    
023    private static IconSize iconSize = IconSize.VERY_SMALL;
024    
025    private final PetriConfig config;
026
027    private final PetriCanvas canvas;
028    private final PetriSidebar sidebar;
029
030    public PetriWindow() {
031
032        super();
033
034        this.setLayout(null);
035
036        this.config = PetriConfig.getInstance();
037
038        this.canvas = new PetriCanvas();
039        this.sidebar = new PetriSidebar();
040
041        this.resetTitle();
042        this.resetSize();
043        this.resetLocation();
044        this.resetCanvas();
045        this.resetSidebar();
046        
047        this.resetListeners();
048
049        this.add(this.canvas);
050        this.add(this.sidebar);
051        
052        this.setBackground(this.config.getWindowBackgroundColor());
053                
054    }
055    
056    public static void setIconSize(IconSize size) {PetriWindow.iconSize = size;}
057    public static IconSize getIconSize() {return PetriWindow.iconSize;}
058
059    private void resetSize() {
060
061        final int windowWidth = this.config.getWindowWidth();
062        final int windowHeight = this.config.getWindowHeight();
063
064        final Dimension windowSize = new Dimension(windowWidth, windowHeight);
065
066        this.setPreferredSize(windowSize);
067        this.setSize(windowSize);
068
069    }
070
071    private void resetLocation() {
072
073        final int windowX = this.config.getWindowX();
074        final int windowY = this.config.getWindowY();
075
076        final Point windowLocation = new Point(windowX, windowY);
077
078        this.setLocation(windowLocation);
079
080    }
081
082    private void resetTitle() {
083
084        this.setTitle(this.config.getWindowTitle());
085
086    }
087
088    private void resetCanvas() {
089
090        this.canvas.setLocation(new Point(0, 0));
091        this.canvas.validate();
092        this.canvas.repaint();
093
094    }
095    
096    private void resetSidebar() {
097        
098        final Point location = new Point(this.config.getWindowWidth() - this.config.getSidebarWidth(), 0);
099        
100        this.sidebar.setLocation(location);
101        
102    }
103    
104    private void resetListeners() {
105        
106        final Dimension minSize = new Dimension(this.config.getWindowMinWidth(), this.config.getWindowMinHeight());
107        
108        this.addComponentListener(new ResizeCanvasListener(minSize, this.canvas));
109        this.addComponentListener(new ResizeSidebarListener(minSize, this.sidebar));
110        
111    }
112
113}
