032package com.nequissimus.university.k1584.ui.elements;
033
034import com.nequissimus.library.data.Singleton;
035import com.nequissimus.university.k1584.logic.PetriConfig;
036import com.nequissimus.university.k1584.ui.enums.IconSize;
037import com.nequissimus.university.k1584.ui.listener.CanvasTransitionListener;
038import com.nequissimus.university.k1584.ui.listener.DragListener;
039import com.nequissimus.university.k1584.ui.listener.SelectListener;
040import com.nequissimus.university.k1584.ui.traits.Draggable;
041
042/**
043 * UI label for transition objects. This label consists of the transition icon
044 * and a name tag.
045 * @author Tim Steinbach
046 */
047public class TransitionLabel extends AbstractLabel implements Draggable {
048
049    /**
050     * Configuration.
051     */
052    private static final PetriConfig CONFIG = Singleton
053        .getObject(PetriConfig.class);
054
055    /**
056     * Serializable UID.
057     */
058    private static final long serialVersionUID = 5447519082475950763L;
059
060    /**
061     * Mouse listener for drag and drop.
062     */
063    private DragListener dragDropListener;
064
065    /**
066     * Mouse listener for selecting multiple labels.
067     */
068    private SelectListener selectListener;
069
070    /**
071     * Create a transition label without connecting it to a logical transition
072     * (for the sidebar). Such a label has no UI behaviour (e.g. drag and drop)
073     * either. It does have a context menu by default.
074     */
075    public TransitionLabel() {
076
077        super();
078
079        this.addSidebarMenu();
080        this.setText(TransitionLabel.CONFIG.getTransitionName());
081
082    }
083
084    /**
085     * Create a new label for a logical transition.
086     * @param name Transition name
087     */
088    public TransitionLabel(final String name) {
089
090        super(name);
091
092        this.registerDraggable();
093        this.addCanvasMenu();
094
095    }
096
097    @Override
098    public final void addCanvasMenu() {
099
100        this.addMouseListener(new CanvasTransitionListener(this));
101
102    }
103
104    /**
105     * Get icon of particular size.
106     * @param size Size
107     * @return Icon
108     */
109    @Override
110    public final AbstractIcon getPetriIcon(final IconSize size) {
111
112        return new TransitionIcon(size);
113
114    }
115
116    @Override
117    public final void registerDraggable() {
118
119        this.selectListener = new SelectListener(this);
120        this.dragDropListener = new DragListener(this);
121
122        this.addMouseListener(this.selectListener);
123        this.addMouseListener(this.dragDropListener);
124        this.addMouseMotionListener(this.dragDropListener);
125
126    }
127
128    @Override
129    public final void unregisterDraggable() {
130
131        this.removeMouseListener(this.selectListener);
132        this.removeMouseListener(this.dragDropListener);
133        this.removeMouseMotionListener(this.dragDropListener);
134
135    }
136
137}
