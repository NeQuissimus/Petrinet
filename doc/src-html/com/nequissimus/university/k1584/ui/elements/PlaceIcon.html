032package com.nequissimus.university.k1584.ui.elements;
033
034import java.awt.Color;
035import java.awt.Dimension;
036import java.awt.Font;
037import java.awt.FontMetrics;
038import java.awt.Graphics2D;
039import java.awt.Image;
040import java.awt.image.BufferedImage;
041
042import javax.swing.ImageIcon;
043
044import com.nequissimus.library.data.Singleton;
045import com.nequissimus.university.k1584.PetriController;
046import com.nequissimus.university.k1584.logic.PetriConfig;
047import com.nequissimus.university.k1584.ui.enums.IconSize;
048import com.nequissimus.university.k1584.ui.enums.TokenSize;
049import com.nequissimus.university.k1584.ui.traits.Resizable;
050
051/**
052 * UI Icon for Petri places.
053 * @author Tim Steinbach
054 */
055public class PlaceIcon extends AbstractIcon implements Resizable {
056
057    /**
058     * Configuration.
059     */
060    private static final PetriConfig CONFIG = Singleton
061        .getObject(PetriConfig.class);
062
063    /**
064     * Serializable UID.
065     */
066    private static final long serialVersionUID = -68079442351799309L;
067
068    /**
069     * Number of tokens.
070     */
071    private int tokens = 0;
072
073    /**
074     * Create a new place icon with a given size.
075     * @param size Icon size
076     */
077    protected PlaceIcon(final IconSize size) {
078
079        super(size);
080
081        this.setDefaultImage(new ImageIcon(PlaceIcon.CONFIG.getImagePlace()));
082
083        this.draw();
084
085    }
086
087    /**
088     * Draw markings onto icon.
089     */
090    public final void drawTokens() {
091
092        final PetriController controller = PetriController.getInstance();
093
094        final TokenSize tokenSize = controller.getTokenSize();
095
096        final Image oldImage = this.getImage();
097
098        final int height = oldImage.getHeight(null);
099        final int width = oldImage.getWidth(null);
100
101        final BufferedImage image =
102            new BufferedImage(height, width, BufferedImage.TYPE_INT_ARGB);
103
104        final Graphics2D graphics = image.createGraphics();
105
106        graphics.setColor(Color.BLACK);
107        graphics.drawImage(oldImage, 0, 0, null);
108
109        if (this.tokens == 1) {
110
111            // final int ovalRadius = this.getIconHeight() / 20;
112            final int ovalRadius = tokenSize.getRadius();
113
114            final int coord = (width / 2) - ovalRadius;
115
116            graphics.fillOval(coord, coord, ovalRadius * 2, ovalRadius * 2);
117
118        } else if (this.tokens > 1) {
119
120            final String text = String.valueOf(this.tokens);
121
122            FontMetrics metrics = graphics.getFontMetrics();
123
124            final Font font = metrics.getFont();
125            final int fontSize = font.getSize();
126            final int newFontSize = tokenSize.getFontSize();
127
128            if (fontSize != newFontSize) {
129
130                final Font newFont =
131                    new Font(font.getFamily(), font.getStyle(), newFontSize);
132                graphics.setFont(newFont);
133                metrics = graphics.getFontMetrics();
134
135            }
136
137            final int textWidth = metrics.stringWidth(text);
138            final int textHeight = metrics.getAscent();
139
140            graphics.drawString(text, (width - textWidth) / 2,
141                (height + textHeight) / 2);
142
143        }
144
145        this.setImage(image);
146
147    }
148
149    @Override
150    public final void resize(final Dimension newSize) {
151
152        this.setSize(newSize);
153        this.draw();
154
155    }
156
157    /**
158     * Set value of tokens.
159     * @param value Tokens
160     */
161    public final void setTokens(final int value) {
162        this.tokens = value;
163        this.drawTokens();
164    }
165
166    @Override
167    final void drawOnFile() {
168
169        this.drawTokens();
170
171    }
172
173}
