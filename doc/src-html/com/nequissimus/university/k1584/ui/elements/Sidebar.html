032package com.nequissimus.university.k1584.ui.elements;
033
034import javax.swing.JPanel;
035
036import com.nequissimus.library.data.Singleton;
037import com.nequissimus.university.k1584.logic.PetriConfig;
038import com.nequissimus.university.k1584.ui.enums.TextPosition;
039
040/**
041 * Sidebar UI for adding new elements to the canvas.
042 * @author Tim Steinbach
043 */
044public class Sidebar extends JPanel {
045
046    /**
047     * Configuration instance.
048     */
049    private static final PetriConfig CONFIG = Singleton
050        .getObject(PetriConfig.class);
051
052    /**
053     * Serializable UID.
054     */
055    private static final long serialVersionUID = -7934305863238811085L;
056
057    /**
058     * Create a new sidebar instance.
059     */
060    Sidebar() {
061
062        super();
063
064        final int width = Sidebar.CONFIG.getSidebarWidth();
065        final int height = Sidebar.CONFIG.getWindowHeight();
066
067        // Use same color as window background (This prevents ugly effects when
068        // resizing the window)
069        this.setOpaque(false);
070
071        this.addIcons();
072
073        this.setSize(width, height);
074
075    }
076
077    /**
078     * Add the default icons to the sidebar. (1 place, 1 transition)
079     */
080    private void addIcons() {
081
082        final PlaceLabel label = new PlaceLabel();
083        label.moveText(TextPosition.BELOW);
084        label.setBounds(10, 10, 100, 100);
085
086        this.add(label);
087
088        final TransitionLabel label2 = new TransitionLabel();
089        label2.moveText(TextPosition.BELOW);
090        label2.setBounds(0, 120, 100, 100);
091
092        this.add(label2);
093
094    }
095
096}
