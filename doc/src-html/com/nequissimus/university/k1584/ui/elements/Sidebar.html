001package com.nequissimus.university.k1584.ui.elements;
002
003import javax.swing.JPanel;
004
005import com.nequissimus.university.k1584.logic.PetriConfig;
006import com.nequissimus.university.k1584.ui.enums.TextPosition;
007
008/**
009 * Sidebar UI for adding new elements to the canvas.
010 * @author Tim Steinbach
011 */
012public class Sidebar extends JPanel {
013
014    /**
015     * Serializable UID.
016     */
017    private static final long serialVersionUID = -7934305863238811085L;
018
019    /**
020     * Configuration instance.
021     */
022    private static final PetriConfig CONFIG = PetriConfig.getInstance();
023
024    /**
025     * Create a new sidebar instance.
026     */
027    Sidebar() {
028
029        super();
030
031        final int width = Sidebar.CONFIG.getSidebarWidth();
032        final int height = Sidebar.CONFIG.getWindowHeight();
033
034        // Use same color as window background (This prevents ugly effects when
035        // resizing the window)
036        this.setOpaque(false);
037
038        this.addIcons();
039
040        this.setSize(width, height);
041
042    }
043
044    /**
045     * Add the default icons to the sidebar. (1 place, 1 transition)
046     */
047    private void addIcons() {
048
049        final PlaceLabel label = new PlaceLabel();
050        label.moveText(TextPosition.BELOW);
051        label.setBounds(10, 10, 100, 100);
052
053        this.add(label);
054
055        final TransitionLabel label2 = new TransitionLabel();
056        label2.moveText(TextPosition.BELOW);
057        label2.setBounds(0, 120, 100, 100);
058
059        this.add(label2);
060
061    }
062
063}
