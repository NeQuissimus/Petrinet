032package com.nequissimus.university.k1584.ui.elements;
033
034import java.awt.Color;
035import java.awt.Dimension;
036import java.awt.Graphics;
037import java.awt.Graphics2D;
038import java.awt.Point;
039import java.awt.Polygon;
040import java.awt.RenderingHints;
041
042import javax.swing.JPanel;
043
044import com.nequissimus.university.k1584.PetriController;
045import com.nequissimus.university.k1584.logic.PetriConstants;
046import com.nequissimus.university.k1584.ui.enums.BoxSide;
047import com.nequissimus.university.k1584.ui.enums.TextPosition;
048
049/**
050 * Arrow connecting two Petri objects. The arrow is automatically aligned to the
051 * objects according to their relative position to each other. That way the
052 * arrow never crosses through the object itself. The arrow aligns on the top,
053 * bottom, left or right side of the label icon and is always centered on the
054 * respective side.
055 * @author Tim Steinbach
056 */
057public class Arrow extends JPanel {
058
059    /**
060     * Controller.
061     */
062    private static final PetriController CONTROLLER = PetriController
063        .getInstance();
064
065    /**
066     * Serializable UID.
067     */
068    private static final long serialVersionUID = 3151568700241791450L;
069
070    /**
071     * Arrow cross line.
072     */
073    private final int[] crossLine = new int[4];
074
075    /**
076     * Connect arrow from this label.
077     */
078    private final AbstractLabel from;
079
080    /**
081     * Arrow head line.
082     */
083    private final int[] headLine = new int[2];
084
085    /**
086     * Connect arrow to this label.
087     */
088    private final AbstractLabel to;
089
090    /**
091     * Create new arrow.
092     * @param from Connect from this label
093     * @param to Connect to this label
094     */
095    public Arrow(final AbstractLabel from, final AbstractLabel to) {
096
097        this.from = from;
098        this.to = to;
099
100    }
101
102    /**
103     * Get arrow origin.
104     * @return Arrow origin
105     */
106    public final AbstractLabel getFrom() {
107        return this.from;
108    }
109
110    /**
111     * Get arrow target.
112     * @return Arrow target
113     */
114    public final AbstractLabel getTo() {
115        return this.to;
116    }
117
118    @Override
119    public final void paintComponent(final Graphics g) {
120
121        ((Graphics2D) g).setRenderingHint(RenderingHints.KEY_ANTIALIASING,
122            RenderingHints.VALUE_ANTIALIAS_ON);
123
124        g.setColor(Color.BLACK);
125
126        final Dimension iconSize = Arrow.CONTROLLER.getIconSize().getSize();
127
128        final Point from = this.from.getLocation();
129        final Point to = this.to.getLocation();
130
131        if (this.from.getHeight() > iconSize.height) {
132
133            final Point fromLoc = this.from.getLocation();
134            final Dimension fromSize = this.from.getSize();
135
136            final int offset = (fromSize.width - iconSize.width) / 2;
137
138            from.setLocation(fromLoc.x + offset, fromLoc.y);
139
140        }
141
142        if (this.to.getHeight() > iconSize.height) {
143
144            final Point toLoc = this.to.getLocation();
145            final Dimension toSize = this.to.getSize();
146
147            final int offset = (toSize.width - iconSize.width) / 2;
148
149            to.setLocation(toLoc.x + offset, toLoc.y);
150
151        }
152
153        boolean fromAbove = false;
154        boolean fromLeft = false;
155        boolean fromBelow = false;
156        boolean fromRight = false;
157
158        if ((from.y + iconSize.height) < to.y) {
159            fromAbove = true;
160        }
161
162        if ((to.y + iconSize.height) < from.y) {
163            fromBelow = true;
164        }
165
166        if ((from.x + iconSize.width) < to.x) {
167            fromLeft = true;
168        }
169
170        if ((to.x + iconSize.width) < from.x) {
171            fromRight = true;
172        }
173
174        if (fromAbove) {
175
176            this.moveToSide(from, BoxSide.BOTTOM);
177            this.moveToSide(to, BoxSide.TOP);
178
179            // this.from.moveText(TextPosition.RIGHT);
180
181        } else if (fromBelow) {
182
183            this.moveToSide(from, BoxSide.TOP);
184            this.moveToSide(to, BoxSide.BOTTOM);
185
186            // this.to.moveText(TextPosition.RIGHT);
187
188        } else if (fromLeft) {
189
190            this.moveToSide(from, BoxSide.RIGHT);
191            this.moveToSide(to, BoxSide.LEFT);
192
193            this.from.moveText(TextPosition.BELOW);
194
195        } else if (fromRight) {
196
197            this.moveToSide(from, BoxSide.LEFT);
198            this.moveToSide(to, BoxSide.RIGHT);
199
200            this.to.moveText(TextPosition.BELOW);
201
202        }
203
204        g.drawLine(from.x, from.y, to.x, to.y);
205        g.fillPolygon(this.getArrow(from, to,
206            PetriConstants.ARROW_HEAD_SIZE));
207
208    }
209
210    /**
211     * Get polygon for painting the arrow.
212     * @param from Run arrow from this point
213     * @param to Run arrow to this point
214     * @param headsize Size of arrow head
215     * @return Arrow polygon
216     */
217    private Polygon getArrow(final Point from, final Point to,
218        final int headsize) {
219
220        this.resetArrowHeadLine(from, to, headsize);
221
222        final int[] crossBase = this.headLine;
223        final int[] headBase = this.headLine;
224
225        this.resetArrowHeadCrossLine(crossBase, to);
226
227        final Polygon head = new Polygon();
228
229        head.addPoint(headBase[0], headBase[1]);
230        head.addPoint(this.crossLine[0], this.crossLine[1]);
231        head.addPoint(to.x, to.y);
232        head.addPoint(this.crossLine[2], this.crossLine[3]);
233        head.addPoint(headBase[0], headBase[1]);
234        head.addPoint(from.x, from.y);
235
236        return head;
237    }
238
239    /**
240     * Move the point attaching the arrow to one side of the label.
241     * @param p Attaching point
242     * @param side Side to move to
243     */
244    private void moveToSide(final Point p, final BoxSide side) {
245
246        final Dimension iconSize = Arrow.CONTROLLER.getIconSize().getSize();
247
248        switch (side) {
249
250            case LEFT:
251                p.y += iconSize.height / 2;
252                break;
253            case TOP:
254                p.x += iconSize.width / 2;
255                break;
256            case BOTTOM:
257                p.x += iconSize.width / 2;
258                p.y += iconSize.height;
259                break;
260            case RIGHT:
261                p.x += iconSize.width;
262                p.y += iconSize.height / 2;
263                break;
264
265            default:
266                break;
267
268        }
269
270    }
271
272    /**
273     * Set arrow head cross line.
274     * @param crosslineBase Base to calculate cross against
275     * @param cross Point to cross at
276     */
277    private void resetArrowHeadCrossLine(final int[] crosslineBase,
278        final Point cross) {
279
280        final int x1 = crosslineBase[0];
281        final int x2 = crosslineBase[1];
282
283        final int crossX =
284            (int) (((cross.x - x1) * PetriConstants.ARROW_HEAD_FACTOR) + x1);
285        final int crossY =
286            (int) (((cross.y - x2) * PetriConstants.ARROW_HEAD_FACTOR) + x2);
287
288        this.crossLine[0] = (((x1 + x2) - crossY));
289        this.crossLine[1] = (((x2 + crossX) - x1));
290        this.crossLine[2] =
291            this.crossLine[0] + ((x1 - this.crossLine[0]) * 2);
292        this.crossLine[3] =
293            this.crossLine[1] + ((x2 - this.crossLine[1]) * 2);
294
295    }
296
297    /**
298     * Set head line for the arrow.
299     * @param from Point to connect from
300     * @param to Point to connect to
301     * @param distance Distance of head to end point
302     */
303    private void resetArrowHeadLine(final Point from, final Point to,
304        final int distance) {
305
306        final double stretch =
307            1 - (distance / (Math.sqrt(((to.x - from.x) * (to.x - from.x))
308                + ((to.y - from.y) * (to.y - from.y)))));
309
310        this.headLine[0] = (int) (stretch * (to.x - from.x)) + from.x;
311        this.headLine[1] = (int) (stretch * (to.y - from.y)) + from.y;
312
313    }
