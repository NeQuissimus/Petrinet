001package com.nequissimus.university.k1584.ui.elements;
002
003import java.awt.Color;
004import java.awt.Dimension;
005import java.awt.Graphics;
006import java.awt.Graphics2D;
007import java.awt.Point;
008import java.awt.Polygon;
009import java.awt.RenderingHints;
010
011import javax.swing.JPanel;
012
013import com.nequissimus.university.k1584.PetriController;
014import com.nequissimus.university.k1584.logic.PetriConfig;
015import com.nequissimus.university.k1584.ui.enums.BoxSide;
016
017/**
018 * Arrow connecting two Petri objects. The arrow is automatically aligned to the
019 * objects according to their relative position to each other. That way the
020 * arrow never crosses through the object itself. The arrow aligns on the top,
021 * bottom, left or right side of the label icon and is always centered on the
022 * respective side.
023 * @author Tim Steinbach
024 */
025public class Arrow extends JPanel {
026
027    /**
028     * Serializable UID.
029     */
030    private static final long serialVersionUID = 3151568700241791450L;
031
032    /**
033     * Configuration.
034     */
035    private static final PetriConfig CONFIG = PetriConfig.getInstance();
036
037    /**
038     * Connect arrow from this label.
039     */
040    private final AbstractLabel from;
041
042    /**
043     * Connect arrow to this label.
044     */
045    private final AbstractLabel to;
046
047    /**
048     * Arrow head line.
049     */
050    private final int[] headLine = new int[2];
051
052    /**
053     * Arrow cross line.
054     */
055    private final int[] crossLine = new int[4];
056
057    /**
058     * Create new arrow.
059     * @param from Connect from this label
060     * @param to Connect to this label
061     */
062    public Arrow(final AbstractLabel from, final AbstractLabel to) {
063
064        this.from = from;
065        this.to = to;
066
067    }
068
069    @Override
070    public final void paintComponent(final Graphics g) {
071        ((Graphics2D) g).setRenderingHint(RenderingHints.KEY_ANTIALIASING,
072            RenderingHints.VALUE_ANTIALIAS_ON);
073
074        g.setColor(Color.BLACK);
075
076        final Point from = this.from.getLocation();
077        final Point to = this.to.getLocation();
078
079        final Dimension iconSize =
080            PetriController.getInstance().getIconSize().getSize();
081
082        boolean fromAbove = false;
083        boolean fromLeft = false;
084        boolean fromBelow = false;
085        boolean fromRight = false;
086
087        if ((from.y + iconSize.height) < to.y) {
088            fromAbove = true;
089        }
090
091        if ((to.y + iconSize.height) < from.y) {
092            fromBelow = true;
093        }
094
095        if ((from.x + iconSize.width) < to.x) {
096            fromLeft = true;
097        }
098
099        if ((to.x + iconSize.width) < from.x) {
100            fromRight = true;
101        }
102
103        if (fromAbove) {
104            this.moveToSide(from, BoxSide.BOTTOM);
105            this.moveToSide(to, BoxSide.TOP);
106        } else if (fromBelow) {
107            this.moveToSide(from, BoxSide.TOP);
108            this.moveToSide(to, BoxSide.BOTTOM);
109        } else if (fromLeft) {
110            this.moveToSide(from, BoxSide.RIGHT);
111            this.moveToSide(to, BoxSide.LEFT);
112        } else if (fromRight) {
113            this.moveToSide(from, BoxSide.LEFT);
114            this.moveToSide(to, BoxSide.RIGHT);
115        }
116
117        g.drawLine(from.x, from.y, to.x, to.y);
118        g.fillPolygon(this.getArrow(from, to,
119            Arrow.CONFIG.getArrowHeadSize()));
120
121    }
122
123    /**
124     * Get polygon for painting the arrow.
125     * @param from Run arrow from this point
126     * @param to Run arrow to this point
127     * @param headsize Size of arrow head
128     * @return Arrow polygon
129     */
130    private Polygon getArrow(final Point from, final Point to,
131        final int headsize) {
132
133        this.resetArrowHeadLine(from, to, headsize);
134
135        final int[] crossBase = this.headLine;
136        final int[] headBase = this.headLine;
137
138        this.resetArrowHeadCrossLine(crossBase, to);
139
140        final Polygon head = new Polygon();
141
142        head.addPoint(headBase[0], headBase[1]);
143        head.addPoint(this.crossLine[0], this.crossLine[1]);
144        head.addPoint(to.x, to.y);
145        head.addPoint(this.crossLine[2], this.crossLine[3]);
146        head.addPoint(headBase[0], headBase[1]);
147        head.addPoint(from.x, from.y);
148
149        return head;
150    }
151
152    /**
153     * Set head line for the arrow.
154     * @param from Point to connect from
155     * @param to Point to connect to
156     * @param distance Distance of head to end point
157     */
158    private void resetArrowHeadLine(final Point from, final Point to,
159        final int distance) {
160
161        final double stretch =
162            1 - (distance / (Math.sqrt(((to.x - from.x) * (to.x - from.x))
163                + ((to.y - from.y) * (to.y - from.y)))));
164
165        this.headLine[0] = (int) (stretch * (to.x - from.x)) + from.x;
166        this.headLine[1] = (int) (stretch * (to.y - from.y)) + from.y;
167
168    }
169
170    /**
171     * Set arrow head cross line.
172     * @param crosslineBase Base to calculate cross against
173     * @param cross Point to cross at
174     */
175    private void resetArrowHeadCrossLine(final int[] crosslineBase,
176        final Point cross) {
177
178        final int x1 = crosslineBase[0];
179        final int x2 = crosslineBase[1];
180
181        final int crossX =
182            (int) (((cross.x - x1) * Arrow.CONFIG.getArrowHeadFactor()) + x1);
183        final int crossY =
184            (int) (((cross.y - x2) * Arrow.CONFIG.getArrowHeadFactor()) + x2);
185
186        this.crossLine[0] = (((x1 + x2) - crossY));
187        this.crossLine[1] = (((x2 + crossX) - x1));
188        this.crossLine[2] =
189            this.crossLine[0] + ((x1 - this.crossLine[0]) * 2);
190        this.crossLine[3] =
191            this.crossLine[1] + ((x2 - this.crossLine[1]) * 2);
192
193    }
194
195    /**
196     * Move the point attaching the arrow to one side of the label.
197     * @param p Attaching point
198     * @param side Side to move to
199     */
200    private void moveToSide(final Point p, final BoxSide side) {
201
202        final Dimension iconSize =
203            PetriController.getInstance().getIconSize().getSize();
204
205        switch (side) {
206
207            case LEFT:
208                p.y += iconSize.height / 2;
209                break;
210            case TOP:
211                p.x += iconSize.width / 2;
212                break;
213            case BOTTOM:
214                p.x += iconSize.width / 2;
215                p.y += iconSize.height;
216                break;
217            case RIGHT:
218                p.x += iconSize.width;
219                p.y += iconSize.height / 2;
220                break;
221
222            default:
223                break;
224
225        }
226
227    }
228
229    /**
230     * Get arrow origin.
231     * @return Arrow origin
232     */
233    public final AbstractLabel getFrom() {
234        return this.from;
235    }
236
237    /**
238     * Get arrow target.
239     * @return Arrow target
240     */
241    public final AbstractLabel getTo() {
242        return this.to;
243    }
244
245}
