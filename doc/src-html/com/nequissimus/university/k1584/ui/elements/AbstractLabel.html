001package com.nequissimus.university.k1584.ui.elements;
002
003import javax.swing.JLabel;
004
005import com.nequissimus.university.k1584.PetriController;
006import com.nequissimus.university.k1584.ui.enums.IconSize;
007import com.nequissimus.university.k1584.ui.enums.TextPosition;
008import com.nequissimus.university.k1584.ui.listener.CanvasIconMenuListener;
009import com.nequissimus.university.k1584.ui.listener.SidebarIconMenuListener;
010import com.nequissimus.university.k1584.ui.traits.Draggable;
011
012/**
013 * Abstract class defining essential properties for each label component.
014 * @author Tim Steinbach
015 */
016public abstract class AbstractLabel extends JLabel implements Draggable {
017
018    /**
019     * Serializable UID.
020     */
021    private static final long serialVersionUID = 6558385524618595255L;
022
023    /**
024     * Application controller.
025     */
026    private final PetriController controller;
027
028    /**
029     * Create a new UI label for a given logical component.
030     * @param name Label name
031     */
032    protected AbstractLabel(final String name) {
033
034        this();
035
036        this.setIcon(this.getPetriIcon(this.controller.getIconSize()));
037
038        this.setText(name);
039
040    }
041
042    /**
043     * Create a new UI label without it being associated to a logical component.
044     */
045    protected AbstractLabel() {
046
047        this.controller = PetriController.getInstance();
048
049        this.setIcon(this.getPetriIcon(IconSize.MEDIUM));
050        this.setText("");
051        this.setOpaque(false);
052
053        this.moveText(TextPosition.RIGHT);
054
055    }
056
057    /**
058     * Get the application controller.
059     * @return Application controller
060     */
061    final PetriController getController() {
062
063        return this.controller;
064
065    }
066
067    /**
068     * Move the text to a different position relative to its icon.
069     * @param pos Text position
070     */
071    final void moveText(final TextPosition pos) {
072
073        this.setHorizontalTextPosition(pos.getX());
074        this.setVerticalTextPosition(pos.getY());
075
076    }
077
078    /**
079     * Return the associated icon.
080     * @param size Icon size
081     * @return Icon
082     */
083    public abstract AbstractIcon getPetriIcon(final IconSize size);
084
085    @Override
086    public final void setText(final String text) {
087
088        super.setText(text);
089        this.setSize(this.getPreferredSize());
090
091    }
092
093    /**
094     * Resize label icon.
095     * @param size New size
096     */
097    public final void resizeIcon(final IconSize size) {
098
099        this.setIcon(this.getPetriIcon(size));
100        this.setSize(this.getPreferredSize());
101
102    }
103
104    /**
105     * Add the context menu for adding new elements to the canvas.
106     */
107    public final void addSidebarMenu() {
108
109        this.addMouseListener(new SidebarIconMenuListener(this));
110
111    }
112
113    /**
114     * Add the context menu for elements on the canvas.
115     */
116    public final void addCanvasMenu() {
117
118        this.addMouseListener(new CanvasIconMenuListener(this));
119
120    }
121
122}
