001package com.nequissimus.university.k1584.ui.elements;
002
003import java.awt.Color;
004import java.awt.Graphics;
005
006import javax.swing.JLabel;
007
008import com.nequissimus.university.k1584.PetriController;
009import com.nequissimus.university.k1584.logic.PetriConfig;
010import com.nequissimus.university.k1584.ui.enums.IconSize;
011import com.nequissimus.university.k1584.ui.enums.TextPosition;
012import com.nequissimus.university.k1584.ui.listener.SidebarIconMenuListener;
013import com.nequissimus.university.k1584.ui.traits.Draggable;
014
015/**
016 * Abstract class defining essential properties for each label component.
017 * @author Tim Steinbach
018 */
019public abstract class AbstractLabel extends JLabel implements Draggable {
020
021    /**
022     * Serializable UID.
023     */
024    private static final long serialVersionUID = 6558385524618595255L;
025
026    /**
027     * Configuration.
028     */
029    private static final PetriConfig CONFIG = PetriConfig.getInstance();
030
031    /**
032     * Controller.
033     */
034    private static final PetriController CONTROLLER = PetriController
035        .getInstance();
036
037    /**
038     * Create a new UI label without it being associated to a logical component.
039     */
040    protected AbstractLabel() {
041
042        this.setIcon(this.getPetriIcon(IconSize.MEDIUM));
043        this.setText("");
044        this.setOpaque(false);
045
046        this.moveText(TextPosition.RIGHT);
047
048    }
049
050    /**
051     * Create a new UI label for a given logical component.
052     * @param name Label name
053     */
054    protected AbstractLabel(final String name) {
055
056        this();
057
058        this.setIcon(this.getPetriIcon(AbstractLabel.CONTROLLER
059            .getIconSize()));
060
061        this.setText(name);
062
063        this.setBackground(AbstractLabel.CONFIG.getNormalColor());
064
065    }
066
067    /**
068     * Add the context menu for elements on the canvas.
069     */
070    public abstract void addCanvasMenu();
071
072    /**
073     * Add the context menu for adding new elements to the canvas.
074     */
075    public final void addSidebarMenu() {
076
077        this.addMouseListener(new SidebarIconMenuListener(this));
078
079    }
080
081    /**
082     * Return the associated icon.
083     * @param size Icon size
084     * @return Icon
085     */
086    public abstract AbstractIcon getPetriIcon(final IconSize size);
087
088    /**
089     * Resize label icon.
090     * @param size New size
091     */
092    public final void resizeIcon(final IconSize size) {
093
094        this.setIcon(this.getPetriIcon(size));
095        this.setSize(this.getPreferredSize());
096
097    }
098
099    @Override
100    public final void setText(final String text) {
101
102        super.setText(text);
103        this.setSize(this.getPreferredSize());
104
105    }
106
107    @Override
108    protected final void paintComponent(final Graphics g) {
109        // Override to draw background
110
111        final Color background = this.getBackground();
112
113        // If alpha == 0, background is invisible (no need to paint)
114        if (background.getAlpha() > 0) {
115
116            g.setColor(background);
117            g.fillRoundRect(0, 0, this.getWidth(), this.getHeight(), 15, 15);
118
119        }
120
121        super.paintComponent(g);
122
123    }
124
125    /**
126     * Move the text to a different position relative to its icon.
127     * @param pos Text position
128     */
129    final void moveText(final TextPosition pos) {
130
131        this.setHorizontalTextPosition(pos.getX());
132        this.setVerticalTextPosition(pos.getY());
133
134    }
135
136}
