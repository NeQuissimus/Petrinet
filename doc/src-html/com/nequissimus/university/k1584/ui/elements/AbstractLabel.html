032package com.nequissimus.university.k1584.ui.elements;
033
034import java.awt.Color;
035import java.awt.Dimension;
036import java.awt.FontMetrics;
037import java.awt.Graphics;
038import java.awt.Graphics2D;
039import java.awt.RenderingHints;
040
041import javax.swing.JLabel;
042import javax.swing.SwingConstants;
043
044import com.nequissimus.university.k1584.PetriController;
045import com.nequissimus.university.k1584.logic.PetriConstants;
046import com.nequissimus.university.k1584.ui.enums.IconSize;
047import com.nequissimus.university.k1584.ui.enums.TextPosition;
048import com.nequissimus.university.k1584.ui.listener.SidebarIconMenuListener;
049import com.nequissimus.university.k1584.ui.traits.Draggable;
050
051/**
052 * Abstract class defining essential properties for each label component.
053 * @author Tim Steinbach
054 */
055public abstract class AbstractLabel extends JLabel implements Draggable {
056
057    /**
058     * Horizontal padding between text and icon.
059     */
060    private static final int PADDING_TEXT_ICON_HOR = 20;
061
062    /**
063     * Vertical padding between text and icon.
064     */
065    private static final int PADDING_TEXT_ICON_VER = 10;
066
067    /**
068     * Controller.
069     */
070    private static final PetriController CONTROLLER = PetriController
071        .getInstance();
072
073    /**
074     * Serializable UID.
075     */
076    private static final long serialVersionUID = 6558385524618595255L;
077
078    /**
079     * Font metrics.
080     */
081    private FontMetrics metrics = null;
082
083    /**
084     * Text position.
085     */
086    private TextPosition textPosition = TextPosition.BELOW;
087
088    /**
089     * Create a new UI label without it being associated to a logical component.
090     */
091    protected AbstractLabel() {
092
093        this.setIcon(this.getPetriIcon(IconSize.MEDIUM));
094        this.setText("");
095        this.setOpaque(false);
096
097        this.setVerticalTextPosition(this.textPosition.getY());
098        this.setHorizontalTextPosition(this.textPosition.getX());
099
100    }
101
102    /**
103     * Create a new UI label for a given logical component.
104     * @param name Label name
105     */
106    protected AbstractLabel(final String name) {
107
108        this();
109
110        this.setIcon(this.getPetriIcon(AbstractLabel.CONTROLLER
111            .getIconSize()));
112
113        this.setText(name);
114
115        this.setBackground(PetriConstants.COLOR_NORMAL);
116
117    }
118
119    /**
120     * Add the context menu for elements on the canvas.
121     */
122    public abstract void addCanvasMenu();
123
124    /**
125     * Add the context menu for adding new elements to the canvas.
126     */
127    public final void addSidebarMenu() {
128
129        this.addMouseListener(new SidebarIconMenuListener(this));
130
131    }
132
133    /**
134     * Return the associated icon.
135     * @param size Icon size
136     * @return Icon
137     */
138    public abstract AbstractIcon getPetriIcon(final IconSize size);
139
140    /**
141     * Resize label icon.
142     * @param size New size
143     */
144    public final void resizeIcon(final IconSize size) {
145
146        this.setIcon(this.getPetriIcon(size));
147        this.setSize(this.getPreferredSize());
148
149    }
150
151    @Override
152    public final void setText(final String text) {
153
154        super.setText(text);
155        this.setSize(this.getPreferredSize());
156
157        AbstractLabel.CONTROLLER.redrawCanvas();
158
159    }
160
161    @Override
162    protected final void paintComponent(final Graphics g) {
163        // Override to draw background
164
165        ((Graphics2D) g).setRenderingHint(RenderingHints.KEY_ANTIALIASING,
166            RenderingHints.VALUE_ANTIALIAS_ON);
167
168        this.metrics = g.getFontMetrics();
169
170        final Color background = this.getBackground();
171
172        // If alpha == 0, background is invisible (no need to paint)
173        if (background.getAlpha() > 0) {
174
175            g.setColor(background);
176            g.fillRoundRect(0, 0, this.getWidth(), this.getHeight(), 15, 15);
177
178        }
179
180        super.paintComponent(g);
181
182    }
183
184    /**
185     * Move the text to a different position relative to its icon.
186     * @param pos Text position
187     */
188    final void moveText(final TextPosition pos) {
189
190        if (this.textPosition != pos) {
191
192            this.setVerticalTextPosition(pos.getY());
193            this.setHorizontalTextPosition(pos.getX());
194            this.setHorizontalAlignment(SwingConstants.LEFT);
195
196            // Resize label to fit text and icon
197            final IconSize iconSize =
198                AbstractLabel.CONTROLLER.getIconSize();
199            final Dimension size = new Dimension(iconSize.getSize());
200
201            if (null != this.metrics) {
202
203                this.textPosition = pos;
204
205                final int fontSize = this.metrics.getFont().getSize();
206                final int textWidth =
207                    this.metrics.stringWidth(this.getText());
208
209                if (pos == TextPosition.BELOW) {
210
211                    int width = (int) size.getWidth();
212                    width = (width > textWidth) ? width : textWidth;
213
214                    final int height =
215                        (int) (size.getHeight()) + fontSize
216                            + AbstractLabel.PADDING_TEXT_ICON_VER;
217
218                    size.setSize(width, height);
219
220                } else if (pos == TextPosition.RIGHT) {
221
222                    size.setSize(size.getWidth() + textWidth
223                        + AbstractLabel.PADDING_TEXT_ICON_HOR,
224                        size.getHeight());
225
226                }
227
228                this.setSize(size);
229                this.setPreferredSize(size);
230
231            }
232
233        }
234
235    }
236
237}
