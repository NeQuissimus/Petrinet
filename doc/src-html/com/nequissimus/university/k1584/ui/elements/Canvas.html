001package com.nequissimus.university.k1584.ui.elements;
002
003import java.awt.Color;
004import java.awt.Component;
005import java.awt.Dimension;
006import java.util.HashSet;
007import java.util.Set;
008
009import javax.swing.JPanel;
010import javax.swing.JScrollPane;
011
012import com.nequissimus.university.k1584.logic.PetriConfig;
013import com.nequissimus.university.k1584.ui.listener.ScrollListener;
014
015/**
016 * This is the main canvas on which all the editing will take place. It consists
017 * of this panel that is scrollable and comes with a predefined size. Its size
018 * can be changed to allow for a larger editing canvas.
019 * @see PetriConfig
020 * @author Tim Steinbach
021 */
022public class Canvas extends JScrollPane {
023
024    /**
025     * Serializable UID.
026     */
027    private static final long serialVersionUID = 6158181636171175402L;
028
029    /**
030     * Configuration.
031     */
032    private static final PetriConfig CONFIG = PetriConfig.getInstance();
033
034    /**
035     * Policy for horizontal scroll bars.
036     */
037    private static final int SCROLL_H =
038        JScrollPane.HORIZONTAL_SCROLLBAR_ALWAYS;
039
040    /**
041     * Policy for vertical scroll bars.
042     */
043    private static final int SCROLL_V =
044        JScrollPane.VERTICAL_SCROLLBAR_ALWAYS;
045
046    /**
047     * Petri net canvas.
048     */
049    private final JPanel canvas;
050
051    /**
052     * Create a new canvas instance.
053     */
054    Canvas() {
055
056        super();
057
058        this.canvas = new JPanel();
059
060        this.resetCanvas();
061
062        this.canvas.validate();
063        this.canvas.repaint();
064
065        final int visibleWidth =
066            Canvas.CONFIG.getWindowWidth()
067                - Canvas.CONFIG.getSidebarWidth();
068        final int visibleHeight =
069            Canvas.CONFIG.getWindowHeight()
070                - Canvas.CONFIG.getScrollbarHeight();
071        final Dimension visibleSize =
072            new Dimension(visibleWidth, visibleHeight);
073
074        this.setPreferredSize(visibleSize);
075        this.setSize(visibleSize);
076
077        this.setViewportView(this.canvas);
078
079        this.setHorizontalScrollBarPolicy(Canvas.SCROLL_H);
080        this.setVerticalScrollBarPolicy(Canvas.SCROLL_V);
081
082        this.getVerticalScrollBar().addAdjustmentListener(
083            new ScrollListener());
084        this.getHorizontalScrollBar().addAdjustmentListener(
085            new ScrollListener());
086
087    }
088
089    /**
090     * Add a UI label to the canvas.
091     * @param label Label to be added
092     */
093    public final void add(final AbstractLabel label) {
094
095        this.canvas.add(label);
096        this.canvas.validate();
097        this.canvas.repaint();
098
099    }
100
101    /**
102     * Get canvas.
103     * @return Canvas
104     */
105    public final JPanel getCanvas() {
106
107        return this.canvas;
108
109    }
110
111    /**
112     * Get all labels currently displayed on the canvas.
113     * @return Set of labels
114     */
115    public final Set<AbstractLabel> getLabels() {
116
117        final Component[] components = this.canvas.getComponents();
118
119        final Set<AbstractLabel> labels = new HashSet<AbstractLabel>();
120
121        for (final Component component : components) {
122
123            if (component instanceof AbstractLabel) {
124                labels.add((AbstractLabel) component);
125            }
126
127        }
128
129        return labels;
130
131    }
132
133    @Override
134    public final void repaint() {
135        this.revalidate();
136        super.repaint();
137
138        if (null != this.canvas) {
139            this.canvas.revalidate();
140            this.canvas.repaint();
141        }
142
143    }
144
145    /**
146     * Reset the canvas to its defaults.
147     */
148    private void resetCanvas() {
149
150        this.canvas.setBackground(Color.WHITE);
151        this.canvas.setOpaque(true);
152
153        final int canvasWidth = Canvas.CONFIG.getCanvasWidth();
154        final int canvasHeight = Canvas.CONFIG.getCanvasHeight();
155
156        final Dimension canvasSize =
157            new Dimension(canvasWidth, canvasHeight);
158
159        this.canvas.setPreferredSize(canvasSize);
160        this.canvas.setSize(canvasSize);
161
162        this.canvas.setLayout(null);
163
164    }
165
166}
