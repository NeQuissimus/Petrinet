001package com.nequissimus.university.k1584.ui.elements;
002
003import javax.swing.JFrame;
004import javax.swing.JMenu;
005import javax.swing.JMenuBar;
006import javax.swing.JMenuItem;
007
008import com.nequissimus.university.k1584.ui.actions.ChangeIconSizeAction;
009import com.nequissimus.university.k1584.ui.actions.CloseWindowAction;
010import com.nequissimus.university.k1584.ui.actions.SaveFileAction;
011import com.nequissimus.university.k1584.ui.enums.IconSize;
012import com.nequissimus.university.k1584.util.StringUtils;
013
014/**
015 * Menu bar for the application window. This menu bar automatically provides all
016 * essential options, such as "save", "close", "load" etc.
017 * @author Tim Steinbach
018 */
019public class PetriMenuBar extends JMenuBar {
020
021    /**
022     * Serializable UID.
023     */
024    private static final long serialVersionUID = -415488481991662662L;
025
026    /**
027     * Application window.
028     */
029    private final JFrame window;
030
031    /**
032     * Create a new menu bar instance for a given window.
033     * @param window Application window
034     */
035    public PetriMenuBar(final JFrame window) {
036
037        super();
038
039        this.window = window;
040
041        this.resetMenu();
042
043    }
044
045    /**
046     * Add all essential menu items to the menu bar.
047     */
048    public final void resetMenu() {
049
050        this.resetFileMenu();
051        this.resetViewMenu();
052
053    }
054
055    /**
056     * Add all items found under "file".
057     */
058    private void resetFileMenu() {
059
060        final JMenu menu = new JMenu("File");
061
062        JMenuItem item = new JMenuItem("Save");
063        item.addActionListener(new SaveFileAction());
064        menu.add(item);
065
066        item = new JMenuItem("Close");
067        item.addActionListener(new CloseWindowAction(this.window));
068        menu.add(item);
069
070        this.add(menu);
071
072    }
073
074    /**
075     * Add all items found under "view".
076     */
077    private void resetViewMenu() {
078
079        final JMenu menu = new JMenu("View");
080
081        JMenu subMenu = new JMenu("Icon size");
082
083        for (IconSize size : IconSize.values()) {
084
085            final JMenuItem item =
086                new JMenuItem(StringUtils.asName(size.name()));
087
088            item.addActionListener(new ChangeIconSizeAction(size));
089
090            subMenu.add(item);
091
092        }
093
094        menu.add(subMenu);
095
096        this.add(menu);
097
098    }
099
100}
