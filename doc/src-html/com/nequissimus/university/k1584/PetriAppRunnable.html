001package com.nequissimus.university.k1584;
002
003import java.awt.EventQueue;
004
005import com.nequissimus.university.k1584.logic.PetriConfig;
006import com.nequissimus.university.k1584.ui.PetriWindow;
007
008
009/**
010 * This is the application launcher which causes the editor to be created and started.
011 * @author Tim Schram
012 *
013 */
014public class PetriApp {
015
016    public static void main(String[] args) {
017        
018        final PetriConfig config = PetriConfig.getInstance();
019        
020        // This is needed for Mac OS X only - It changes the application name in the menu bar
021        // from the fully-qualified class name to the given value
022        System.setProperty("com.apple.mrj.application.apple.menu.about.name", config.get(PetriConfig.APPLICATION_NAME));
023            
024        EventQueue.invokeLater(new PetriAppRunnable());
025
026    }
027
028}
029
030class PetriAppRunnable implements Runnable {
031
032    final PetriWindow window;
033    
034    public PetriAppRunnable() {
035        
036        this.window = new PetriWindow();
037        
038    }
039
040    @Override
041    public void run() {
042        
043        window.validate();
044        window.repaint();
045        window.setVisible(true);
046
047    }
048
049}
