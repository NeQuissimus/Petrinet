001package com.nequissimus.university.k1584;
002
003import java.awt.Component;
004import java.awt.Dimension;
005import java.awt.Point;
006import java.io.File;
007
008import javax.print.attribute.standard.Severity;
009import javax.swing.JOptionPane;
010
011import com.nequissimus.university.k1584.logic.PetriConfig;
012import com.nequissimus.university.k1584.logic.PetriNet;
013import com.nequissimus.university.k1584.logic.PetriObject;
014import com.nequissimus.university.k1584.logic.PetriPlace;
015import com.nequissimus.university.k1584.logic.PetriSnapshots;
016import com.nequissimus.university.k1584.logic.PetriTransition;
017import com.nequissimus.university.k1584.logic.pnml.PetriMarkup;
018import com.nequissimus.university.k1584.logic.pnml.PnmlException;
019import com.nequissimus.university.k1584.ui.AbstractLabel;
020import com.nequissimus.university.k1584.ui.PetriCanvas;
021import com.nequissimus.university.k1584.ui.PetriPlaceLabel;
022import com.nequissimus.university.k1584.ui.PetriTransitionLabel;
023import com.nequissimus.university.k1584.ui.PetriWindow;
024import com.nequissimus.university.k1584.ui.enums.IconSize;
025
026/**
027 * Main controller for the Petri net application
028 * that is implemented as a singleton to make sure
029 * no application has more than one controller/UI/logic.
030 * @author Tim Steinbach
031 *
032 */
033public final class PetriController implements Runnable {
034
035    /**
036     * Singleton instance of controller.
037     */
038    private static final PetriController CONTROLLER =
039        new PetriController();
040
041    /**
042     * Graphical user interface.
043     */
044    private PetriWindow ui;
045
046    /**
047     * Snapshots of all logical Petri nets.
048     */
049    private final PetriSnapshots logic;
050
051    /**
052     * Configuration.
053     */
054    private final PetriConfig config = PetriConfig.getInstance();
055
056    /**
057     * The Petri net that is currently being edited.
058     */
059    private PetriNet currentNet = null;
060
061    /**
062     * Size for all icons.
063     */
064    private IconSize iconSize = IconSize.LARGE;
065
066    /**
067     * Instantiate the controller.
068     */
069    private PetriController() {
070
071        this.logic = new PetriSnapshots();
072
073        this.logic.add(this.config.getNetName());
074        this.currentNet = this.logic.getCurrent();
075
076    }
077
078    /**
079     * Get the singleton instance of PetriController.
080     * @return Controller instance
081     */
082    public static PetriController getInstance() {
083        return PetriController.CONTROLLER;
084    }
085
086    @Override
087    public void run() {
088
089        Component ui = this.getWindow();
090
091        ui.validate();
092        ui.repaint();
093        ui.setVisible(true);
094
095    }
096
097    /**
098     * Get the current size for all icons.
099     * @return Current size
100     */
101    public Dimension getIconSize() {
102        return this.iconSize.getSize();
103    }
104
105    /**
106     * Set the new size for all icons.
107     * @param size New size
108     */
109    public void setIconSize(final IconSize size) {
110        this.iconSize = size;
111    }
112
113    /**
114     * Create a new place object and add it to both, the canvas and logical net.
115     */
116    public void addPlace() {
117
118        PetriPlace place = this.currentNet.addPlace();
119
120        PetriPlaceLabel label = new PetriPlaceLabel(place);
121
122        addLabel(label);
123
124    }
125
126    /**
127     * Create a new transition object and add it to both, the canvas and logical
128     * net.
129     */
130    public void addTransition() {
131
132        PetriTransition transition = this.currentNet.addTransition();
133
134        PetriTransitionLabel label = new PetriTransitionLabel(transition);
135
136        addLabel(label);
137
138    }
139
140    /**
141     * Add a new label to the canvas.
142     * @param label Label to be added
143     */
144    private void addLabel(final AbstractLabel label) {
145
146        PetriCanvas canvas = PetriWindow.getCanvas();
147
148        label.setLocation(0, 0);
149        canvas.add(label);
150
151    }
152
153    /**
154     * Get a Petri object's name.
155     * @param object Object
156     * @return Object's name
157     */
158    public String getName(final PetriObject object) {
159
160        return this.currentNet.getName(object);
161
162    }
163
164    /**
165     * Resize a Petri object.
166     * @param object Object to be resized
167     * @param newSize New size
168     */
169    public void setSize(final PetriObject object, final Dimension newSize) {
170
171        this.currentNet.setSize(object, newSize);
172
173    }
174
175    /**
176     * Reposition a Petri object.
177     * @param object Object to be moved
178     * @param position New position
179     */
180    public void
181        setPosition(final PetriObject object, final Point position) {
182
183        this.currentNet.setPosition(object, position);
184
185    }
186
187    /**
188     * Save the current Petri net to a given file.
189     * @param file File to save to
190     * @throws PnmlException Error turning nets into PNML
191     */
192    public void save(final File file) throws PnmlException {
193
194        PetriMarkup.toPnmlFile(file, this.logic);
195
196    }
197
198    /**
199     * Lazily initialize UI singleton and return it.
200     * @return PetriWindow singleton
201     */
202    public PetriWindow getWindow() {
203
204        if (null == this.ui) {
205
206            this.ui = new PetriWindow();
207
208        }
209
210        return this.ui;
211
212    }
213
214    /**
215     * Report a message to the UI/user.
216     * @param severity Severity of message
217     * @param message Message text
218     */
219    public void
220        reportMessage(final Severity severity, final String message) {
221
222        int messageType = JOptionPane.INFORMATION_MESSAGE;
223
224        if (severity.equals(Severity.ERROR)) {
225            messageType = JOptionPane.ERROR_MESSAGE;
226        } else if (severity.equals(Severity.WARNING)) {
227            messageType = JOptionPane.WARNING_MESSAGE;
228        }
229
230        JOptionPane.showMessageDialog(this.ui, message, null, messageType);
231
232    }
233
234}
