001package com.nequissimus.university.k1584;
002
003import java.awt.Component;
004import java.awt.Dimension;
005import java.awt.Point;
006import java.io.File;
007import java.util.Set;
008
009import javax.print.attribute.standard.Severity;
010import javax.swing.JOptionPane;
011import javax.swing.JPanel;
012
013import com.nequissimus.university.k1584.logic.PetriConfig;
014import com.nequissimus.university.k1584.logic.PetriNet;
015import com.nequissimus.university.k1584.logic.PetriObject;
016import com.nequissimus.university.k1584.logic.PetriPlace;
017import com.nequissimus.university.k1584.logic.PetriSnapshots;
018import com.nequissimus.university.k1584.logic.PetriTransition;
019import com.nequissimus.university.k1584.logic.pnml.PetriMarkup;
020import com.nequissimus.university.k1584.logic.pnml.PnmlException;
021import com.nequissimus.university.k1584.ui.elements.AbstractLabel;
022import com.nequissimus.university.k1584.ui.elements.PetriCanvas;
023import com.nequissimus.university.k1584.ui.elements.PetriPlaceLabel;
024import com.nequissimus.university.k1584.ui.elements.PetriTransitionLabel;
025import com.nequissimus.university.k1584.ui.elements.PetriWindow;
026import com.nequissimus.university.k1584.ui.enums.IconSize;
027
028/**
029 * Main controller for the Petri net application that is implemented as a
030 * singleton to make sure no application has more than one controller/UI/logic.
031 * @author Tim Steinbach
032 */
033public final class PetriController implements Runnable {
034
035    /**
036     * Singleton instance of controller.
037     */
038    private static final PetriController CONTROLLER = new PetriController();
039
040    /**
041     * Graphical user interface.
042     */
043    private PetriWindow ui;
044
045    /**
046     * Snapshots of all logical Petri nets.
047     */
048    private final PetriSnapshots logic;
049
050    /**
051     * Configuration.
052     */
053    private final PetriConfig config = PetriConfig.getInstance();
054
055    /**
056     * The Petri net that is currently being edited.
057     */
058    private PetriNet currentNet = null;
059
060    /**
061     * Size for all icons.
062     */
063    private IconSize iconSize = IconSize.LARGE;
064
065    /**
066     * Instantiate the controller.
067     */
068    private PetriController() {
069
070        this.logic = new PetriSnapshots();
071
072        this.logic.add(this.config.getNetName());
073        this.currentNet = this.logic.getCurrent();
074
075    }
076
077    /**
078     * Get the singleton instance of PetriController.
079     * @return Controller instance
080     */
081    public static PetriController getInstance() {
082        return PetriController.CONTROLLER;
083    }
084
085    @Override
086    public void run() {
087
088        Component ui = this.getWindow();
089
090        ui.validate();
091        ui.repaint();
092        ui.setVisible(true);
093
094    }
095
096    /**
097     * Get the current size for all icons.
098     * @return Current size
099     */
100    public Dimension getIconSize() {
101        return this.iconSize.getSize();
102    }
103
104    /**
105     * Set the new size for all icons.
106     * @param size New size
107     */
108    public void setIconSize(final IconSize size) {
109
110        if (size != this.iconSize) {
111
112            this.iconSize = size;
113
114            Set<AbstractLabel> labels = PetriWindow.getCanvas().getLabels();
115
116            for (AbstractLabel label : labels) {
117                label.resizeIcon(size);
118            }
119
120        }
121
122    }
123
124    /**
125     * Create a new place object and add it to both, the canvas and logical net.
126     */
127    public void addPlace() {
128
129        PetriPlace place = this.currentNet.addPlace();
130
131        PetriPlaceLabel label = new PetriPlaceLabel(place);
132
133        addLabel(label);
134
135    }
136
137    /**
138     * Create a new transition object and add it to both, the canvas and logical
139     * net.
140     */
141    public void addTransition() {
142
143        PetriTransition transition = this.currentNet.addTransition();
144
145        PetriTransitionLabel label = new PetriTransitionLabel(transition);
146
147        addLabel(label);
148
149    }
150
151    /**
152     * Add a new label to the canvas.
153     * @param label Label to be added
154     */
155    private void addLabel(final AbstractLabel label) {
156
157        PetriCanvas canvas = PetriWindow.getCanvas();
158
159        label.setLocation(0, 0);
160        canvas.add(label);
161
162    }
163
164    /**
165     * Get a Petri object's name.
166     * @param object Object
167     * @return Object's name
168     */
169    public String getName(final PetriObject object) {
170
171        return this.currentNet.getName(object);
172
173    }
174
175    /**
176     * Resize a Petri object.
177     * @param object Object to be resized
178     * @param newSize New size
179     */
180    public void setSize(final PetriObject object, final Dimension newSize) {
181
182        this.currentNet.setSize(object, newSize);
183
184    }
185
186    /**
187     * Reposition a Petri object.
188     * @param object Object to be moved
189     * @param position New position
190     */
191    public void setPosition(final PetriObject object, final Point position) {
192
193        this.currentNet.setPosition(object, position);
194
195    }
196
197    /**
198     * Save the current Petri net to a given file.
199     * @param file File to save to
200     * @throws PnmlException Error turning nets into PNML
201     */
202    public void save(final File file) throws PnmlException {
203
204        PetriMarkup.toPnmlFile(file, this.logic);
205
206    }
207
208    /**
209     * Lazily initialize UI singleton and return it.
210     * @return PetriWindow singleton
211     */
212    public PetriWindow getWindow() {
213
214        if (null == this.ui) {
215
216            this.ui = new PetriWindow();
217
218        }
219
220        return this.ui;
221
222    }
223
224    /**
225     * Report a message to the UI/user.
226     * @param severity Severity of message
227     * @param message Message text
228     */
229    public void
230        reportMessage(final Severity severity, final String message) {
231
232        int messageType = JOptionPane.INFORMATION_MESSAGE;
233
234        if (severity.equals(Severity.ERROR)) {
235            messageType = JOptionPane.ERROR_MESSAGE;
236        } else if (severity.equals(Severity.WARNING)) {
237            messageType = JOptionPane.WARNING_MESSAGE;
238        }
239
240        JOptionPane.showMessageDialog(this.ui, message, null, messageType);
241
242    }
243
244    /**
245     * Remove an object and its label from the logical net and the canvas,
246     * respectively.
247     * @param label Label/object to be removed
248     */
249    public void removeObject(final AbstractLabel label) {
250
251        PetriObject object = label.getObject();
252
253        if (object instanceof PetriPlace) {
254            this.currentNet.remove((PetriPlace) object);
255        } else if (object instanceof PetriTransition) {
256            this.currentNet.remove((PetriTransition) object);
257        }
258
259        final JPanel canvas = PetriWindow.getCanvas().getCanvas();
260
261        canvas.remove(label);
262        canvas.invalidate();
263        canvas.repaint();
264
265    }
266
267}
